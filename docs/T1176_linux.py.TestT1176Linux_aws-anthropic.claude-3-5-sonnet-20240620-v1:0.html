   let timeData = {};
   let currentUrl = '';
   let startTime = 0;

   function updateTime() {
     if (currentUrl && startTime) {
       const currentTime = Date.now();
       const elapsedTime = currentTime - startTime;
       timeData[currentUrl] = (timeData[currentUrl] || 0) + elapsedTime;
       startTime = currentTime;
       browser.storage.local.set({ timeData: timeData });
     }
   }

   function handleTabChange(tabId, changeInfo, tab) {
     if (changeInfo.status === 'complete') {
       updateTime();
       currentUrl = new URL(tab.url).hostname;
       startTime = Date.now();
     }
   }

   function handleTabActivated(activeInfo) {
     browser.tabs.get(activeInfo.tabId, (tab) => {
       updateTime();
       currentUrl = new URL(tab.url).hostname;
       startTime = Date.now();
     });
   }

   browser.tabs.onUpdated.addListener(handleTabChange);
   browser.tabs.onActivated.addListener(handleTabActivated);

   // Update time data every minute
   setInterval(updateTime, 60000);

   // Load existing time data on startup
   browser.storage.local.get('timeData', (result) => {
     if (result.timeData) {
       timeData = result.timeData;
     }
   });

   let activeTabId = null;

   function updateTime() {
     if (activeTabId && startTime) {
       const currentTime = new Date().getTime();
       const elapsedTime = Math.round((currentTime - startTime) / 1000);
       
       browser.tabs.get(activeTabId).then((tab) => {
         const url = new URL(tab.url);
         const domain = url.hostname;
         
         if (!timeData[domain]) {
           timeData[domain] = 0;

         timeData[domain] += elapsedTime;
         
         browser.storage.local.set({ timeData: timeData });

   let currentTab = null;
   let startTime = null;

   function updateTimeSpent(url, duration) {
     browser.storage.local.get(url, (result) => {
       let timeSpent = result[url] || 0;
       timeSpent += duration;
       browser.storage.local.set({ [url]: timeSpent });
     });
   }

   function handleTabChange(tabId, changeInfo, tab) {
     if (changeInfo.status === 'complete' && tab.active) {
       if (currentTab) {
         const endTime = Date.now();
         const duration = (endTime - startTime) / 1000; // Convert to seconds
         updateTimeSpent(currentTab, duration);
       }
       currentTab = new URL(tab.url).hostname;
       startTime = Date.now();
     }
   }

   browser.tabs.onUpdated.addListener(handleTabChange);

   browser.tabs.onActivated.addListener((activeInfo) => {
     browser.tabs.get(activeInfo.tabId, (tab) => {
       if (currentTab) {
         const endTime = Date.now();
         const duration = (endTime - startTime) / 1000; // Convert to seconds
         updateTimeSpent(currentTab, duration);
       }
       currentTab = new URL(tab.url).hostname;
       startTime = Date.now();
     });
   });

   function formatTime(seconds) {
     const hours = Math.floor(seconds / 3600);
     const minutes = Math.floor((seconds % 3600) / 60);
     const remainingSeconds = Math.floor(seconds % 60);
     return `${hours}h ${minutes}m ${remainingSeconds}s`;
   }

   function displayTimeSpent() {
     const timeList = document.getElementById('timeList');
     timeList.innerHTML = '';

     browser.storage.local.get(null, (result) => {
       const sortedEntries = Object.entries(result).sort((a, b) => b[1] - a[1]);
       
       sortedEntries.forEach(([url, time]) => {
         const listItem = document.createElement('div');
         listItem.className = 'time-entry';
         listItem.textContent = `${url}: ${formatTime(time)}`;
         timeList.appendChild(listItem);
       });
     });
   }

   document.addEventListener('DOMContentLoaded', displayTimeSpent);
