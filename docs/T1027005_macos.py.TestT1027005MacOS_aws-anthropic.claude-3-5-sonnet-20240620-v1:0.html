import subprocess
import argparse
import os

def get_log_entries(process_name=None, pid=None):
    cmd = ["log", "show"]
    
    if process_name:
        cmd.extend(["--predicate", f'processImagePath CONTAINS[c] "{process_name}"'])
    elif pid:
        cmd.extend(["--predicate", f'processID == {pid}'])
    else:
        raise ValueError("Either process_name or pid must be provided")

    result = subprocess.run(cmd, capture_output=True, text=True)
    return result.stdout

def delete_log_entries(entries):
    lines = entries.split('\n')
    for line in lines:
        if line.strip():
            timestamp = line.split()[0]
            cmd = ["log", "erase", "--oldest", timestamp]
            subprocess.run(cmd, check=True)

def main():
    parser = argparse.ArgumentParser(description="Delete macOS log entries for a specific process or PID")
    group = parser.add_mutually_exclusive_group(required=True)
    group.add_argument("-n", "--name", help="Process name to search for")
    group.add_argument("-p", "--pid", type=int, help="Process ID to search for")
    args = parser.parse_args()

    try:
        if args.name:
            entries = get_log_entries(process_name=args.name)
        else:
            entries = get_log_entries(pid=args.pid)

        if entries:
            print(f"Found the following log entries:")
            print(entries)
            
            confirm = input("Do you want to delete these entries? (y/n): ").lower()
            if confirm == 'y':
                delete_log_entries(entries)
                print("Log entries deleted successfully.")
            else:
                print("Operation cancelled.")
        else:
            print("No matching log entries found.")

    except subprocess.CalledProcessError as e:
        print(f"Error: {e}")
        print("This script may require administrative privileges.")

if __name__ == "__main__":
    main()

import subprocess
import argparse
import re

def get_log_entries(process_identifier, identifier_type):
    """
    Retrieve log entries for the specified process identifier.
    """
    cmd = ['log', 'show', '--predicate', f'{identifier_type} == "{process_identifier}"', '--style', 'syslog']
    result = subprocess.run(cmd, capture_output=True, text=True)
    return result.stdout.splitlines()

def delete_log_entries(entries):
    """
    Delete the specified log entries.
    """
    for entry in entries:
        # Extract the timestamp from the log entry
        timestamp_match = re.search(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})', entry)
        if timestamp_match:
            timestamp = timestamp_match.group(1)
            # Delete the log entry
            cmd = ['log', 'erase', '--oldest', timestamp]
            subprocess.run(cmd, check=True)
            print(f"Deleted entry: {entry}")
        else:
            print(f"Could not parse timestamp for entry: {entry}")

def main():
    parser = argparse.ArgumentParser(description="Delete macOS log entries for a specific process.")
    parser.add_argument("identifier", help="Process name or PID")
    parser.add_argument("--type", choices=["processImagePath", "processID"], default="processImagePath",
                        help="Identifier type: processImagePath for process name, processID for PID")
    args = parser.parse_args()

    print(f"Searching for log entries related to {args.type} '{args.identifier}'...")
    entries = get_log_entries(args.identifier, args.type)

    if not entries:
        print("No matching log entries found.")
        return

    print(f"Found {len(entries)} matching log entries.")
    confirm = input("Do you want to delete these entries? (y/n): ").lower()
    if confirm == 'y':
        delete_log_entries(entries)
        print("Log entries deleted.")
    else:
        print("Operation cancelled.")

if __name__ == "__main__":
    main()

import subprocess

import sys

def search_and_delete_log_entries(process_identifier, is_pid=False):
    # Construct the log show command
    if is_pid:
        log_command = f"log show --predicate 'processID == {process_identifier}'"
    else:
        log_command = f"log show --predicate 'processImagePath contains \"{process_identifier}\"'"

    try:
        # Run the log show command and capture the output
        result = subprocess.run(log_command, shell=True, check=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)
        log_entries = result.stdout.strip().split('\n')

        if not log_entries:
            print(f"No log entries found for the specified {'PID' if is_pid else 'process name'}.")
            return

        # Extract the time stamps from the log entries
        time_stamps = [entry.split()[0] + ' ' + entry.split()[1] for entry in log_entries if entry]

        # Construct the log erase command
        erase_command = f"sudo log erase --older '{time_stamps[-1]}' --predicateString '{log_command.split('--predicate ')[1]}'"

        # Run the log erase command
        subprocess.run(erase_command, shell=True, check=True)

        print(f"Successfully deleted {len(time_stamps)} log entries related to the specified {'PID' if is_pid else 'process name'}.")

    except subprocess.CalledProcessError as e:
        print(f"An error occurred while executing the command: {e}")
        print(f"Error output: {e.stderr}")
    except Exception as e:
        print(f"An unexpected error occurred: {e}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Search and delete macOS log entries for a specific process.")
    parser.add_argument("identifier", help="Process name or PID to search for in the logs")
    parser.add_argument("--pid", action="store_true", help="Treat the identifier as a PID instead of a process name")

    args = parser.parse_args()

    if os.geteuid() != 0:
        print("This script requires root privileges. Please run with sudo.")
        sys.exit(1)

    search_and_delete_log_entries(args.identifier, args.pid)
