    var captureSession: AVCaptureSession!

        super.viewDidLoad()
        
        // Initialize the capture session
        captureSession = AVCaptureSession()
        
        // Find the default video device (webcam)

            print("No video device found")
            return
        }

        do {
            // Create video device input
            let videoInput = try AVCaptureDeviceInput(device: videoDevice)
            
            // Add video input to the capture session

                captureSession.addInput(videoInput)
            } else {
                print("Could not add video input")
                return
            }

            // Create a video data output
            let videoOutput = AVCaptureVideoDataOutput()
            videoOutput.setSampleBufferDelegate(self, queue: DispatchQueue(label: "videoQueue"))
            
            // Add video output to the capture session
            if captureSession.canAddOutput(videoOutput) {
                captureSession.addOutput(videoOutput)
            } else {
                print("Could not add video output")
                return
            }
            
            // Display the video feed using a preview layer
            videoPreviewLayer = AVCaptureVideoPreviewLayer(session: captureSession)
            videoPreviewLayer.videoGravity = .resizeAspectFill
            videoPreviewLayer.frame = videoPreviewView.bounds
            
            // Add the preview layer to the video view
            videoPreviewView.layer = videoPreviewLayer
            
            // Start the capture session
            captureSession.startRunning()

        } catch {
            print("Error setting up video capture: \(error)")
        }
    }
}
