import os
from subprocess import Popen, PIPE

# Get the path to the software
software_path = "/Applications/Software.app"

# Check if the software is already in the login items
login_items = os.popen("osascript -e 'tell application \"System Events\" to get user login items'").read()
if software_path in login_items:
    print("The software is already in the login items.")
    exit(0)

# Add the software to the login items
os.popen("osascript -e 'tell application \"System Events\" to make login item at end with properties {path:\"%s\"}' %s" % (software_path, software_path))

print("The software has been added to the login items.")

import os

def add_login_item(app_path):
    """Adds an item to the login items list."""

    # Convert app path to quoted form for script compatibility
    app_path = os.path.expanduser(app_path)
    app_path = '"{}"'.format(app_path)

    # Build AppleScript command
    script = """
        set appPath to {}
        tell application "System Events"
            set newListItem to make new login item
            set path of newListItem to appPath
        end tell
    """.format(app_path)

    # Run AppleScript command
    os.system("osascript -e '{}'".format(script))

# Example usage
add_login_item("/Applications/MySoftware.app")

#!/bin/bash

# Get the path to the application
app_path=$(osascript -e 'on run argv
    tell application "Finder"
        return file of item 1 of argv
    end tell

# Create a launchd plist file

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">

    <array>
        <string>$app_path</string>
    </array>
    <key>RunAtLoad</key>
    <true/>
</dict>
</plist>
"""

# Write the plist file to disk
mkdir -p ~/Library/LaunchAgents
echo "$plist_file" > ~/Library/LaunchAgents/com.example.myapp.plist

# Reload the launchd daemon
launchctl load -w ~/Library/LaunchAgents/com.example.myapp.plist

echo "Software added to login items"

on load_at_startup(app_path)
    -- Get the path to the Login Items folder
    set login_items_folder to (path to library folder from user domain as text) & "LoginItems:"
    
    -- Create a symbolic link to the app in the Login Items folder
    tell application "Terminal"
        do shell script "ln -sf " & quoted form of app_path & " " & quoted form of login_items_folder
    end tell
end load_at_startup

try

on error
    display dialog "Failed to add software to Login Items."
end try
