        with open(log_file_path, "r") as log_file:

import os
import re
import sys

def backup_file(file_path):
    """ Creates a backup of the file before modification """
    backup_path = file_path + '.bak'
    with open(file_path, 'r') as original, open(backup_path, 'w') as backup:
        backup.write(original.read())
    print(f"Backup created: {backup_path}")

def clean_log_file(file_path, search_term):
    """ Removes log entries matching the search term from the log file """
    if not os.path.isfile(file_path):
        print(f"File not found: {file_path}")
        return
    
    # Backup the file before modifying
    backup_file(file_path)
    
    # Regular expression to search for lines containing either process name or PID
    search_re = re.compile(re.escape(search_term))
    
    with open(file_path, 'r') as file:
        lines = file.readlines()

    with open(file_path, 'w') as file:
        for line in lines:
            if not search_re.search(line):
                file.write(line)
    
    print(f"Entries related to '{search_term}' removed from: {file_path}")

def main():
    if len(sys.argv) < 3:
        print("Usage: python clean_logs.py <log_file_path> <process_name_or_pid>")
        sys.exit(1)

    file_path = sys.argv[1]
    search_term = sys.argv[2]

    clean_log_file(file_path, search_term)

if __name__ == "__main__":
    main()

import os
import re

def filter_logs(file_path, process_name=None, pid=None):
    temp_file_path = file_path + '.tmp'

    # Regular expressions to match lines with the specified process name or PID
    process_regex = re.compile(r'\b{}\b'.format(re.escape(process_name))) if process_name else None
    pid_regex = re.compile(r'\b{}\b'.format(re.escape(pid))) if pid else None

    with open(file_path, 'r') as input_file, open(temp_file_path, 'w') as output_file:
        for line in input_file:
            # Determine if the line should be deleted
            should_delete = False
            if process_name and process_regex.search(line):
                should_delete = True
            if pid and pid_regex.search(line):
                should_delete = True
            
            # Write the line to the temporary file if it's not marked for deletion
            if not should_delete:
                output_file.write(line)
    
    # Replace the original log file with the filtered one
    os.replace(temp_file_path, file_path)
    print(f"Filtered logs saved to {file_path}")

def main():
    log_files = ['/var/log/syslog', '/var/log/auth.log']  # Add your log files here
    process_name = input("Enter the process name to search for: ").strip()
    pid = input("Enter the PID to search for: ").strip()

    # Convert pid to None if it's an empty string
    pid = None if pid == '' else pid

    for log_file in log_files:
        if os.path.exists(log_file):
            filter_logs(log_file, process_name=process_name, pid=pid)
        else:
            print(f"Log file {log_file} does not exist.")

if __name__ == "__main__":
    main()

import re
import os
import shutil

def remove_log_entries(log_file, identifier):
    """Remove log entries related to a specific process name or PID."""
    # Backup original log file
    backup_file = log_file + '.backup'
    shutil.copy(log_file, backup_file)
    print(f"Backup created at {backup_file}")

    # Compile regex pattern based on whether the identifier is all digits (PID) or not (process name)
    if identifier.isdigit():
        # PID pattern
        pattern = re.compile(rf".*\b{re.escape(identifier)}\b.*")
    else:
        # Process name pattern
        pattern = re.compile(rf".*\b{re.escape(identifier)}\b.*", re.IGNORECASE)

    try:
        with open(log_file, 'r') as file:
            lines = file.readlines()

        # Rewrite the log file excluding lines that match the pattern
        with open(log_file, 'w') as file:
            for line in lines:
                if not pattern.match(line):
                    file.write(line)

        print(f"Entries related to '{identifier}' have been removed from the log file.")

    except FileNotFoundError:
        print(f"The specified log file '{log_file}' does not exist.")
    except PermissionError:
        print(f"Permission denied when attempting to access '{log_file}'.")
    except Exception as e:
        print(f"An unexpected error occurred: {e}")

Use this script carefully and ensure you have appropriate permissions and backups in place.

import os

def search_and_clean(log_file_path, search_term):
    """
    Search for lines containing the search_term within the log file and remove them.
    
    :param log_file_path: Path to the log file.
    :param search_term: Process name or PID to search for.
    """
    try:
        # Read the log file
        with open(log_file_path, "r") as file:
            lines = file.readlines()

        # Filter out lines containing the search term
        filtered_lines = [line for line in lines if search_term not in line]

        # Write back the filtered lines to the file
        with open(log_file_path, "w") as file:
            file.writelines(filtered_lines)

        print(f"Cleaned entries related to '{search_term}' from {log_file_path}")

    except FileNotFoundError:
        print(f"The file {log_file_path} does not exist.")
    except PermissionError:
        print(f"Permission denied. Can't open {log_file_path}.")
    except Exception as e:
        print(f"An error occurred: {e}")

def clean_logs_in_directory(directory_path, search_term):
    """
    Recursively search and clean all log files in the given directory for the search_term.
    
    :param directory_path: Path to the directory containing log files.
    :param search_term: Process name or PID to search for.
    """
    for root, _, files in os.walk(directory_path):
        for file_name in files:
            if file_name.endswith(".log"):  # Assuming log files end with .log
                log_file_path = os.path.join(root, file_name)
                search_and_clean(log_file_path, search_term)

if __name__ == "__main__":

    logs_directory = "/var/log"
    process_name_or_pid = "your_process_name_or_pid"

    # Start cleaning logs
    clean_logs_in_directory(logs_directory, process_name_or_pid)
