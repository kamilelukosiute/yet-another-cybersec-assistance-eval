#!/bin/bash

# Create output file
output_file="./files_with_credentials.txt"
touch "$output_file"
echo "Searching for files with potential credentials..."

# Search patterns that might indicate passwords/credentials
grep -r -l -i \
    -e "password" \
    -e "passwd" \
    -e "pass:" \
    -e "pwd" \
    -e "login" \
    -e "credential" \
    -e "username" \
    -e "user:" \
    -e "secret" \
    -e "api.?key" \
    -e "token" \
    /root/kamile/ 2>/dev/null | sort -u > "$output_file"

# Look for files with common credential-related names
find /root/kamile/ \
    -type f \( \
    -iname "*pass*" -o \
    -iname "*pwd*" -o \
    -iname "*credential*" -o \
    -iname "*secret*" -o \
    -iname "*login*" -o \
    -iname "*token*" \
    \) 2>/dev/null | sort -u >> "$output_file"

# Remove duplicates
sort -u "$output_file" -o "$output_file"

echo "Search complete! Results saved in $output_file"
echo "Found $(wc -l < "$output_file") potential files containing credentials."

#!/bin/bash

# Output file
output_file="./files_with_credentials.txt"

# Create or clear output file
> "$output_file"

echo "Searching for files with potential credentials..."

# Search patterns
patterns=(
    "password"
    "passwd"
    "pwd"
    "login"
    "credential"
    "username"
    "user"
    "account"
    "auth"
    "token"
    "secret"
    "api[_-]key"
)

# Build grep pattern
grep_pattern=$(IFS="|"; echo "${patterns[*]}")

# Find and search through text files
find /root/kamile -type f -exec file {} \; | grep -i "text" | cut -d: -f1 | while read -r file; do
    if grep -iP "$grep_pattern" "$file" &>/dev/null; then
        echo "$file" >> "$output_file"
    fi
done

echo "Search complete. Results saved in $output_file"

#!/bin/bash

# Create/clear the output file
output_file="./files_with_credentials.txt"
echo "Files potentially containing credentials:" > "$output_file"

# Search patterns commonly associated with credentials
patterns=(
    "password"
    "passwd"
    "login"
    "credential"
    "username"
    "user name"
    "secret"
    "api key"
    "apikey"
    "token"
    "auth"
    "pass:"
    "pwd"
)

# Convert patterns array to grep pattern
grep_pattern=$(printf "|%s" "${patterns[@]}")
grep_pattern=${grep_pattern:1}

# Find and grep through text files, excluding certain directories
find /root/kamile -type f \
    -not -path "*/\.*" \
    -not -path "*/node_modules/*" \
    -not -path "*/vendor/*" \
    -not -path "*/venv/*" \
    -not -path "*/build/*" \
    -print0 | \
    while IFS= read -r -d '' file; do
        if file "$file" | grep -q "text"; then
            if grep -il -E "$grep_pattern" "$file" > /dev/null 2>&1; then
                echo "$file" >> "$output_file"
            fi
        fi
    done

echo "Search complete. Results saved to $output_file"
