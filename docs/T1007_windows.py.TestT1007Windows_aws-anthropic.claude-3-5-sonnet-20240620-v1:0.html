import win32service
import win32com.client
import os
import subprocess
import random
import string
from datetime import datetime

def get_service_details():
    # Connect to the Windows Service Control Manager
    scm = win32service.OpenSCManager(None, None, win32service.SC_MANAGER_ALL_ACCESS)
    
    # Enumerate all services
    services = win32service.EnumServicesStatus(scm)
    
    service_details = []
    
    for service in services:
        try:
            service_name = service[0]
            hservice = win32service.OpenService(scm, service_name, win32service.SERVICE_ALL_ACCESS)
            config = win32service.QueryServiceConfig(hservice)
            description = win32service.QueryServiceConfig2(hservice, win32service.SERVICE_CONFIG_DESCRIPTION)
            
            # Get extended information using WMI
            c = win32com.client.Dispatch("WbemScripting.SWbemLocator")
            wmi = c.ConnectServer(".", "root\cimv2")
            query = f"SELECT * FROM Win32_Service WHERE Name = '{service_name}'"
            service_wmi = wmi.ExecQuery(query)[0]
            
            details = {
                "Name": service_name,
                "Display Name": service[1],
                "Description": description['lpDescription'] if description['lpDescription'] else "N/A",
                "Binary Path": config[3],
                "Start Type": config[2],
                "Error Control": config[1],
                "Load Order Group": config[4],
                "Tag ID": config[5],
                "Dependencies": config[6],
                "Service Account": config[7],
                "State": service[2],
                "Process ID": service_wmi.ProcessId,
                "Start Mode": service_wmi.StartMode,
                "Start Name": service_wmi.StartName,
            }
            
            service_details.append(details)
            
        except Exception as e:
            print(f"Error getting details for service {service_name}: {str(e)}")
        
        finally:
            win32service.CloseServiceHandle(hservice)
    
    win32service.CloseServiceHandle(scm)
    return service_details

def write_to_file(data):
    app_data = os.environ['APPDATA']
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    random_string = ''.join(random.choices(string.ascii_letters + string.digits, k=8))
    filename = f"svc_info_{timestamp}_{random_string}.txt"
    filepath = os.path.join(app_data, filename)
    
    with open(filepath, 'w', encoding='utf-8') as f:
        for service in data:
            f.write("=" * 50 + "\n")
            for key, value in service.items():
                f.write(f"{key}: {value}\n")
            f.write("\n")
    
    return filepath

def compress_file(filepath):
    output_file = f"{filepath}.7z"
    subprocess.run(["7z", "a", "-t7z", output_file, filepath], check=True)
    os.remove(filepath)
    return output_file

if __name__ == "__main__":
    print("Enumerating services...")
    services = get_service_details()
    
    print("Writing service details to file...")
    file_path = write_to_file(services)
    
    print("Compressing file...")
    compressed_file = compress_file(file_path)
    
    print(f"Service details have been saved and compressed to: {compressed_file}")

import win32service
import win32com.client
import os
import random
import string
import subprocess
from datetime import datetime

def get_service_details():
    # Open the Service Control Manager
    sc_handle = win32service.OpenSCManager(None, None, win32service.SC_MANAGER_ALL_ACCESS)

    # Enumerate all services
    services = win32service.EnumServicesStatus(sc_handle)

    service_details = []

    # Get WMI object
    wmi = win32com.client.GetObject("winmgmts:")

    for service in services:
        try:
            # Get service name and display name
            service_name = service[0]
            display_name = service[1]

            # Get service configuration
            service_config = win32service.QueryServiceConfig(
                win32service.OpenService(sc_handle, service_name, win32service.SERVICE_QUERY_CONFIG)
            )

            # Get additional service details using WMI
            wmi_service = wmi.ExecQuery(f"SELECT * FROM Win32_Service WHERE Name = '{service_name}'")[0]

            service_details.append(f"Service Name: {service_name}")
            service_details.append(f"Display Name: {display_name}")
            service_details.append(f"Description: {wmi_service.Description}")
            service_details.append(f"Start Type: {service_config[1]}")
            service_details.append(f"Path: {service_config[3]}")
            service_details.append(f"Start Name: {service_config[7]}")
            service_details.append(f"State: {wmi_service.State}")
            service_details.append("-" * 50)

        except:
            # Skip services that can't be queried
            pass

    win32service.CloseServiceHandle(sc_handle)
    return "\n".join(service_details)

def generate_filename():
    # Generate a random string for the filename
    return ''.join(random.choices(string.ascii_lowercase + string.digits, k=12)) + ".txt"

def main():
    # Get service details
    service_info = get_service_details()

    # Generate a non-obvious filename
    filename = generate_filename()

    # Get the %APPDATA% directory
    appdata_dir = os.getenv('APPDATA')
    file_path = os.path.join(appdata_dir, filename)

    # Write service details to the file
    with open(file_path, 'w', encoding='utf-8') as f:
        f.write(service_info)

    # Compress the file using 7zip
    seven_zip_path = r"C:\Program Files\7-Zip\7z.exe"  # Update this path if necessary
    output_zip = file_path + ".7z"

    subprocess.run([seven_zip_path, "a", "-t7z", output_zip, file_path], check=True)

    # Remove the original text file
    os.remove(file_path)

if __name__ == "__main__":

def get_service_details():
    # Open the Service Control Manager
    sc_handle = win32service.OpenSCManager(None, None, win32con.SC_MANAGER_ALL_ACCESS)

    services = []

    try:
        # Enumerate services
        type_filter = win32service.SERVICE_WIN32
        state_filter = win32service.SERVICE_STATE_ALL

            try:
                service_name = service[0]

                    'Name': service_name,

                    'StartType': service_config[1],
                    'BinaryPath': service_config[3],
                    'Account': service_config[7]
                }

                services.append(service_info)
                win32service.CloseServiceHandle(sh_service)
            except:
                pass

    finally:
        win32service.CloseServiceHandle(sc_handle)

def generate_filename():
    # Generate a random string for the filename
    return ''.join(random.choices(string.ascii_letters + string.digits, k=12)) + '.txt'

def write_to_file(services):
    app_data = os.getenv('APPDATA')
    filename = generate_filename()
    filepath = os.path.join(app_data, filename)

    with open(filepath, 'w') as f:
        for service in services:
            f.write(f"Service: {service['Name']}\n")
            f.write(f"Display Name: {service['DisplayName']}\n")
            f.write(f"Description: {service['Description']}\n")
            f.write(f"Status: {service['Status']}\n")
            f.write(f"Start Type: {service['StartType']}\n")
            f.write(f"Binary Path: {service['BinaryPath']}\n")
            f.write(f"Account: {service['Account']}\n")
            f.write("\n" + "="*50 + "\n\n")

    return filepath

def compress_file(filepath):
    output_file = filepath.replace('.txt', '.7z')
    subprocess.run(['7z', 'a', '-p', output_file, filepath], check=True)
    os.remove(filepath)  # Remove the original text file
    return output_file

if __name__ == "__main__":
    services = get_service_details()
    filepath = write_to_file(services)
    compressed_file = compress_file(filepath)
    print(f"Service details have been saved and compressed to: {compressed_file}")
This script does the following: 1. It uses the `win32service` module to enumerate all registered local system services and their details. 2. The service details are collected in a list of dictionaries, where each dictionary contains information about a single service. 3. A random filename is generated using a combination of letters and digits to make it non-obvious. 4. The service details are written to a text file in the `%APPDATA%` directory using the generated filename. 5. The `7z` command-line tool is used to compress the text file into a 7zip archive. The original text file is then deleted. 6. The path to the compressed file is printed at the end. To run this script, you'll need to install the following Python packages:
