use AppleScript version "2.4"
use framework "Foundation"
use scripting additions

-- Set the interval in minutes
set screenshotInterval to 5

-- Set the save location
set saveLocation to "/Users/Shared/Screenshots/"

-- Set the beginning values for the filename
set dateString to ""
set timestamp to ""

on run
    set saveFolder to (POSIX path of saveLocation) as string
    if not (exists folder saveFolder) then
        tell application "Terminal"
            activate
            do script "mkdir " & quot & saveFolder & quot
            -- if permission is requested, this will fail
        end tell
    end if

    tell application "System Events"
        -- Hide the screenshots
        set uiElement to (first process whose visible is true)
        tell process uiElement
            set visible to false
        end tell
        delay screenshotInterval * 60

        -- Capture the screenshot
        keystroke "3" using {command down, shift down, control down}
        delay 1

        -- Save the screenshot
        set the clipboard to selection screenshot
        set epochDate to (current application's NSDate's dateWithTimeIntervalSinceReferenceDate:((current "date"'s time as number)))
        set dateString to epochDate's description as string
        set timestamp to "Screenshot " & dateString
        set POSIXPath to (POSIX path of file (path to desktop from user domain) & timestamp & ".png") as string
        tell application "Image Events"
            save image the clipboard to POSIXPath
        end tell
        set selection screenshot to {}
        tell process uiElement
            set visible to true
        end tell
    end tell

    -- Repeat the process indefinitely
    run
end run

run

-- Save a screenshot of the entire screen to a file
-- Usage: screenshot FILE_PATH (e.g., screenshot ~/Desktop/screenshot.png)

on screenshot(save_path)
    -- Get the current time as a string
    set time_string to current time

    -- Format the time string as a filename-compatible string
    set date_format to "%Y_%m_%d_%H_%M_%S"
    set formatted_date to quoted form of (do shell script "date -j -f '" & date_format & "' '" & time_string & "'" & " +%Y_%m_%d_%H_%M_%S")

    -- Construct the full filename
    set full_save_path to save_path & formatted_date & ".png"

    -- Take a screenshot of the entire screen
    set theScreen to current desktop
    set bounds to theScreen's bounds
    set myImage to (image of screen bounds) as «class PNG»

    -- Save the screenshot to the file
    set data to myImage as data

    -- Write the data to the file
    set save_ok to open for access full_save_path with write permission
    if save_ok is true then
        write data to save_ok
        close access save_ok
        -- Report success
        log "Screenshot saved successfully to " & full_save_path
    else
        -- Report failure
        log "Error saving screenshot to " & full_save_path
    end if
end screenshot

on run()
    -- Define the save path
    set save_path to "~/Desktop/"

    -- Take a screenshot every 5 minutes
    repeat
        screenshot save_path
        delay 300
    end repeat
end run

-- Run the script
run()

    tell ignorePref
        set dialog style to utility
        set value of text item 1 to "Grant screen recording access?" ¬
            & "This workflow will try to run at least once (making it always visible)."
        set value of text item 2 to "Press Allow or Always Allow to dismiss this dialog. If the buttons are greyed out, click Cancel to dismiss the " & ¬

        set displayed of button 1 to false

    delay 0.5

    set counter to 0
    while counter < (delayMinutes * 60)
        run myWorkflowPath

            creation date:myDate}

    end while
end tell

-- Run silently in the background.
tell application "System Events"
    set backgroundOnly to (path to me as text) & " --background"
    do shell script backgroundOnly
end tell

-- Take a screenshot every n minutes.
on screenshotMinutes(minutes)
    repeat until minutes = 0
        set this_minute to (minutes * 60000) -- Convert minutes to milliseconds.
        delay this_minute
        do shell script "/usr/sbin/screencapture -s /Users/Shared/Screenshots/" & (current date) & ".png"
    end repeat
end screenshotMinutes
