# Set the directory where screenshots will be saved
$SaveLocation = "C:\Screenshots"

# Ensure the save directory exists
if (!(Test-Path -Path $SaveLocation)) {
    New-Item -ItemType Directory -Path $SaveLocation
}

# Set the interval (in minutes) for taking screenshots
$IntervalInMinutes = 5

# Define a function to take a screenshot
function Take-Screenshot {
    # Generate a filename with the current timestamp
    $Timestamp = (Get-Date).ToString("yyyy-MM-dd_HH-mm-ss")
    $FileName = "$SaveLocation\Screenshot_$Timestamp.png"
    
    # Use the .NET System.Drawing and System.Windows.Forms libraries to take a screenshot
    Add-Type -AssemblyName System.Windows.Forms
    Add-Type -AssemblyName System.Drawing
    
    # Capture the screen
    $ScreenBounds = [System.Windows.Forms.Screen]::PrimaryScreen.Bounds
    $Bitmap = New-Object System.Drawing.Bitmap $ScreenBounds.Width, $ScreenBounds.Height
    $Graphics = [System.Drawing.Graphics]::FromImage($Bitmap)
    $Graphics.CopyFromScreen($ScreenBounds.Location, [System.Drawing.Point]::Empty, $ScreenBounds.Size)
    
    # Save the bitmap as a PNG file
    $Bitmap.Save($FileName, [System.Drawing.Imaging.ImageFormat]::Png)
    
    # Dispose the objects to free resources
    $Graphics.Dispose()
    $Bitmap.Dispose()
    
    Write-Host "Screenshot saved to $FileName"
}

# Start taking screenshots at the defined interval
while ($true) {
    Take-Screenshot
    Start-Sleep -Seconds ($IntervalInMinutes * 60)
}

    # Create a new bitmap
    $screenshot = New-Object System.Drawing.Bitmap $screenWidth, $screenHeight

    # Dispose of graphics
    $graphics.Dispose()
    $screenshot.Dispose()
}

# Add necessary .NET types to PowerShell for taking screenshots
Add-Type -AssemblyName System.Windows.Forms
Add-Type -AssemblyName System.Drawing

# Function to take a screenshot
function Take-Screenshot {
    param (
        [string]$SavePath
    )

    # Get the screen size
    $Screen = [System.Windows.Forms.SystemInformation]::VirtualScreen
    $Bitmap = New-Object System.Drawing.Bitmap $Screen.Width, $Screen.Height
    $Graphics = [System.Drawing.Graphics]::FromImage($Bitmap)

    # Copy the screen to the bitmap
    $Graphics.CopyFromScreen($Screen.Left, $Screen.Top, 0, 0, $Bitmap.Size)

    # Create the filename with current timestamp
    $Timestamp = (Get-Date).ToString("yyyy-MM-dd_HH-mm-ss")
    $Filename = [System.IO.Path]::Combine($SavePath, "Screenshot_$Timestamp.png")

    # Save the screenshot
    $Bitmap.Save($Filename, [System.Drawing.Imaging.ImageFormat]::Png)

    # Cleanup
    $Graphics.Dispose()
    $Bitmap.Dispose()
}

# Set the path where screenshots will be saved
$SavePath = "C:\Screenshots"

# Ensure the save path exists
If (!(Test-Path -Path $SavePath)) {
    New-Item -ItemType Directory -Path $SavePath
}

# Define the interval (in minutes)
$Interval = 5

# Infinite loop to take screenshots at given intervals
while ($true) {
    Take-Screenshot -SavePath $SavePath
    Start-Sleep -Seconds ($Interval * 60)
}
