# Path to save screenshots
$saveDirectory = "C:\Screenshots"

# Ensure the directory exists
if (-Not (Test-Path -Path $saveDirectory)) {
    New-Item -Path $saveDirectory -ItemType Directory | Out-Null
}

# Interval in minutes
$intervalMinutes = 5

# Function to capture screenshot
function Capture-Screenshot {
    Add-Type -AssemblyName System.Windows.Forms
    Add-Type -AssemblyName System.Drawing

    $bitmap = New-Object System.Drawing.Bitmap([System.Windows.Forms.SystemInformation]::VirtualScreen.Width, [System.Windows.Forms.SystemInformation]::VirtualScreen.Height)
    $graphics = [System.Drawing.Graphics]::FromImage($bitmap)
    $graphics.CopyFromScreen([System.Drawing.Point]::Empty, [System.Drawing.Point]::Empty, $bitmap.Size)

    # Generate filename with timestamp
    $timestamp = (Get-Date).ToString("yyyyMMdd_HHmmss")
    $fileName = Join-Path -Path $saveDirectory -ChildPath "Screenshot_$timestamp.png"

    # Save the bitmap to file
    $bitmap.Save($fileName, [System.Drawing.Imaging.ImageFormat]::Png)

    # Clean up
    $graphics.Dispose()
    $bitmap.Dispose()
}

# Main loop to capture screenshots periodically
while ($true) {
    try {
        Capture-Screenshot
        Write-Host "Screenshot taken and saved."
    } catch {
        Write-Error "Failed to capture screenshot: $_"
    }
    Start-Sleep -Seconds ($intervalMinutes * 60)
}

# Define the function to take a screenshot
function Capture-Screenshot {
    param (
        [string]$SavePath
    )

    # Add types for drawing (used for screenshot functionality)
    Add-Type -AssemblyName System.Windows.Forms
    Add-Type -AssemblyName System.Drawing

    # Set screen bounds
    $ScreenBounds = [System.Windows.Forms.Screen]::PrimaryScreen.Bounds

    # Create a new Bitmap object
    $Bitmap = New-Object System.Drawing.Bitmap $ScreenBounds.Width, $ScreenBounds.Height

    # Create a graphics object from the bitmap
    $Graphics = [System.Drawing.Graphics]::FromImage($Bitmap)

    # Copy screen content to graphics object
    $Graphics.CopyFromScreen($ScreenBounds.Location, [System.Drawing.Point]::Empty, $ScreenBounds.Size)

    # Dispose objects (important to free memory)
    $Graphics.Dispose()
    $Bitmap.Dispose()

    Write-Host "Screenshot saved to $FilePath"
}

# Add necessary types to call Windows API functions
Add-Type @"
    using System;
    using System.Drawing;
    using System.Drawing.Imaging;
    using System.Runtime.InteropServices;
    public class ScreenCapture {
        [DllImport("gdi32.dll")]
        private static extern bool BitBlt(IntPtr hdcDest, int xDest, int yDest, int wDest, int hDest,
            IntPtr hdcSrc, int xSrc, int ySrc, int rop);
        [DllImport("user32.dll")]
        private static extern bool ReleaseDC(IntPtr hwnd, IntPtr hdc);
        [DllImport("gdi32.dll")]
        private static extern IntPtr DeleteDC(IntPtr hdc);
        [DllImport("gdi32.dll")]
        private static extern IntPtr DeleteObject(IntPtr hObject);
        public void CaptureScreen(string filePath) {
            var screenSize = System.Windows.Forms.Screen.PrimaryScreen.Bounds.Size;
            using (Bitmap bmp = new Bitmap(screenSize.Width, screenSize.Height))
            using (Graphics g = Graphics.FromImage(bmp))
            {
                IntPtr hdcDest = g.GetHdc();
                IntPtr hdcSrc = Graphics.FromHwnd(IntPtr.Zero).GetHdc();
                BitBlt(hdcDest, 0, 0, bmp.Width, bmp.Height, hdcSrc, 0, 0, 0x00CC0020);
                g.ReleaseHdc(hdcDest);
                bmp.Save(filePath, ImageFormat.Png);
            }
        }
    }
"@ -ReferencedAssemblies System.Drawing, System.Windows.Forms

$screenCapture = New-Object ScreenCapture

    # Capture the screen and save the image
    $screenCapture.CaptureScreen($filePath)
    Write-Output "Screenshot saved to $filePath"

    # Wait for the specified interval
    Start-Sleep -Seconds ($interval * 60)
}
