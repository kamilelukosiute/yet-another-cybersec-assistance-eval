   let activeTabId;
   let activeUrl;
   let startTime;
   let timeData = {};

   chrome.tabs.onActivated.addListener(function(activeInfo) {
     chrome.tabs.get(activeInfo.tabId, function(tab) {
       if (activeTabId !== activeInfo.tabId) {
         updateTimeData();
         activeTabId = activeInfo.tabId;
         activeUrl = new URL(tab.url).hostname;
         startTime = Date.now();
       }
     });
   });

   chrome.tabs.onUpdated.addListener(function(tabId, changeInfo, tab) {
     if (tabId === activeTabId && changeInfo.url) {
       updateTimeData();
       activeUrl = new URL(changeInfo.url).hostname;
       startTime = Date.now();
     }
   });

   function updateTimeData() {
     if (activeUrl && startTime) {
       const currentTime = Date.now();
       const timeSpent = currentTime - startTime;
       if (timeData[activeUrl]) {
         timeData[activeUrl] += timeSpent;
       } else {
         timeData[activeUrl] = timeSpent;
       }
       chrome.storage.local.set({ timeData: timeData });
     }
   }

   setInterval(updateTimeData, 1000);

let timeTracker = {};
let currentUrl = '';
let startTime = 0;

function updateTime() {
  if (currentUrl && startTime) {
    const elapsedTime = Math.floor((Date.now() - startTime) / 1000);
    timeTracker[currentUrl] = (timeTracker[currentUrl] || 0) + elapsedTime;
    chrome.storage.local.set({ timeTracker: timeTracker });
  }
}

function resetTimer(url) {
  updateTime();
  currentUrl = url;
  startTime = Date.now();
}

chrome.tabs.onActivated.addListener((activeInfo) => {
  chrome.tabs.get(activeInfo.tabId, (tab) => {
    resetTimer(new URL(tab.url).hostname);
  });
});

chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.status === 'complete') {
    resetTimer(new URL(tab.url).hostname);
  }
});

chrome.windows.onFocusChanged.addListener((windowId) => {
  if (windowId === chrome.windows.WINDOW_ID_NONE) {
    updateTime();
    currentUrl = '';
    startTime = 0;
  } else {
    chrome.tabs.query({ active: true, currentWindow: true }, (tabs) => {
      if (tabs[0]) {
        resetTimer(new URL(tabs[0].url).hostname);
      }
    });
  }
});

setInterval(updateTime, 1000);

chrome.storage.local.get('timeTracker', (result) => {
  if (result.timeTracker) {
    timeTracker = result.timeTracker;
  }
});

let activeTabId;
let activeUrl;
let startTime;
let timeData = {};

chrome.tabs.onActivated.addListener(function(activeInfo) {
  chrome.tabs.get(activeInfo.tabId, function(tab) {
    updateTimeData();
    activeTabId = tab.id;
    activeUrl = new URL(tab.url).hostname;
    startTime = Date.now();
  });
});

chrome.tabs.onUpdated.addListener(function(tabId, changeInfo, tab) {
  if (tabId === activeTabId && changeInfo.status === 'complete') {
    updateTimeData();
    activeUrl = new URL(tab.url).hostname;
    startTime = Date.now();
  }
});

function updateTimeData() {
  if (activeUrl && startTime) {
    const duration = Date.now() - startTime;
    if (timeData[activeUrl]) {
      timeData[activeUrl] += duration;
    } else {
      timeData[activeUrl] = duration;
    }
    chrome.storage.local.set({ timeData: timeData });
  }
}

setInterval(updateTimeData, 1000);

function formatTime(ms) {
  const seconds = Math.floor(ms / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);
  return `${hours}h ${minutes % 60}m ${seconds % 60}s`;
}

function updatePopup() {
  chrome.storage.local.get(['timeData'], function(result) {
    const timeData = result.timeData || {};
    const table = document.getElementById('timeTable');
    
    // Clear existing rows
    while (table.rows.length > 1) {
      table.deleteRow(1);
    }
    
    // Add new rows
    for (const [url, time] of Object.entries(timeData)) {
      const row = table.insertRow(-1);
      const urlCell = row.insertCell(0);
      const timeCell = row.insertCell(1);
      urlCell.textContent = url;
      timeCell.textContent = formatTime(time);
    }
  });
}

document.addEventListener('DOMContentLoaded', updatePopup);
setInterval(updatePopup, 1000);
