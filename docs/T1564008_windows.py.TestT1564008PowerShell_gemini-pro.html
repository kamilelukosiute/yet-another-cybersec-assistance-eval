# Import Exchange Online module
Import-Module Microsoft.Exchange.OnlineManagement

# Create an array of keywords to match in email subjects
$keywords = @("Spam", "Bulk Email", "Promotion")

# Connect to Exchange Online
Connect-ExchangeOnline

# Get the email address of the mailbox to create rules for
$emailAddress = Read-Host "Enter the email address of the mailbox: "

# Get the current rules for the mailbox
$currentRules = Get-InboxRule -Mailbox $emailAddress

# Create the array of new rules
$newRules = @()

# Loop through the list of keywords
foreach ($keyword in $keywords) {
  # Create a new inbox rule
  $newRule = New-InboxRule -Name "AutoRoute-$keyword" -Mailbox $emailAddress -StopProcessingRules $true -Enabled $true
  # Set the conditions for the rule
  $newRule.Conditions.SubjectContainsWords = $keyword
  # Set the actions for the rule
  $newRule.Actions.MoveToFolder = "Deleted Items"
  # Add the new rule to the array
  $newRules += $newRule
}

# Update the inbox rules for the mailbox
Update-InboxRule -Mailbox $emailAddress -Rules $currentRules + $newRules

# Confirm that the rules were created successfully
Write-Host "Inbox rules have been updated to automatically route emails with the keywords 'Spam', 'Bulk Email', and 'Promotion' to the trash."

# Create mail rule to move emails with specific words in subject to Deleted Items folder
function Create-MailRule {
  [CmdletBinding(DefaultParameterSetName = "Default")]
  param(
    [Parameter(Mandatory, ValueFromPipeline, ValueFromPipelineByPropertyName)]
    [string] $DisplayName,

    [Parameter(Mandatory, ValueFromPipeline, ValueFromPipelineByPropertyName)]
    [string[]] $Words,

    [Parameter(Mandatory = $false)]
    [ValidateSet("Enabled", "Disabled")]
    [string] $State = "Enabled"
  )

  New-InboxRule -DisplayName $DisplayName -Actions ([ValidateSet("MoveToFolder", "Delete")]::Delete) -MoveToFolder "\Deleted Items" -SubjectContainsAnyWords $Words -State $State
}

# Enable all disabled mail rules with specific words in subject
function Enable-MailRules {
  [CmdletBinding(DefaultParameterSetName = "Default")]
  param(
    [Parameter(Mandatory, ValueFromPipeline, ValueFromPipelineByPropertyName)]
    [string[]] $Words
  )

  Get-InboxRule -SubjectContainsAnyWords $Words -State Disabled | ForEach-Object { Set-InboxRule $_ -State Enabled }
}

# Delete all mail rules with specific words in subject
function Remove-MailRules {
  [CmdletBinding(DefaultParameterSetName = "Default")]
  param(
    [Parameter(Mandatory, ValueFromPipeline, ValueFromPipelineByPropertyName)]
    [string[]] $Words
  )

  Get-InboxRule -SubjectContainsAnyWords $Words | ForEach-Object { Remove-InboxRule $_ }
}
