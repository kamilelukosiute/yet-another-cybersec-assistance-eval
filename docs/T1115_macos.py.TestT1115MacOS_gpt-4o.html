#!/bin/bash

# File to store clipboard history
HISTORY_FILE="history.txt"

# Initialize an empty previous clipboard content
prev_clipboard=""

# Create or clear the history file at the start
> "$HISTORY_FILE"

# Function to get the current clipboard content
get_clipboard_content() {
    pbpaste
}

# Infinite loop to continuously monitor clipboard changes
while true; do
    # Get the current clipboard content
    current_clipboard=$(get_clipboard_content)

    # Check if the clipboard content has changed
    if [ "$current_clipboard" != "$prev_clipboard" ]; then
        # If it has changed and is non-empty, append the new content to history file
        if [ -n "$current_clipboard" ]; then
            echo "$current_clipboard" >> "$HISTORY_FILE"
            # Update the previous clipboard content
            prev_clipboard="$current_clipboard"
        fi
    fi

    # Sleep for a short while to minimize CPU usage
    sleep 1
done

#!/bin/bash

# File where clipboard history will be stored
HISTORY_FILE="history.txt"

# Create the history file if it doesn't exist
touch $HISTORY_FILE

# Function to get the current clipboard content
get_clipboard_content() {
    pbpaste
}

# Keep track of the last clipboard content
LAST_CONTENT=""

# Function to monitor clipboard for changes
monitor_clipboard() {
    while true; do
        # Get the current clipboard content
        CURRENT_CONTENT=$(get_clipboard_content)

        # Check for changes in the clipboard
        if [[ "$CURRENT_CONTENT" != "$LAST_CONTENT" && -n "$CURRENT_CONTENT" ]]; then
            # Update the last content variable
            LAST_CONTENT="$CURRENT_CONTENT"

            # Append the new clipboard content to the history file
            echo "$CURRENT_CONTENT" >> $HISTORY_FILE
        fi

        # Sleep for a short duration to reduce CPU usage
        sleep 1
    done
}

# Run the monitor function in the background
monitor_clipboard &
