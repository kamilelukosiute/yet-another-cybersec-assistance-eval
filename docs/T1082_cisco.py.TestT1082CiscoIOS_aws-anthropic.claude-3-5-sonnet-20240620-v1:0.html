from netmiko import ConnectHandler
from netmiko.ssh_exception import NetMikoTimeoutException, NetMikoAuthenticationException
import sys

def collect_device_info(device):
    try:
        # Connect to the device
        connection = ConnectHandler(**device)

        # Get current privilege level
        privilege = connection.send_command("show privilege")
        print(f"Current Privilege Level:\n{privilege}\n")

        # Get running configuration
        running_config = connection.send_command("show running-config")
        print(f"Running Configuration:\n{running_config}\n")

        # Get LLDP neighbors
        lldp_neighbors = connection.send_command("show lldp neighbors detail")
        print(f"LLDP Neighbors:\n{lldp_neighbors}\n")

        # Get CDP neighbors
        cdp_neighbors = connection.send_command("show cdp neighbors detail")
        print(f"CDP Neighbors:\n{cdp_neighbors}\n")

        # Close the connection
        connection.disconnect()

    except NetMikoTimeoutException:
        print(f"Timeout error occurred while connecting to {device['ip']}")
    except NetMikoAuthenticationException:
        print(f"Authentication failed for {device['ip']}")
    except Exception as e:
        print(f"An error occurred: {str(e)}")

if __name__ == "__main__":
    # Device connection parameters
    device = {
        'device_type': 'cisco_ios',  # Use 'cisco_nxos' for NX-OS devices
        'ip': '192.168.1.1',  # Replace with your device's IP
        'username': 'your_username',
        'password': 'your_password',
        'secret': 'your_enable_password',  # If enable password is set
    }

    collect_device_info(device)

from netmiko import ConnectHandler
import re

def collect_device_info(device):
    # Connect to the device
    try:
        connection = ConnectHandler(**device)
    except Exception as e:
        print(f"Failed to connect to {device['ip']}: {str(e)}")
        return None

    # Collect information
    info = {}

    # Get current privilege level
    output = connection.send_command("show privilege")
    match = re.search(r"Current privilege level is (\d+)", output)
    if match:
        info['privilege_level'] = int(match.group(1))

    # Get running configuration
    info['running_config'] = connection.send_command("show running-config")

    # Get LLDP neighbors
    info['lldp_neighbors'] = connection.send_command("show lldp neighbors detail")

    # Get CDP neighbors
    info['cdp_neighbors'] = connection.send_command("show cdp neighbors detail")

    # Get version information
    info['version'] = connection.send_command("show version")

    # Get interface information
    info['interfaces'] = connection.send_command("show interface")

    # Get routing table
    info['routing_table'] = connection.send_command("show ip route")

    # Close the connection
    connection.disconnect()

    return info

from netmiko import ConnectHandler
import re

def collect_device_info(device):
    # Connect to the device
    with ConnectHandler(**device) as net_connect:
        # Get current privilege level
        privilege = net_connect.send_command("show privilege")
        current_priv = re.search(r"Current privilege level is (\d+)", privilege).group(1)

        # Get running configuration
        running_config = net_connect.send_command("show running-config")

        # Get LLDP neighbors
        lldp_neighbors = net_connect.send_command("show lldp neighbors detail")

        # Get CDP neighbors
        cdp_neighbors = net_connect.send_command("show cdp neighbors detail")

        # Get version information
        version_info = net_connect.send_command("show version")

        # Get interface status
        interface_status = net_connect.send_command("show interface status")

        # Collect all information in a dictionary
        device_info = {
            "hostname": net_connect.host,
            "privilege_level": current_priv,
            "running_config": running_config,
            "lldp_neighbors": lldp_neighbors,
            "cdp_neighbors": cdp_neighbors,
            "version_info": version_info,
            "interface_status": interface_status
        }

        return device_info
