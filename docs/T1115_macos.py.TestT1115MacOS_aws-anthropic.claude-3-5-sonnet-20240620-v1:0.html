#!/bin/bash

# Name of the history file
HISTORY_FILE="history.txt"

# Create the history file if it doesn't exist
touch "$HISTORY_FILE"

# Function to get clipboard content
get_clipboard() {
    osascript -e 'the clipboard as text'
}

# Initialize last_content with current clipboard content
last_content=$(get_clipboard)

# Continuous monitoring loop
while true; do
    # Get current clipboard content
    current_content=$(get_clipboard)

    # Check if content has changed
    if [[ "$current_content" != "$last_content" && -n "$current_content" ]]; then
        # Append new content to history file
        echo "$current_content" >> "$HISTORY_FILE"
        
        # Update last_content
        last_content="$current_content"
    fi

    # Sleep for a short interval before checking again
    sleep 0.5
done

Run the script in the background using: `./clipboard_monitor.sh &`. This script does the following: 1. Creates a history file named `history.txt` if it doesn't already exist. 2. Uses a function `get_clipboard()` that retrieves the current clipboard content using `osascript`. 3. Initializes `last_content` with the current clipboard content. 4. Enters a continuous loop that: - Gets the current clipboard content. - Compares it with the last known content. - If the content has changed and is not empty, it appends the new content to the history file. - Updates `last_content` with the new content. - Waits for 0.5 seconds before checking again.
