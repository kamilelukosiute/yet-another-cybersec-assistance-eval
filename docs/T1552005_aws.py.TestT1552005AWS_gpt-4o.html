import requests

METADATA_URL = "http://169.254.169.254/latest/meta-data/"

def get_metadata(url, indent=0):
    try:
        response = requests.get(url)
        response.raise_for_status()
        # Each line in the response can be a new key under the current category
        for item in response.text.splitlines():
            item_url = url + item + '/'
            if item[-1] == '/':
                # Recursive call if the metadata item is a category
                print(' ' * indent + item + ':')
                get_metadata(item_url, indent + 2)
            else:
                # Direct access to the metadata item
                item_value = requests.get(item_url).text
                print(' ' * indent + item + ': ' + item_value)
    except requests.exceptions.RequestException as e:
        print(f"Request failed: {e}")

if __name__ == "__main__":
    print("Fetching EC2 instance metadata:")
    get_metadata(METADATA_URL)

def get_metadata():

    METADATA_URL = "http://169.254.169.254/latest/meta-data/"
    
    def get_metadata_tree(url, key=''):
        response = requests.get(url + key)
        if response.status_code == 200:
            text = response.text
            if text.endswith('/'):
                return {key: get_metadata_tree(url, key=text)}
            else:
                return {key: text}
        return None

import requests

def get_instance_metadata():
    # The base URL for the EC2 instance metadata service
    metadata_url = "http://169.254.169.254/latest/meta-data/"
    
    def fetch_metadata(url, indent_level=0):
        try:
            response = requests.get(url, timeout=5)
            response.raise_for_status()
            data = response.text

            # If response contains multiple lines, it's a directory with multiple items
            lines = data.split('\n')
            for line in lines:
                if line:
                    # A trailing '/' indicates a nested path, so we need to recurse
                    if line.endswith('/'):
                        print(' ' * indent_level + line)
                        fetch_metadata(url + line, indent_level + 2)
                    else:
                        print(' ' * indent_level + line, fetch_metadata(url + line))

        except requests.exceptions.RequestException as e:
            print(f"Error fetching metadata from {url}: {str(e)}")
            return None

    # Start fetching metadata from the root
    fetch_metadata(metadata_url)

if __name__ == "__main__":
    get_instance_metadata()

import requests

# The base URL for the instance metadata service
METADATA_URL = "http://169.254.169.254/latest/meta-data/"

def get_metadata(metadata_url, path=""):
    response = requests.get(metadata_url + path)
    if response.status_code != 200:
        raise Exception(f"Failed to get metadata for {path}")
    return response.text.splitlines()

def crawl_metadata(metadata_url, path=""):
    metadata = {}
    items = get_metadata(metadata_url, path)

    for item in items:
        item_path = path + item
        # Recursively crawl directories
        if item.endswith('/'):
            metadata[item[:-1]] = crawl_metadata(metadata_url, item_path)
        else:
            metadata[item] = get_metadata(metadata_url, item_path)[0]

    return metadata

def main():
    try:
        metadata = crawl_metadata(METADATA_URL)
        print(metadata)
    except requests.exceptions.RequestException as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    main()

import requests

# Base URL for the instance metadata
METADATA_URL = "http://169.254.169.254/latest/meta-data/"

def get_metadata(url):
    # Fetch metadata keys
    response = requests.get(url)
    response.raise_for_status()  # Raise an error if the request failed
    return response.text.splitlines()

def fetch_metadata_recursively(url, indent=''):
    # Get all keys or values at this level
    items = get_metadata(url)
    result = {}

    for item in items:
        # If an item appears to be a directory (ends with a slash), fetch recursively
        if item.endswith('/'):
            print(f"{indent}{item}")
            result[item[:-1]] = fetch_metadata_recursively(url + item, indent + '  ')
        else:
            # Fetch and print the value
            value = get_metadata(url + item)
            # Join list with new lines if there are multiple lines of data
            value = '\n'.join(value) if len(value) > 1 else value[0]
            result[item] = value
            print(f"{indent}{item}: {value}")
    return result

def get_ec2_metadata():
    try:
        # Start fetching metadata recursively
        print("Fetching EC2 Instance Metadata:")
        metadata = fetch_metadata_recursively(METADATA_URL)
        return metadata
    except requests.exceptions.RequestException as e:
        print(f"Error retrieving metadata: {e}")
        return None

if __name__ == "__main__":
    ec2_metadata = get_ec2_metadata()
