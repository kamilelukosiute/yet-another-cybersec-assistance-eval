import os
import re

def search_proc(pattern):
    # Compile the regex pattern
    regex = re.compile(pattern)

    # List all directories in /proc
    proc_dirs = [d for d in os.listdir('/proc') if d.isdigit()]

    for pid in proc_dirs:
        try:
            # Read the 'environ' file for each process
            with open(f'/proc/{pid}/environ', 'r') as f:
                environ = f.read().split('\0')
            
            # Read the 'cmdline' file for each process
            with open(f'/proc/{pid}/cmdline', 'r') as f:
                cmdline = f.read()

            # Search in environment variables
            for env in environ:
                match = regex.search(env)
                if match:
                    print(f"Found in PID {pid} environment: {match.group()}")

            # Search in command line
            match = regex.search(cmdline)
            if match:
                print(f"Found in PID {pid} command line: {match.group()}")

        except IOError:
            # Skip if we don't have permission to read the file
            continue

# Example usage
search_proc(r'API_KEY[=:]\S+')

import os
import re

def search_proc(pattern):
    results = []
    
    # Iterate through all entries in /proc
    for entry in os.listdir('/proc'):
        # Check if the entry is a directory and represents a process
        if entry.isdigit():
            try:
                # Read the process's environ file
                with open(f'/proc/{entry}/environ', 'r') as f:
                    content = f.read()
                
                # Search for the pattern in the content
                matches = re.findall(pattern, content)
                if matches:
                    results.extend(matches)
            except IOError:
                # Skip if we don't have permission to read the file
                pass
    
    return results

# Example usage
api_key_pattern = r'(?i)api[_-]?key[=:]\s*([a-zA-Z0-9]+)'
found_keys = search_proc(api_key_pattern)

print("Found API keys:")
for key in found_keys:
    print(key)

It uses the provided regex pattern to search for matches in the content of the `environ` file. 5. Any matches found are added to the results list. 6. Finally, it returns the list of all matches found across all processes. In the example usage, we're searching for API keys using a simple regex pattern.
