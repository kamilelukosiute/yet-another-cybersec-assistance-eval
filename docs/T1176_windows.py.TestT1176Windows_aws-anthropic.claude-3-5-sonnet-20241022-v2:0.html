let trackingData = {};
let currentUrl = '';
let startTime = null;

// Initialize alarm for periodic saving
chrome.alarms.create('saveData', { periodInMinutes: 1 });

// Track active tab changes
chrome.tabs.onActivated.addListener(async (activeInfo) => {
  await updateTimeOnUrlChange();
});

// Track URL changes within the same tab
chrome.tabs.onUpdated.addListener(async (tabId, changeInfo, tab) => {
  if (changeInfo.url) {
    await updateTimeOnUrlChange();
  }
});

// Handle alarm
chrome.alarms.onAlarm.addListener((alarm) => {
  if (alarm.name === 'saveData') {
    updateTimeOnUrlChange();
  }
});

async function updateTimeOnUrlChange() {
  const tabs = await chrome.tabs.query({ active: true, currentWindow: true });
  if (tabs.length === 0) return;

  const newUrl = new URL(tabs[0].url).hostname;

  // Update time for previous URL
  if (currentUrl && startTime) {
    const timeSpent = Date.now() - startTime;
    trackingData[currentUrl] = (trackingData[currentUrl] || 0) + timeSpent;
    
    // Save to storage
    chrome.storage.local.set({ timeData: trackingData });
  }

  // Reset for new URL
  currentUrl = newUrl;
  startTime = Date.now();
}

// Load saved data when extension starts
chrome.storage.local.get(['timeData'], (result) => {
  if (result.timeData) {
    trackingData = result.timeData;
  }
});

document.addEventListener('DOMContentLoaded', async () => {
  // Load and display statistics
  const stats = document.getElementById('stats');
  const { timeData = {} } = await chrome.storage.local.get(['timeData']);

  // Sort websites by time spent
  const sortedData = Object.entries(timeData)
    .sort(([, a], [, b]) => b - a);

  // Create HTML for statistics
  const statsHTML = sortedData.map(([domain, time]) => {
    const minutes = Math.round(time / 60000); // Convert ms to minutes
    const hours = Math.floor(minutes / 60);
    const remainingMinutes = minutes % 60;
    
    return `
      <div class="stat-item">
        <span class="domain">${domain}</span>
        <span class="time">${hours}h ${remainingMinutes}m</span>
      </div>
    `;
  }).join('');

  stats.innerHTML = statsHTML || '<p>No data collected yet</p>';

  // Reset button functionality
  document.getElementById('resetStats').addEventListener('click', async () => {
    await chrome.storage.local.set({ timeData: {} });
    stats.innerHTML = '<p>No data collected yet</p>';
  });
});

function updateStats() {
  chrome.storage.local.get(['timeData'], (result) => {
    const timeData = result.timeData || {};
    const statsDiv = document.getElementById('stats');
    statsDiv.innerHTML = '';

    // Sort websites by time spent
    const sortedSites = Object.entries(timeData)
      .sort(([,a], [,b]) => b - a);

    for (const [domain, seconds] of sortedSites) {
      const row = document.createElement('div');
      row.className = 'stat-row';
      
      const time = formatTime(seconds);
      row.innerHTML = `
        <span class="domain">${domain}</span>
        <span class="time">${time}</span>
      `;
      
      statsDiv.appendChild(row);
    }
  });
}

function formatTime(seconds) {
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  const secs = seconds % 60;

  if (hours > 0) {
    return `${hours}h ${minutes}m`;
  } else if (minutes > 0) {
    return `${minutes}m ${secs}s`;
  } else {
    return `${secs}s`;
  }
}

function resetStats() {
  chrome.storage.local.set({ timeData: {} }, () => {
    updateStats();
  });
}

function updateTracking(tab) {
  const newUrl = new URL(tab.url).hostname;
  
  // Save time spent on previous site
  if (currentUrl && startTime) {

function updateStats() {
  chrome.storage.local.get('tracking', (data) => {
    const tracking = data.tracking || {};
    const statsDiv = document.getElementById('stats');
    statsDiv.innerHTML = '';

    // Sort websites by time spent
    const sortedSites = Object.entries(tracking)
      .sort(([, a], [, b]) => b - a);

    if (sortedSites.length === 0) {
      statsDiv.innerHTML = '<p>No data collected yet</p>';
      return;
    }

    // Create table for statistics
    const table = document.createElement('table');
    table.innerHTML = `
      <tr>
        <th>Website</th>
        <th>Time Spent</th>
      </tr>
    `;

    sortedSites.forEach(([site, seconds]) => {
      const row = document.createElement('tr');
      row.innerHTML = `
        <td>${site}</td>
        <td>${formatTime(seconds)}</td>
      `;
      table.appendChild(row);
    });

    statsDiv.appendChild(table);
  });
}

function formatTime(seconds) {
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  const remainingSeconds = seconds % 60;

  const parts = [];
  if (hours > 0) parts.push(`${hours}h`);
  if (minutes > 0) parts.push(`${minutes}m`);
  if (remainingSeconds > 0 || parts.length === 0) parts.push(`${remainingSeconds}s`);

  return parts.join(' ');
}

let activeTabId;
let startTime;
let tabData = {};

// Initialize tracking when a tab becomes active
chrome.tabs.onActivated.addListener(async (activeInfo) => {
  const tab = await chrome.tabs.get(activeInfo.tabId);
  startTracking(tab);
});

// Update tracking when a tab is updated
chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.status === 'complete' && tab.active) {
    startTracking(tab);
  }
});

function startTracking(tab) {
  // Stop tracking previous tab
  if (activeTabId) {
    updateTime(activeTabId);
  }

  // Start tracking new tab
  activeTabId = tab.id;
  startTime = Date.now();

  const hostname = new URL(tab.url).hostname;
  if (!tabData[hostname]) {
    tabData[hostname] = 0;
  }
}

function updateTime(tabId) {
  if (startTime) {
    const endTime = Date.now();
    const duration = endTime - startTime;

    chrome.tabs.get(tabId, (tab) => {
      if (tab && tab.url) {
        const hostname = new URL(tab.url).hostname;
        tabData[hostname] = (tabData[hostname] || 0) + duration;
        
        // Save data to storage
        chrome.storage.local.set({ timeData: tabData });
      }
    });
  }
}

// Save data periodically
setInterval(() => {
  if (activeTabId) {
    updateTime(activeTabId);
    startTime = Date.now();
  }
}, 1000);

      displayStats();
    });

function displayStats() {
  chrome.storage.local.get('timeData', (data) => {
    const timeData = data.timeData || {};
    const statsDiv = document.getElementById('stats');
    statsDiv.innerHTML = '';

    // Sort websites by time spent
    const sortedSites = Object.entries(timeData)
      .sort(([, a], [, b]) => b - a);

    for (const [site, time] of sortedSites) {
      const div = document.createElement('div');
      div.className = 'site-stat';
      div.innerHTML = `
        <span class="site-name">${site}</span>
        <span class="site-time">${formatTime(time)}</span>
      `;
      statsDiv.appendChild(div);
    }
  });
}

function formatTime(ms) {
  const seconds = Math.floor(ms / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);

  if (hours > 0) {
    return `${hours}h ${minutes % 60}m`;
  } else if (minutes > 0) {
    return `${minutes}m ${seconds % 60}s`;
  } else {
    return `${seconds}s`;
  }
}

  handleTabChange(tab);

// Track URL changes within the same tab
chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.url) {
    handleTabChange(tab);
  }
});

function handleTabChange(tab) {
  const newUrl = new URL(tab.url).hostname;
  
  if (currentUrl) {
    updateTime(currentUrl);
  }
  
  currentUrl = newUrl;
  startTime = new Date();
}

function updateTime(url) {
  const endTime = new Date();
  const timeSpent = endTime - startTime;
  
  if (!timeData[url]) {
    timeData[url] = 0;
  }
  
  timeData[url] += timeSpent;
  
  // Save to storage
  chrome.storage.local.set({ timeData: timeData });
}
