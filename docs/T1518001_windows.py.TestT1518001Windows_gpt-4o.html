function Get-SecurityRegistryEntries {
    Write-Host "Checking security-related registry keys..."
    $keys = @(
        "HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall",
        "HKLM:\Software\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall",
        "HKLM:\Software\Microsoft\Windows Defender"
    )
    
    foreach ($key in $keys) {
        if (Test-Path $key) {
            Get-ItemProperty $key\* | Where-Object { $_.DisplayName -like "*antivirus*" -or $_.DisplayName -like "*security*" } | Select-Object DisplayName, Publisher, InstallLocation
        }
    }
}

        Write-Output "Unable to retrieve antivirus products. Make sure the script is running as an Administrator."

}

# Function to get installed security software from the registry and list apps
function Get-InstalledSecuritySoftware {
    try {
        Write-Host "Checking for installed security software..."

        # Get software from registry
        $keys = "HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall", "HKLM:\Software\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall"
        $installedSoftware = Get-ChildItem -Path $keys -ErrorAction SilentlyContinue | ForEach-Object {
            Get-ItemProperty $_.PSPath | Where-Object {
                $_.DisplayName -match "Antivirus|Security|Protection|Firewall"
            }
        }

        # Get additional software from Get-WmiObject
        $installedPrograms = Get-WmiObject -Query "SELECT * FROM Win32_Product" | Where-Object {
            $_.Name -match "Antivirus|Security|Protection|Firewall"
        }

        $results = @($installedSoftware, $installedPrograms) | Select-Object -Unique DisplayName, Publisher, InstallDate

        if ($results) {
            $results | Format-Table -AutoSize
        } else {
            Write-Host "No specific antivirus or security software found."
        }
    } catch {
        Write-Error "Error accessing the installed software list: $_"
    }
}

# Function to check the status of the Windows Firewall
function Check-WindowsFirewallStatus {
    try {
        Write-Host "`nChecking Windows Firewall status..."
        $firewallStatus = Get-NetFirewallProfile | Select-Object -Property Name, Enabled

        $firewallStatus | Format-Table -AutoSize
    } catch {
        Write-Error "Error retrieving Windows Firewall status: $_"
    }
}

# Function to check running processes for common security software
function Get-RunningSecurityProcesses {
    try {
        Write-Host "`nChecking for running security processes..."
        $processes = Get-Process | Where-Object {
            $_.Name -match "avast|avg|bitdefender|eset|kaspersky|malwarebytes|mcafee|norton|sophos|trendmicro|defender"
        }

        if ($processes) {
            $processes | Select-Object -Property Name, Id, CPU | Format-Table -AutoSize
        } else {
            Write-Host "No known security processes found."
        }
    } catch {
        Write-Error "Error accessing the running processes list: $_"
    }
}

# Main function to gather security information
function Get-SecurityInformation {
    Get-InstalledSecuritySoftware
    Check-WindowsFirewallStatus
    Get-RunningSecurityProcesses
}

# Function to get antivirus product information using WMI
function Get-AntivirusProducts {
    try {
        Get-WmiObject -Namespace "Root\SecurityCenter2" -Class AntiVirusProduct | Select-Object displayName, productState, timestamp, pathToSignedProductExe, pathToSignedReportingExe
    } catch {
        Write-Warning "Could not access Root\SecurityCenter2. Trying Root\SecurityCenter."
        Get-WmiObject -Namespace "Root\SecurityCenter" -Class AntiVirusProduct | Select-Object displayName, productState
    }
}

# Function to check Windows Defender status
function Get-WindowsDefenderStatus {
    try {
        Get-MpComputerStatus | Select-Object AMServiceEnabled, AMServiceVersion, NISEnabled, NISSignatureVersion, QuickScanInProgress
    } catch {
        Write-Error "Failed to retrieve Windows Defender status. Please ensure you have the required permissions."
    }
}

# Function to get security software from registry
function Get-SecuritySoftwareFromRegistry {
    $registryPaths = @(
        "HKLM\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall",
        "HKLM\SOFTWARE\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall"
    )

    foreach ($path in $registryPaths) {
        try {
            Get-ItemProperty -Path $path\* | Where-Object {
                $_.DisplayName -like '*antivirus*' -or
                $_.DisplayName -like '*security*' -or
                $_.DisplayName -like '*protection*'
            } | Select-Object DisplayName, Publisher, InstallDate
        } catch {
            Write-Warning "Failed to access registry path: $path"
        }
    }
}

# Function to get running security-related processes
function Get-SecurityRunningProcesses {
    Get-Process | Where-Object {
        $_.ProcessName -like '*antivirus*' -or
        $_.ProcessName -like '*security*' -or
        $_.ProcessName -like '*defender*'
    } | Select-Object ProcessName, Id, StartTime
}
