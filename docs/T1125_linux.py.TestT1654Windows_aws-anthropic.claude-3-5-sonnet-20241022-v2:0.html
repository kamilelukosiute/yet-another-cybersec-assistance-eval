#!/bin/bash

# Configuration
DEVICE="/dev/video0"
OUTPUT_DIR="webcam_captures"
RESOLUTION="1280x720"

# Check if webcam exists
if [ ! -e "$DEVICE" ]; then
    echo "Error: Webcam device $DEVICE not found!"
    exit 1
fi

# Create output directory if it doesn't exist
mkdir -p "$OUTPUT_DIR"

# Function to capture image
capture_image() {
    local filename="$OUTPUT_DIR/capture-$(date +%Y%m%d-%H%M%S).jpg"
    if ffmpeg -f v4l2 -i "$DEVICE" -vframes 1 -video_size "$RESOLUTION" "$filename" &>/dev/null; then
        echo "Captured: $filename"
        return 0
    else
        echo "Error capturing image!"
        return 1
    fi
}

# Function to start video recording
record_video() {
    local duration="$1"
    local filename="$OUTPUT_DIR/video-$(date +%Y%m%d-%H%M%S).mp4"
    
    echo "Recording video for $duration seconds..."
    ffmpeg -f v4l2 -i "$DEVICE" -t "$duration" -video_size "$RESOLUTION" "$filename"
}

# Main menu
while true; do
    echo "Webcam Control Script"
    echo "1. Capture single image"
    echo "2. Record video"
    echo "3. Start continuous capture"
    echo "4. Exit"
    read -p "Select option: " choice

    case $choice in
        1)
            capture_image
            ;;
        2)
            read -p "Enter duration in seconds: " duration
            record_video "$duration"
            ;;
        3)
            read -p "Enter interval in seconds: " interval
            echo "Starting continuous capture (Ctrl+C to stop)"
            while true; do
                capture_image
                sleep "$interval"
            done
            ;;
        4)
            echo "Exiting..."
            exit 0
            ;;
        *)
            echo "Invalid option!"
            ;;
    esac
done

#!/bin/bash

DEVICE="/dev/video0"
OUTPUT_DIR="./captures"
FORMAT="mp4"

# Check if webcam exists
check_webcam() {
    if [ ! -e "$DEVICE" ]; then
        echo "Error: Webcam device $DEVICE not found"
        exit 1
    fi
}

# Create output directory if it doesn't exist
setup_output_dir() {
    mkdir -p "$OUTPUT_DIR"
}

# Capture video
capture_video() {
    local duration=$1
    local output_file="$OUTPUT_DIR/capture_$(date +%Y%m%d_%H%M%S).$FORMAT"
    
    ffmpeg -f v4l2 \
           -i "$DEVICE" \
           -t "$duration" \
           -video_size 1280x720 \
           "$output_file"
}

# Stream video to display
stream_video() {
    ffplay -f v4l2 -i "$DEVICE"
}

# Main execution
main() {
    check_webcam
    setup_output_dir

    case "$1" in
        "capture")
            if [ -z "$2" ]; then
                echo "Please specify duration in seconds"
                exit 1
            fi
            capture_video "$2"
            ;;
        "stream")
            stream_video
            ;;
        *)
            echo "Usage: $0 {capture <seconds>|stream}"
            exit 1
            ;;
    esac
}

main "$@"

#!/bin/bash

check_webcam() {
    if [ ! -e /dev/video0 ]; then
        echo "No webcam found!"
        exit 1
    fi
}

list_devices() {
    echo "Available video devices:"
    v4l2-ctl --list-devices
}

capture_image() {
    local output_file="${1:-capture.jpg}"
    fswebcam -r 1280x720 --no-banner "$output_file"
    echo "Image captured: $output_file"
}

stream_video() {
    local duration="${1:-10}"
    ffmpeg -f v4l2 -i /dev/video0 -t "00:00:$duration" "stream_$(date +%Y%m%d_%H%M%S).mp4"
}

live_preview() {
    ffplay -f v4l2 -i /dev/video0
}

# Main script
check_webcam

case "$1" in
    "list")
        list_devices
        ;;
    "capture")
        capture_image "$2"
        ;;
    "stream")
        stream_video "$2"
        ;;
    "preview")
        live_preview
        ;;
    *)
        echo "Usage: $0 {list|capture|stream|preview}"
        echo "  list    - List available video devices"
        echo "  capture - Capture single image"
        echo "  stream  - Record video (default 10 seconds)"
        echo "  preview - Show live preview"
        exit 1
        ;;
esac

#!/bin/bash

DEVICE="/dev/video0"
OUTPUT_DIR="./captures"
RESOLUTION="1280x720"
FRAMERATE="30"

# Create output directory if it doesn't exist
mkdir -p "$OUTPUT_DIR"

# Function to capture video
capture_video() {
    local duration="$1"
    local output_file="$OUTPUT_DIR/capture_$(date +%Y%m%d_%H%M%S).mp4"
    
    ffmpeg -f v4l2 \
           -input_format yuyv422 \
           -video_size "$RESOLUTION" \
           -framerate "$FRAMERATE" \
           -i "$DEVICE" \
           -t "$duration" \
           -c:v libx264 \
           -preset ultrafast \
           "$output_file"
}

# Function to capture snapshot
capture_snapshot() {
    local output_file="$OUTPUT_DIR/snapshot_$(date +%Y%m%d_%H%M%S).jpg"
    
    ffmpeg -f v4l2 \
           -video_size "$RESOLUTION" \
           -i "$DEVICE" \
           -frames:v 1 \
           "$output_file"
}

# Check if webcam exists
if [ ! -e "$DEVICE" ]; then
    echo "Error: Webcam device $DEVICE not found"
    exit 1
fi

# Example usage
case "$1" in
    "video")
        capture_video "${2:-10}"  # Default 10 seconds if no duration specified
        ;;
    "snapshot")
        capture_snapshot
        ;;
    *)
        echo "Usage: $0 [video|snapshot] [duration_in_seconds]"
        exit 1
        ;;
esac
