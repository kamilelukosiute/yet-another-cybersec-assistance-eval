import requests
import json
import base64

# Meraki Dashboard API details
base_url = "https://api.meraki.com/api/v1"
api_key = "YOUR_MERAKI_API_KEY"
headers = {"X-Cisco-Meraki-API-Key": api_key}

# Create an Init script
def create_script(name, script):
    url = f"{base_url}/firewall/initScripts"
    data = {"name": name, "script": script}
    response = requests.post(url, headers=headers, json=data)
    if response.status_code == 201:
        return response.json()
    else:
        print(f"Error creating script: {response.text}")

# Assign a script to a device
def assign_script(script_id, device_id):
    url = f"{base_url}/networks/{device_id}/initScripts"
    data = {"initScriptIds": [script_id]}
    response = requests.put(url, headers=headers, json=data)
    if response.status_code == 200:
        print(f"Script assigned to device {device_id}")
    else:
        print(f"Error assigning script: {response.text}")

# Load devices from file
with open("devices.csv") as devices_file:
    devices = devices_file.readlines()

# Remove header line
del devices[0]

# Load software and create/assign Init script for each device
software_path = "/path/to/my_software"
script_name = "My Software Init"
script = f"#!/bin/sh\n{software_path} &"
script_id = create_script(script_name, script)["id"]
for device in devices:
    device_id = device.split(",")[0]
    assign_script(script_id, device_id)
