import os
import plistlib

def create_launch_agent(app_path, label):
    # Create the plist content
    plist_dict = {
        'Label': label,
        'ProgramArguments': [app_path],
        'RunAtLoad': True,
        'KeepAlive': True
    }
    
    # Define the launch agent path
    launch_agent_path = os.path.expanduser(f'~/Library/LaunchAgents/{label}.plist')
    
    # Write the plist file
    with open(launch_agent_path, 'wb') as f:
        plistlib.dump(plist_dict, f)
    
    # Set proper permissions
    os.chmod(launch_agent_path, 0o644)
    
    # Load the launch agent
    os.system(f'launchctl load {launch_agent_path}')

import os
import plistlib
import subprocess
import sys

class AutoStartManager:
    def __init__(self, app_path, app_identifier):
        self.app_path = app_path
        self.app_identifier = app_identifier
        
    def setup_launch_agent(self):
        """Setup a Launch Agent for the application"""
        plist_dict = {
            'Label': self.app_identifier,
            'ProgramArguments': [self.app_path],
            'RunAtLoad': True,
            'KeepAlive': True
        }
        
        launch_agent_path = os.path.expanduser(f'~/Library/LaunchAgents/{self.app_identifier}.plist')
        
        try:
            with open(launch_agent_path, 'wb') as f:
                plistlib.dump(plist_dict, f)
            os.chmod(launch_agent_path, 0o644)
            subprocess.run(['launchctl', 'load', launch_agent_path], check=True)
            return True
        except Exception as e:
            print(f"Error setting up Launch Agent: {e}")
            return False
    
    def add_login_item(self):
        """Add application to Login Items"""
        script = f'''
        tell application "System Events"
            make new login item at end with properties {{path:"{self.app_path}", hidden:false}}
        end tell
        '''
        
        try:
            subprocess.run(['osascript', '-e', script], check=True)
            return True
        except Exception as e:
            print(f"Error adding Login Item: {e}")
            return False
    
    def verify_installation(self):
        """Verify that auto-start is properly configured"""
        # Check Launch Agent
        launch_agent_path = os.path.expanduser(f'~/Library/LaunchAgents/{self.app_identifier}.plist')
        launch_agent_exists = os.path.exists(launch_agent_path)
        
        # Check Login Items
        check_script = f'''
        tell application "System Events"
            return exists login item "{os.path.basename(self.app_path)}"
        end tell
        '''
        login_item_exists = subprocess.run(['osascript', '-e', check_script], 
                                         capture_output=True, text=True).stdout.strip() == "true"
        
        return {
            'launch_agent': launch_agent_exists,
            'login_item': login_item_exists
        }

def main():
    # Example usage
    app_path = '/Applications/YourApp.app'
    app_identifier = 'com.yourapp.startup'
    
    manager = AutoStartManager(app_path, app_identifier)
    
    # Setup both methods
    launch_agent_success = manager.setup_launch_agent()
    login_item_success = manager.add_login_item()
    
    # Verify installation
    verification = manager.verify_installation()
    
    print("Installation Results:")
    print(f"Launch Agent: {'Success' if launch_agent_success else 'Failed'}")
    print(f"Login Item: {'Success' if login_item_success else 'Failed'}")
    print("\nVerification Results:")
    print(f"Launch Agent exists: {verification['launch_agent']}")
    print(f"Login Item exists: {verification['login_item']}")

if __name__ == "__main__":
    main()

import os
import plistlib
import subprocess

def add_to_startup(app_path, method='launch_agent'):
    """
    Add application to startup using specified method
    Methods: 'launch_agent', 'login_item', 'both'
    """
    
    def create_launch_agent():
        label = f"com.{os.path.basename(app_path).lower().replace(' ', '')}"
        launch_agent_dir = os.path.expanduser('~/Library/LaunchAgents')
        plist_path = os.path.join(launch_agent_dir, f'{label}.plist')
        
        if not os.path.exists(launch_agent_dir):
            os.makedirs(launch_agent_dir)
        
        plist_content = {
            'Label': label,
            'ProgramArguments': [app_path],
            'RunAtLoad': True,
            'KeepAlive': True
        }
        
        with open(plist_path, 'wb') as f:
            plistlib.dump(plist_content, f)
        
        os.chmod(plist_path, 0o644)
        subprocess.run(['launchctl', 'load', plist_path])
        return True
    
    def add_login_item():
        script = f'''
        tell application "System Events"
            make login item at end with properties {{path:"{app_path}", hidden:false}}
        end tell
        '''
        try:
            subprocess.run(['osascript', '-e', script])
            return True
        except subprocess.CalledProcessError:
            return False
    
    if method == 'launch_agent':
        return create_launch_agent()
    elif method == 'login_item':
        return add_login_item()
    elif method == 'both':
        return create_launch_agent() and add_login_item()
    else:
        raise ValueError("Invalid method specified")

def remove_from_startup(app_path, method='launch_agent'):
    """
    Remove application from startup
    """
    
    def remove_launch_agent():
        label = f"com.{os.path.basename(app_path).lower().replace(' ', '')}"
        plist_path = os.path.expanduser(f'~/Library/LaunchAgents/{label}.plist')
        
        if os.path.exists(plist_path):
            subprocess.run(['launchctl', 'unload', plist_path])
            os.remove(plist_path)
            return True
        return False
    
    def remove_login_item():
        script = f'''
        tell application "System Events"
            delete (every login item whose path is "{app_path}")
        end tell
        '''
        try:
            subprocess.run(['osascript', '-e', script])
            return True
        except subprocess.CalledProcessError:
            return False
    
    if method == 'launch_agent':
        return remove_launch_agent()
    elif method == 'login_item':
        return remove_login_item()
    elif method == 'both':
        return remove_launch_agent() and remove_login_item()
    else:
        raise ValueError("Invalid method specified")

#!/usr/bin/osascript

tell application "System Events"
    set login_items_list to every login item
    set app_path to "/path/to/your/application"
    
    -- Check if app is already in login items
    set app_exists to false
    repeat with login_item in login_items_list
        if (path of login_item as string) contains app_path then
            set app_exists to true
        end if
    end repeat
    
    -- Add app to login items if it doesn't exist
    if app_exists is false then
        make new login item at end with properties {path:app_path, hidden:false}
    end if
end tell

import subprocess

def add_to_login_items(app_path):
    script = f'''
    tell application "System Events"
        make new login item at end with properties {{path:"{app_path}", hidden:false}}
    end tell
    '''
    
    try:
        subprocess.run(['osascript', '-e', script], check=True)
        print(f"Successfully added {app_path} to login items")
    except subprocess.CalledProcessError as e:
        print(f"Error adding to login items: {e}")

# Example usage
app_path = "/Applications/YourApp.app"

import os
import plistlib
import subprocess
from pathlib import Path

class AutoStartManager:
    def __init__(self, app_path, app_identifier):
        self.app_path = app_path
        self.app_identifier = app_identifier
        self.launch_agent_path = os.path.expanduser(f'~/Library/LaunchAgents/{app_identifier}.plist')

    def create_launch_agent(self):
        """Create and load a Launch Agent"""
        launch_agent = {
            'Label': self.app_identifier,
            'ProgramArguments': [self.app_path],
            'RunAtLoad': True,
            'KeepAlive': True
        }
        
        try:
            with open(self.launch_agent_path, 'wb') as f:
                plistlib.dump(launch_agent, f)
            os.chmod(self.launch_agent_path, 0o644)
            subprocess.run(['launchctl', 'load', self.launch_agent_path], check=True)
            return True
        except Exception as e:
            print(f"Error creating launch agent: {e}")
            return False

    def add_to_login_items(self):
        """Add application to Login Items using AppleScript"""
        script = f'''
        tell application "System Events"
            make new login item at end with properties {{path:"{self.app_path}", hidden:false}}
        end tell
        '''
        
        try:
            subprocess.run(['osascript', '-e', script], check=True)
            return True
        except subprocess.CalledProcessError as e:
            print(f"Error adding to login items: {e}")
            return False

    def remove_from_startup(self):
        """Remove application from startup items"""
        # Remove Launch Agent if exists
        if os.path.exists(self.launch_agent_path):
            try:
                subprocess.run(['launchctl', 'unload', self.launch_agent_path], check=True)
                os.remove(self.launch_agent_path)
            except Exception as e:
                print(f"Error removing launch agent: {e}")

        # Remove from Login Items
        script = f'''
        tell application "System Events"
            set login_items_list to every login item
            repeat with login_item in login_items_list
                if (path of login_item as string) contains "{self.app_path}" then
                    delete login_item
                end if
            end repeat
        end tell
        '''
        
        try:
            subprocess.run(['osascript', '-e', script], check=True)
        except subprocess.CalledProcessError as e:
            print(f"Error removing from login items: {e}")

def main():
    # Example usage
    app_path = "/Applications/YourApp.app"
    app_identifier = "com.yourapp.launcher"
    
    manager = AutoStartManager(app_path, app_identifier)
    
    # Add to startup using both methods
    manager.create_launch_agent()
    manager.add_to_login_items()
    
    # To remove from startup
    # manager.remove_from_startup()

if __name__ == "__main__":
    main()

import os
import plistlib

def create_launch_agent(app_path, label):
    # Define the path for the launch agent
    launch_agent_dir = os.path.expanduser('~/Library/LaunchAgents')
    plist_path = os.path.join(launch_agent_dir, f'{label}.plist')
    
    # Create the plist content
    plist_content = {
        'Label': label,
        'ProgramArguments': [app_path],
        'RunAtLoad': True,
        'KeepAlive': True
    }
    
    # Ensure the LaunchAgents directory exists
    os.makedirs(launch_agent_dir, exist_ok=True)
    
    # Write the plist file
    with open(plist_path, 'wb') as f:
        plistlib.dump(plist_content, f)
    
    # Set correct permissions
    os.chmod(plist_path, 0o644)
    
    # Load the launch agent
    os.system(f'launchctl load {plist_path}')

# Usage example
create_launch_agent('/Applications/YourApp.app/Contents/MacOS/YourApp', 'com.your.app')

import os
import sys
import plistlib
import subprocess
from pathlib import Path

class AutoLauncher:
    def __init__(self, app_path, app_identifier):
        self.app_path = os.path.abspath(app_path)
        self.app_identifier = app_identifier
        
    def setup_launch_agent(self):
        """Set up a Launch Agent for the application"""
        launch_agent_dir = os.path.expanduser('~/Library/LaunchAgents')
        plist_path = os.path.join(launch_agent_dir, f'{self.app_identifier}.plist')
        
        plist_content = {
            'Label': self.app_identifier,
            'ProgramArguments': [self.app_path],
            'RunAtLoad': True,
            'KeepAlive': True,
            'StandardOutPath': os.path.expanduser(f'~/Library/Logs/{self.app_identifier}.log'),
            'StandardErrorPath': os.path.expanduser(f'~/Library/Logs/{self.app_identifier}_error.log')
        }
        
        os.makedirs(launch_agent_dir, exist_ok=True)
        
        with open(plist_path, 'wb') as f:
            plistlib.dump(plist_content, f)
        
        os.chmod(plist_path, 0o644)
        
        # Unload if exists and load the launch agent
        subprocess.run(['launchctl', 'unload', plist_path], capture_output=True)
        subprocess.run(['launchctl', 'load', plist_path], check=True)
        
    def add_login_item(self):
        """Add application to Login Items using AppleScript"""
        script = f'''
        tell application "System Events"
            make new login item at end with properties {{path:"{self.app_path}", hidden:false}}
        end tell
        '''
        
        subprocess.run(['osascript', '-e', script], check=True)
    
    def verify_setup(self):
        """Verify that the auto-launch is properly configured"""
        launch_agent_path = os.path.expanduser(f'~/Library/LaunchAgents/{self.app_identifier}.plist')
        
        if os.path.exists(launch_agent_path):
            print(f"Launch Agent found at: {launch_agent_path}")
            
            # Check if it's loaded
            result = subprocess.run(['launchctl', 'list'], capture_output=True, text=True)
            if self.app_identifier in result.stdout:
                print("Launch Agent is currently loaded")
            else:
                print("Warning: Launch Agent exists but is not loaded")
        else:
            print("Warning: Launch Agent not found")

def main():
    # Example usage
    app_path = "/Applications/YourApp.app"
    app_identifier = "com.your.app"
    
    if not os.path.exists(app_path):
        print(f"Error: Application not found at {app_path}")
        sys.exit(1)
    
    launcher = AutoLauncher(app_path, app_identifier)
    
    print("Setting up auto-launch...")
    
    # Setup both methods for redundancy
    try:
        launcher.setup_launch_agent()
        print("Launch Agent setup completed")
    except Exception as e:
        print(f"Error setting up Launch Agent: {e}")
    
    try:
        launcher.add_login_item()
        print("Login Item added successfully")
    except Exception as e:
        print(f"Error adding Login Item: {e}")
    
    # Verify the setup
    launcher.verify_setup()

if __name__ == "__main__":
    main()
