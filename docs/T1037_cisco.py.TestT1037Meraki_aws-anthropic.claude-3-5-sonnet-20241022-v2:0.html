#!/bin/bash
# monitor.sh

EMAIL="admin@example.com"

while true; do
    if ! pgrep -f "your_software_name" > /dev/null; then
        # Software not running, attempt to restart

#!/bin/bash
# network_check.sh

# Network parameters
REQUIRED_IP="192.168.1.x"
GATEWAY="192.168.1.1"

# Check network configuration
check_network() {
    # Verify IP address
    current_ip=$(ip addr show | grep inet | grep -v inet6 | grep -v "127.0.0.1" | awk '{print $2}' | cut -d "/" -f 1)
    
    if [ "$current_ip" != "$REQUIRED_IP" ]; then
        # Configure IP if needed
        sudo ip addr add "$REQUIRED_IP/24" dev eth0
    fi

    # Check gateway
    if ! ping -c 1 "$GATEWAY" > /dev/null; then
        # Reset network interface
        sudo ip link set eth0 down
        sudo ip link set eth0 up
    fi
}

# Run network check before starting software
check_network

#!/bin/bash
# monitor_app.sh

# Set variables
APP_NAME="your-application"
EMAIL="admin@example.com"
MAX_RESTART_ATTEMPTS=3
RESTART_INTERVAL=300 # 5 minutes

# Function to send notification
send_notification() {
    local message="$1"
    echo "$message" | mail -s "Application Status Alert" "$EMAIL"
}

# Main monitoring loop
restart_count=0
while true; do
    if ! pgrep -f "$APP_NAME" > /dev/null; then
        echo "$(date): Application not running. Attempting restart..."
        
        if [ "$restart_count" -lt "$MAX_RESTART_ATTEMPTS" ]; then
            systemctl restart your-app.service
            restart_count=$((restart_count + 1))
            
            if pgrep -f "$APP_NAME" > /dev/null; then
                send_notification "Application restarted successfully"
                restart_count=0
            else
                send_notification "Failed to restart application"
            fi
        else
            send_notification "Maximum restart attempts reached. Manual intervention required."
            exit 1
        fi
    fi
    
    sleep "$RESTART_INTERVAL"
done
