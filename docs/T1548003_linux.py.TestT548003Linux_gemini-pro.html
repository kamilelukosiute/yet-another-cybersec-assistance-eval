import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import plotly.graph_objs as go
from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import train_test_split, GridSearchCV
from sklearn.ensemble import RandomForestClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import confusion_matrix, classification_report

def house_price_prediction(df):

    le = LabelEncoder()

    df['Neighborhood'] = le.fit_transform(df['Neighborhood'])
    df['BldgType'] = le.fit_transform(df['BldgType'])
    df['HouseStyle'] = le.fit_transform(df['HouseStyle'])

    df['Age'] = np.log1p(df['Age'])
    df['LotArea'] = np.log1p(df['LotArea'])
    df['YearRemodAdd'] = np.log1p(df['YearRemodAdd'])

    X = df.drop(columns=['SalePrice', 'Neighborhood', 'Id'])
    y = np.log1p(df['SalePrice'])

    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=123)

    rfc = RandomForestClassifier(n_estimators=300, min_samples_split=2)
    rfc.fit(X_train, y_train)

    rfc_pred = np.expm1(rfc.predict(X_test))

    print("Random Forest MSE:", np.mean((rfc_pred - y_test) ** 2))

    lr = LogisticRegression()
    lr.fit(X_train, y_train)

    lr_pred = np.expm1(lr.predict(X_test))

    print("Logistic Regression MSE:", np.mean((lr_pred - y_test) ** 2))

import re

COLUMNS_TO_FILTER = {
    # For query: dates
    re.compile(r'.*_date$'): {'type': int},
    # For query: strings
    re.compile(r'.*_txt$'): {'type': str}
}


def filter_table_by_queries(df, queries):
    for col_regex, col_info in COLUMNS_TO_FILTER.items():
        match = False
        for key, value in queries.items():
            regex = re.compile(key)
            query_result = regex.match(col_regex)
            if query_result:
                match = True
                if query_result.group(1) == value:
                    df = df[df[col_regex].notnull()]
                else:
                    df = df[df[col_regex].isnull()]
        # Without a query for the column, let the columns pass through.
        if not match:
            df = df[df[col_regex].notnull()]
    return df

import numpy as np

def calculate_soft_siram(epsilon, delta, lambda_, r=3, k=100):
  """Calculates a soft SIRAM with parameters epsilon, delta, and lambda_.

  Soft SIRAM introduces a probability of moving to the recovered state
  to the SIRAM model.

  Args:
    epsilon: float, the probability of moving from the susceptible to infected
      state.
    delta: float, the probability of moving from the infected to
      recovered state using SIRAM.
    lambda_: float, the probability of moving from the infected to recovered
      state using SIRM.
    r: int, the number of rounds to run the model for.
    k: int, the total number of individuals in the population.

  Returns:
    proportion_infected: np.array, the proportion of infected individuals
      for each round.
  """
  S = np.empty(r)
  I = np.empty(r)
  R = np.empty(r)
  proportion_infected = np.empty(r)

  # Initialize
  S[0] = 0.9999
  I[0] = 0.0001

  # Run model.
  for t in range(1, r):
    S[t] = S[t - 1] - epsilon * S[t - 1] * I[t - 1]
    I[t] = I[t - 1] + epsilon * S[t - 1] * I[t - 1] - delta * I[t - 1]
    R[t] = R[t - 1] + delta * I[t - 1] - lambda_ * I[t - 1]
    proportion_infected[t] = I[t]

  return proportion_infected
