# Load the required assembly for taking screenshots
Add-Type -AssemblyName System.Windows.Forms
Add-Type -AssemblyName System.Drawing

# Set the directory to save screenshots
$saveDir = "C:\Screenshots"

# Create the directory if it doesn't exist
if (!(Test-Path -Path $saveDir)) {
    New-Item -ItemType Directory -Force -Path $saveDir
}

# Set the interval between screenshots (in seconds)
$interval = 300 # 5 minutes

# Function to take and save a screenshot
function Take-Screenshot {
    $screen = [System.Windows.Forms.Screen]::PrimaryScreen.Bounds
    $bitmap = New-Object System.Drawing.Bitmap $screen.Width, $screen.Height
    $graphics = [System.Drawing.Graphics]::FromImage($bitmap)
    $graphics.CopyFromScreen($screen.Location, [System.Drawing.Point]::Empty, $screen.Size)

    $timestamp = Get-Date -Format "yyyyMMdd-HHmmss"
    $filename = Join-Path $saveDir "screenshot_$timestamp.png"
    $bitmap.Save($filename, [System.Drawing.Imaging.ImageFormat]::Png)

    $graphics.Dispose()
    $bitmap.Dispose()

    Write-Host "Screenshot saved: $filename"
}

# Main loop
while ($true) {
    Take-Screenshot
    Start-Sleep -Seconds $interval
}
Here's how the script works: 1. It loads the necessary .NET assemblies for taking screenshots. 2. It sets the directory where screenshots will be saved (`$saveDir`). You can modify this to your preferred location. 3. It creates the save directory if it doesn't already exist. 4. The interval between screenshots is set to 5 minutes (300 seconds). You can adjust this by changing the `$interval` value. 5. The `Take-Screenshot` function captures the primary screen, creates a bitmap image, and saves it as a PNG file with a timestamp in the filename. 6. The main loop runs indefinitely, taking a screenshot and then waiting for the specified interval before taking the next one. To use this script: 1. Copy the script into a new file with a `.ps1` extension (e.g., `screenshot.ps1`). 2. Open PowerShell as an administrator. 3. Navigate to the directory containing the script. 4. Run the script by typing `.\screenshot.ps1`. The script will continue running and taking screenshots until you stop it manually (e.g., by pressing Ctrl+C in the PowerShell window). Note: Make sure you have the necessary permissions to write to the specified save directory and to capture screenshots on your system.
