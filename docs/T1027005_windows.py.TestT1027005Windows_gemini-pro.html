import win32evtlog
import win32api

def delete_log_entries(log_name, process_name, pid=None):
    """
    Deletes log entries from the specified log that relate to a specific process name or PID.

    Args:
        log_name (str): The name of the log to search.
        process_name (str): The name of the process to search for.
        pid (int, optional): The PID of the process to search for.
    """

    # Open the log file with write access
    handle = win32evtlog.OpenEventLog(None, log_name)
    win32evtlog.WriteEventLog(handle, win32evtlog.EVENTLOG_INFORMATION_TYPE, 0, 0, 'Cleaning up log entries')

    # Get the number of log entries
    num_entries = win32evtlog.GetNumberOfEventLogRecords(handle)

    # Loop through each log entry
    for i in range(num_entries):

        # Get the log entry
        entry = win32evtlog.ReadEventLog(handle, i)

        # Check if the log entry is related to the specified process
        if (process_name and process_name in entry['SourceName']) or (pid and pid == entry['EventID']):

            # Delete the log entry
            win32evtlog.ClearEventLog(handle, entry['Logfile'])

    # Close the log file
    win32evtlog.CloseEventLog(handle)

import os
import subprocess

def delete_process_logs(process_name_or_pid):
    """
    Search Windows log files for events that relate to a specific process name or PID and then deletes those entries.

    Args:
        process_name_or_pid (str): The name or PID of the process to search for.
    """

    # Get the current date and time in the format YYYY-MM-DD HH:MM:SS.
    current_date_time = subprocess.check_output("date /T", shell=True).decode("utf-8").strip()

    # Get the list of log files.
    log_files = os.listdir("C:\\Windows\\System32\\winevt\\Logs")

    # Iterate over the log files.
    for log_file in log_files:

        # Get the list of events in the log file.
        events = subprocess.check_output(f"wevtutil qe {log_file} /rd:true /f:text /query:\"*[EventData[Data[@Name='ProcessName'] = '{process_name_or_pid}']]\"", shell=True).decode("utf-8").strip()

        # Iterate over the events.
        for event in events.split("\n"):

            # Get the event ID.
            event_id = event.split("Event ID: ")[1].split(" [")[0]

            # Delete the event.
            subprocess.call(f"wevtutil delevent /log:{log_file} /id:{event_id}", shell=True)

            # Print a message to indicate that the event was deleted.
            print(f"Deleted event {event_id} from log file {log_file}.")

# Search for events related to the specified process
events = subprocess.check_output(["wevtutil", "query-events", log_path, "/q:{EventID=4688}", "/f:Text", "/r:*", "/n:*"]).decode("utf-8")

    if search_term in event:
        event_ids.append(event.split("EventID=")[1].split(",")[0])

# Delete the events
for event_id in event_ids:
    subprocess.call(["wevtutil", "delete", log_path, "/e:{}".format(event_id)])

import os
import re

# Get the process name or PID to search for
process_name_or_pid = input("Enter the process name or PID to search for: ")

# Get the log files to search
log_files = input("Enter the log files to search (comma separated): ")

# Convert the log files to a list
log_files = log_files.split(",")

# Search the log files for events that relate to the process name or PID
for log_file in log_files:
    with open(log_file, "r") as f:
        log_data = f.read()
        
        # Get the events that relate to the process name or PID
        events = re.findall(r".*Process Name: (?P<process_name>.+)\s+Process ID: (?P<process_id>\d+).*", log_data)

        # Delete the events that relate to the process name or PID
        with open(log_file, "w") as f:
            for event in events:
                if event.group("process_name") == process_name_or_pid or event.group("process_id") == process_name_or_pid:
                    log_data = log_data.replace(event.group(0), "")
    
            f.write(log_data)
