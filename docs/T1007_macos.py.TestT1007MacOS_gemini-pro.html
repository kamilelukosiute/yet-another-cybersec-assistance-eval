import subprocess
import tarfile

services = subprocess.check_output(['launchctl', 'list']).decode('utf-8').splitlines()

# Parse the output and get the details of each service

# Write the service details to a text file

import subprocess
import tempfile
import tarfile

# Get a list of all registered local system services
services = subprocess.check_output(["/usr/bin/systemsetup", "-getlocalservices"])

# Convert the output to a list of strings
services = services.decode("utf-8").split("\n")

# Remove any empty strings from the list
services = [service for service in services if service]

# Create a temporary file to store the service details
with tempfile.NamedTemporaryFile(dir="/tmp", delete=False) as temp_file:
    # Write the service details to the file
    temp_file.write("Local System Services:\n\n".encode("utf-8"))
    for service in services:
        details = subprocess.check_output(["/usr/bin/systemsetup", "-getlocalservice", service])

        temp_file.write("\n".encode("utf-8"))

# Compress the temporary file using tar.gz
with tarfile.open("/tmp/services.tar.gz", "w:gz") as tar:
    tar.add(temp_file.name, arcname="services.txt")

# Create a temporary file to store the information.

import os
import platform
import subprocess
import tarfile
import tempfile

if platform.system() != 'Darwin':
    raise Exception('This script is only designed for macOS systems.')

# Define a non-obvious file name for storing the service details
filename = '.tmp_macOS_services.txt'

# Enumerate all registered local system services and their details
command = ["launchctl", "list", "-x", "-F"]
process = subprocess.Popen(command, stdout=subprocess.PIPE)
output, _ = process.communicate()

# Parse the output and extract the service details
services = {}
for line in output.decode('utf-8').splitlines():
    if line.startswith('Service'):
        tokens = line.split()
        service_name = tokens[2]
        services[service_name] = {}

# Gather additional details for each service
for service_name in services:
    command = ["launchctl", "get", f"{service_name}.plist"]
    process = subprocess.Popen(command, stdout=subprocess.PIPE)
    output, _ = process.communicate()
    services[service_name]['plist'] = output.decode('utf-8')

# Write the service details to a text file
with open(os.path.join('/tmp', filename), 'w') as f:
    for service_name, details in services.items():
        f.write(f"Service Name: {service_name}\n")
        f.write(f"Service plist:\n{details['plist']}\n\n")

# Compress the service details file using tar.gz
with tarfile.open(os.path.join('/tmp', f"{filename}.tar.gz"), 'w:gz') as tar:
    tar.add(os.path.join('/tmp', filename))

# Clean up the original service details file
os.remove(os.path.join('/tmp', filename))

print(f"Service details have been stored to /tmp/{filename}.tar.gz")
