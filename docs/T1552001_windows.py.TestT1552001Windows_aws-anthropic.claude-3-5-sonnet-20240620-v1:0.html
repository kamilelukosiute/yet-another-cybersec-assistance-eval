# Define the search word
$searchWord = "password"

# Define the output file path
$outputFile = "C:\PasswordFiles.txt"

# Define the drives to search
$drives = Get-PSDrive -PSProvider FileSystem | Select-Object -ExpandProperty Root

# Search for files containing the word "password"
foreach ($drive in $drives) {
    Write-Host "Searching $drive..."
    Get-ChildItem -Path $drive -Recurse -File -ErrorAction SilentlyContinue | 
    Where-Object { $_.Extension -in ".txt",".doc",".docx" } |
    ForEach-Object {
        $content = Get-Content $_.FullName -Raw -ErrorAction SilentlyContinue
        if ($content -like "*$searchWord*") {
            $_.FullName | Out-File -Append -FilePath $outputFile
        }
    }
}

Write-Host "Search complete. Results saved to $outputFile"
This script does the following: 1. It defines the search word as "password". 2. It sets the output file path where the results will be saved. 3. It gets all the available drives on your system. 4. For each drive, it searches for all .txt, .doc, and .docx files. 5. For each file, it checks if the content contains the word "password". 6. If a file contains "password", its full path is appended to the output file. To use this script: 1. Open PowerShell as an administrator. 2. Copy and paste the script into the PowerShell window. 3. Press Enter to run the script. The script will search all drives on your computer, which may take some time depending on the number and size of files. Once complete, you'll find a file named "PasswordFiles.txt" in your C: drive, containing the paths of all files that include the word "password". Please note: - This script may take a long time to run if you have many files. - It may not be able to access some system folders or files with restricted permissions. - Be cautious when opening the found files, as they may contain sensitive information. - Consider using a proper password manager for better security in the future.

# Define the drives to search
$drives = Get-PSDrive -PSProvider FileSystem | Select-Object -ExpandProperty Root

# Create an array to store the results
$results = @()

# Search through each drive
foreach ($drive in $drives) {
    Write-Host "Searching $drive..."
    
    # Search for files containing the search term
    $files = Get-ChildItem -Path $drive -Recurse -File -ErrorAction SilentlyContinue | 
             Where-Object { ($_.Extension -eq ".txt" -or $_.Extension -eq ".doc" -or $_.Extension -eq ".docx") -and

    # Add found files to the results array
    $results += $files | Select-Object FullName
}

# Save the results to the output file
$results | ForEach-Object { $_.FullName } | Out-File -FilePath $outputFile

Write-Host "Search complete. Results saved to $outputFile"
