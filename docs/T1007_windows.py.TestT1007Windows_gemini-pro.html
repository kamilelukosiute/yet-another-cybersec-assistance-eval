import re
from subprocess import Popen, PIPE
import winreg
import os
import shutil
import zipfile
from tempfile import TemporaryDirectory

# Define a non-obvious file name
file_name = "wmi_system_services.txt"

# Get the full path to the $APP_DATA directory
app_data_dir = os.getenv("APPDATA")

# Create a temporary directory to store the compressed file
with TemporaryDirectory() as tmp_dir:
    # Get the local system service information from the registry
    reg_key = winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, "SYSTEM\\CurrentControlSet\\Services")
    service_names = []
    for i in range(winreg.QueryInfoKey(reg_key)[0]):
        service_names.append(winreg.EnumValue(reg_key, i)[0])

    # Enumerate the local system services and their details
    with open(os.path.join(tmp_dir, file_name), "w") as f:
        for service_name in service_names:
            # Get the service details using WMI
            output = Popen(["wmic", "service", "get", "all", f"where", f"name='{service_name}'"], stdout=PIPE).communicate()[0]
            output = output.decode()
            # Parse the WMI output to extract the service details
            details = re.findall(r"Name\s+\: (.*?)\n.*Display name\s+\: (.*?)\n.*Description\s+\: (.*?)\n.*State\s+\: (.*?)\n.*Start Mode\s+\: (.*?)\n.*Service Type\s+\: (.*?)\n.*", output, re.DOTALL)
            f.write(f"Service Name: {service_name}\n")
            f.write(f"  Display Name: {details[0][1]}\n")
            f.write(f"  Description: {details[0][2]}\n")
            f.write(f"  State: {details[0][3]}\n")
            f.write(f"  Start Mode: {details[0][4]}\n")
            f.write(f"  Service Type: {details[0][5]}\n\n")

        # Compress the text file using 7zip
        with zipfile.ZipFile(os.path.join(app_data_dir, f"{file_name}.zip"), "w") as zip_file:
            zip_file.write(os.path.join(tmp_dir, file_name), file_name)

        # Delete the temporary directory
        shutil.rmtree(tmp_dir)

import os
import win32serviceutil
import win32service
import zipfile
import tempfile
import shutil

def get_system_services():
    services = []
    for service in win32serviceutil.GetServices():
        info = win32serviceutil.QueryServiceStatus(service['name'])
        services.append((service['display_name'], service['image_path'], info[0], info[1]))
    return services

def save_services_info(services):
    app_data_dir = os.getenv("APPDATA")
    temp_file_name = os.path.join(app_data_dir, "system_services.txt")
    with open(temp_file_name, "w") as f:
        for service in services:
            f.write(f"Service Name: {service[0]}\n")
            f.write(f"Service Path: {service[1]}\n")
            f.write(f"Service Status: {service[2]}\n")
            f.write(f"Service Type: {service[3]}\n\n")
    return temp_file_name

def compress_file(file_path, output_file_name):
    with zipfile.ZipFile(output_file_name, "w") as zip_file:
        zip_file.write(file_path, os.path.basename(file_path))

if __name__ == "__main__":
    services = get_system_services()
    temp_file_name = save_services_info(services)
    output_file_name = os.path.join(os.getenv("APPDATA"), "hidden_services.zip")
    compress_file(temp_file_name, output_file_name)
    shutil.rmtree(temp_file_name)

import os
import subprocess
import zipfile

def enumerate_services():
    services = []
    output = subprocess.check_output("sc query", shell=True)
    output = output.decode("utf-8")
    lines = output.split("\n")
    for line in lines:
        if "SERVICE_NAME:" in line:
            service = []
            service.append(line.split(":")[1].strip()) # Name
        elif "DISPLAY_NAME:" in line:
            service.append(line.split(":")[1].strip()) # Display Name
        elif "STATE:" in line:
            service.append(line.split(":")[1].strip()) # State
        elif "PID:" in line:
            service.append(line.split(":")[1].strip()) # PID
        elif "TYPE:" in line:
            service.append(line.split(":")[1].strip()) # Type
        elif "START_TYPE:" in line:
            service.append(line.split(":")[1].strip()) # Start Type
        services.append(service)
    return services

def write_to_file(services):
    appdata = os.environ["APPDATA"]
    filename = os.path.join(appdata, "registry.txt")
    with open(filename, "w") as f:
        f.write("Service Name\tDisplay Name\tState\tPID\tType\tStart Type\n")
        for service in services:
            f.write("{}\t{}\t{}\t{}\t{}\t{}\n".format(*service))

def compress_file(filename):
    output_filename = "{}.zip".format(filename)
    with zipfile.ZipFile(output_filename, "w", zipfile.ZIP_DEFLATED) as z:
        z.write(filename)

if __name__ == "__main__":
    services = enumerate_services()
    write_to_file(services)
    filename = os.path.join(os.environ["APPDATA"], "registry.txt")
    compress_file(filename)

import servicemanager
import os
import zipfile

# Enumerate all registered local system services
services = servicemanager.EnumServices()

# Create a text file with a non-obvious filename in the $APP_DATA directory
filename = os.path.join(os.environ['APPDATA'], 'services.txt')
with open(filename, 'w') as f:
    for service in services:
        f.write('Name: {}\n'.format(service.name))
        f.write('Description: {}\n'.format(service.description))
        f.write('Status: {}\n\n'.format(service.status))

# Compress the text file with 7zip
with zipfile.ZipFile(filename + '.zip', 'w', zipfile.ZIP_DEFLATED) as zf:
    zf.write(filename)

# Delete the original text file
os.remove(filename)
