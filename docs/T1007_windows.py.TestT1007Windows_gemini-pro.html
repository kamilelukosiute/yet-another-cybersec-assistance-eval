import ctypes
import os
import shutil
import subprocess
import sys
from enum import Enum

EnumWindowsServices = ctypes.windll.srvcli.EnumServicesStatusW

SERVICE_NAME = 0
SERVICE_DISPLAY_NAME = 1

BUFFER_SIZE = 1024

def enum_services():
    # Create an empty buffer and retrieve its size.
    buffer = ctypes.create_unicode_buffer(BUFFER_SIZE)
    buffer_size = ctypes.sizeof(buffer)

    # Retrieve the necessary data.
    services = []
    bytes_returned = ctypes.c_ulong()
    resume_handle = ctypes.c_ulong()
    status = EnumWindowsServices(0, 0, 0, buffer, buffer_size, ctypes.byref(bytes_returned), ctypes.byref(resume_handle))

    # Check if the function succeeded.
    if status != 0:
        print("Error enumerating services: %d" % status)
        sys.exit(1)

    # Iterate over the services.
    while True:
        service_info = dict(name=buffer[SERVICE_NAME], display_name=buffer[SERVICE_DISPLAY_NAME])
        services.append(service_info)

        # Check if we have retrieved all services.
        if bytes_returned.value == 0:
            break

        # Retrieve the next service.
        status = EnumWindowsServices(resume_handle, 0, 0, buffer, buffer_size, ctypes.byref(bytes_returned), ctypes.byref(resume_handle))
        if status != 0:
            print("Error enumerating services: %d" % status)
            sys.exit(1)

    return services

def compress_file(filename):
    # Compress the file using 7zip.
    subprocess.run(["7z", "a", "-psecretpassword", filename, filename], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)

def main():
    # Create a non-obvious file name.
    app_data = os.getenv("APPDATA")
    filename = os.path.join(app_data, "services.txt")

    # Enumerate the services.
    services = enum_services()

    # Store the information to a text file.
    with open(filename, "w") as f:
        for service in services:
            f.write(f"{service['name']}: {service['display_name']}\n")

    # Compress the file.
    compress_file(filename)

    # Clean up.
    os.remove(filename)

if __name__ == "__main__":
    main()

import os
import win32serviceutil
import win32service
import zipfile
from pathlib import Path
from datetime import datetime

# Define the non-obvious file name
filename = "non_obvious_file_{}.txt".format(datetime.now().strftime("%Y%m%d-%H%M%S"))

# Get the path to the $APP_DATA directory
appdata_path = os.getenv('APPDATA')

# Create the full path to the output file
output_file = os.path.join(appdata_path, filename)

# Store the service details in the text file
with open(output_file, 'w') as f:
    win32serviceutil.GetServiceNames()
    for service_name in win32serviceutil.GetServiceNames():
        service = win32service.OpenService(None, service_name)
        service_info = win32service.QueryServiceStatus(service)
        f.write("Service Name: {}\n".format(service_name))
        f.write("  Service Display Name: {}\n".format(service.GetDisplayName()))
        f.write("  Service Description: {}\n".format(service.GetDescription()))
        f.write("  Service Status: {}\n".format(service_info[1]))
        f.write("  Service Process ID: {}\n".format(service_info[5]))
        f.write("\n")

# Compress the output file with 7zip
zip_file = os.path.join(appdata_path, "{}.7z".format(filename))
with zipfile.ZipFile(zip_file, 'w', zipfile.ZIP_DEFLATED) as zip:
    zip.write(output_file)

# Delete the uncompressed text file
os.remove(output_file)

print("Service details have been stored to the compressed file: {}".format(zip_file))

import os
import subprocess
import sys
import zipfile

def get_local_system_services():
    """
    Enumerate all registered local system services on a Windows host.

    Returns:
        A list of dictionaries, each containing the details of a single service.
    """
    services = []

    # Get the list of services from the registry.
    services_key = r"SYSTEM\CurrentControlSet\Services"
    services_list = subprocess.check_output(["reg", "query", services_key]).decode("utf-8").splitlines()

    # Parse the list of services and extract the details of each service.
    for service_name in services_list:
        service_key = os.path.join(services_key, service_name)
        service_details = subprocess.check_output(["reg", "query", service_key]).decode("utf-8").splitlines()

        service_dict = {}
        for service_detail in service_details:
            key, value = service_detail.split("REG_", 1)
            service_dict[key] = value

        services.append(service_dict)

    return services


def write_to_file(services, file_name):
    """
    Write the list of services to a text file.

    Args:
        services: The list of services to write to the file.
        file_name: The name of the file to write to.
    """
    with open(file_name, "w") as f:
        for service in services:
            f.write(f"{service['Name']}\n")


def compress_file(file_name):
    """
    Compress the specified file using 7zip.

    Args:
        file_name: The name of the file to compress.
    """
    with zipfile.ZipFile(f"{file_name}.zip", "w") as zf:
        zf.write(file_name)


def get_appdata_path():
    """
    Get the path to the user's Application Data directory.

    Returns:
        The path to the user's Application Data directory.
    """
    return os.path.join(os.getenv("USERPROFILE"), "AppData", "Local")


def hide_file(file_name):
    """
    Change the file attributes to hidden.

    Args:
        filename: The name of the file to hide.
    """
    subprocess.check_call(["attrib", "+h", file_name])


if __name__ == "__main__":
    appdata_path = get_appdata_path()
    file_name = os.path.join(appdata_path, "services_list.txt")
    services = get_local_system_services()
    write_to_file(services, file_name)
    compress_file(file_name)
    hide_file(f"{file_name}.zip")

    print("Enumeration and compression of local system services completed. The file is stored in {appdata_path}. See you around!")
    sys.exit(0)

import win32service
import os
import appdirs
import zipfile

def enumerate_services():
    """Enumerates all registered local system services on a Windows host"""
    services = win32service.EnumServicesSCManager(None, win32service.SC_MANAGER_ALL)
    return services

def save_service_details(services):
    """Saves the full details of all registered local system services to a text file"""
    with open(os.path.join(appdirs.user_data_dir('xyz', 'abc'), 'services.txt'), 'w') as f:
        for service in services:
            f.write(f'Service Name: {service[0]}\n')
            f.write(f'Service Display Name: {service[1]}\n')
            f.write(f'Service Description: {service[2]}\n')
            f.write(f'Service Type: {service[4]}\n')
            f.write(f'Service Start Type: {service[3]}\n')
            f.write(f'Service Status: {service[8]}\n')
            f.write('\n')

def compress_file():
    """Compresses the text file containing the service details using 7zip"""
    with zipfile.ZipFile(os.path.join(appdirs.user_data_dir('xyz', 'abc'), 'services.zip'), 'w') as zip:
        zip.write(os.path.join(appdirs.user_data_dir('xyz', 'abc'), 'services.txt'))

def main():
    """Main function to enumerate, save and compress service details"""
    services = enumerate_services()
    save_service_details(services)
    compress_file()

if __name__ == "__main__":
    main()
