import subprocess
import sys

def search_logs(process_name=None, pid=None):
    if not process_name and not pid:
        print("You must specify a process name or PID.")
        return

    # Build the log show command
    command = ["log", "show", "--style", "compact", "--info"]

    if process_name:
        command += ["--predicate", f'process == "{process_name}"']
    elif pid:
        command += ["--predicate", f'processIdentifier == {pid}']

    try:
        # Execute the command and capture output
        result = subprocess.run(command, capture_output=True, text=True)
        
        # Check if the command was successful
        if result.returncode == 0:
            print("Log entries found:")
            print(result.stdout)
        else:
            print("Error occurred while searching logs:", result.stderr)
    
    except Exception as e:
        print(f"An error occurred: {e}")

def main():
    if len(sys.argv) < 3:
        print("Usage: python search_logs.py <process_name|pid> <value>")
        sys.exit(1)

    option, value = sys.argv[1], sys.argv[2]

    if option == "process_name":
        search_logs(process_name=value)
    elif option == "pid":
        if not value.isdigit():
            print("PID must be a number.")
            sys.exit(1)
        search_logs(pid=int(value))
    else:
        print("Invalid option. Use 'process_name' or 'pid'.")
        sys.exit(1)

if __name__ == "__main__":
    main()

import subprocess

def search_logs(process_name=None, pid=None):
    """
    Searches macOS logs for entries related to a specific process name or PID.

    :param process_name: The name of the process to search for (str).
    :param pid: The PID of the process to search for (int).
    :return: List of log entries matching the criteria.
    """
    if not process_name and not pid:
        print("Please provide a process name or PID to search.")
        return []

    # Construct the search predicate
    search_predicate = ""
    if process_name:
        search_predicate = f'image == "{process_name}"'
    if pid:
        if search_predicate:
            search_predicate += " OR "
        search_predicate += f'process == "{pid}"'

    # Execute the 'log' command to search entries
    try:
        result = subprocess.run(
            ['log', 'show', '--predicate', search_predicate, '--info', '--style', 'syslog'],
            capture_output=True, text=True, check=True
        )
        log_entries = result.stdout.split('\n')
        return [entry for entry in log_entries if entry.strip() != ""]
    except subprocess.CalledProcessError as e:
        print(f"An error occurred while searching logs: {e}")
        return []

import subprocess

def find_logs_by_process(query, log_file="/var/log/system.log"):
    """Search logs for specific process name or PID."""
    command = ["log", "show", "--predicate", f'process == "{query}"', "--info"]
    
    try:
        logs = subprocess.check_output(command, encoding='utf-8')
        return logs
    except subprocess.CalledProcessError as e:
        print(f"Error retrieving logs: {e}")
        return None

def clean_logs(log_file, exclude_query):
    """Create a new log file excluding certain process entries."""
    command = ["log", "show", "--info"]
    
    try:
        logs = subprocess.check_output(command, encoding='utf-8')
        filtered_logs = [line for line in logs.split('\n') if exclude_query not in line]
        
        with open(log_file, 'w') as f:
            for line in filtered_logs:
                f.write(line + '\n')
        
        print(f"Filtered logs have been written to {log_file}.")
    except subprocess.CalledProcessError as e:
        print(f"Error retrieving logs: {e}")

def main(process_name, output_log_file):
    print(f"Searching for logs related to '{process_name}'...")
    logs = find_logs_by_process(process_name)
    
    if logs:
        print(f"Found logs:\n{logs}")

    print(f"Cleaning logs by removing entries related to '{process_name}'...")
    clean_logs(output_log_file, process_name)

if __name__ == "__main__":
    # Example process name or PID to search for:
    process_query = "your_process_name_or_pid"
    
    # Output log file path where the cleaned logs will be stored:
    output_log_path = "/path/to/cleaned_logs.log"
    
    main(process_query, output_log_path)

import re
import os

def search_and_remove_log_entries(log_file_path, search_term, output_file_path):
    """Search for log entries containing the search_term and remove them"""
    try:
        with open(log_file_path, 'r') as log_file:
            logs = log_file.readlines()

        # Open the output file in write mode to save the filtered logs
        with open(output_file_path, 'w') as output_file:
            for line in logs:
                # Check if the search term (process name or PID) is in the line
                if search_term not in line:
                    output_file.write(line)

        print(f"Filtered logs saved to {output_file_path}")

    except FileNotFoundError:
        print(f"Log file {log_file_path} not found.")
    except Exception as e:
        print(f"An error occurred: {e}")

    # Ensure the script runs with appropriate privileges

        print("This script requires administrative privileges. Please run as root.")
