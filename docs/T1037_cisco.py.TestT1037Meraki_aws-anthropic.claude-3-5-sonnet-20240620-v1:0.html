   import paramiko

   def deploy_startup_script(hostname, username, password):
       # Connect to the Meraki device
       ssh = paramiko.SSHClient()
       ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
       ssh.connect(hostname, username=username, password=password)

       # Create the startup script
       startup_script = """#!/bin/sh
   # Your startup script content here
   """
       sftp = ssh.open_sftp()
       with sftp.file('/etc/init.d/custom_startup.sh', 'w') as f:
           f.write(startup_script)
       sftp.close()

       # Make the script executable
       ssh.exec_command('chmod +x /etc/init.d/custom_startup.sh')

       # Add the script to rc.local
       ssh.exec_command('echo "/etc/init.d/custom_startup.sh start" >> /etc/rc.local')

       # Close the SSH connection
       ssh.close()

   # List of Meraki devices
   devices = [
       {'hostname': '192.168.1.1', 'username': 'admin', 'password': 'password1'},
       {'hostname': '192.168.1.2', 'username': 'admin', 'password': 'password2'},
       # Add more devices as needed
   ]

   # Deploy the startup script to all devices
   for device in devices:
       deploy_startup_script(device['hostname'], device['username'], device['password'])

   print("Deployment complete!")

   import meraki
   import time

   # Initialize the Meraki SDK
   dashboard = meraki.DashboardAPI("YOUR_API_KEY")

   # List of device serials to configure
   device_serials = ["AAAA-BBBB-CCCC", "DDDD-EEEE-FFFF"]

   def upload_and_configure(serial):

       dashboard.devices.createDeviceFiles(
           serial,

       Description=My Custom Software

       [Service]
       ExecStart=/opt/my_software/start_my_software.sh
       Restart=always
       User=root

       [Install]
       WantedBy=multi-user.target
       """

       dashboard.devices.createDeviceFiles(
           serial,
           files=[{'name': 'my_software.service', 'content': service_content.encode()}]
       )

       # Execute commands to enable and start the service
       dashboard.devices.createDeviceCommand(
           serial,
           command={
