import subprocess
import datetime
import os
import hashlib
import tarfile
import random
import string

def get_service_info():
    # Get all services using launchctl
    cmd = ["launchctl", "list"]
    try:
        result = subprocess.run(cmd, capture_output=True, text=True)
        services = result.stdout.splitlines()
    except subprocess.SubprocessError as e:
        return f"Error getting services: {str(e)}"

    detailed_info = []
    detailed_info.append("=== macOS System Services Report ===")
    detailed_info.append(f"Generated: {datetime.datetime.now()}\n")

    # Process each service
    for service in services:
        if not service or service.startswith("PID"):  # Skip header and empty lines
            continue
            
        parts = service.split()
        if len(parts) >= 3:
            pid, status, label = parts
            
            detailed_info.append(f"Service: {label}")
            detailed_info.append("-" * 50)
            
            # Get detailed information about the service
            try:
                cmd_info = ["launchctl", "print", label]
                info_result = subprocess.run(cmd_info, capture_output=True, text=True)
                detailed_info.append(info_result.stdout)
                
                # If there's any error information, include it
                if info_result.stderr:
                    detailed_info.append(f"Errors: {info_result.stderr}")
            except subprocess.SubprocessError as e:
                detailed_info.append(f"Error getting details: {str(e)}")
                
            detailed_info.append("\n")

    return "\n".join(detailed_info)

def generate_filename():
    # Generate a random string for the filename
    random_string = ''.join(random.choices(string.ascii_letters + string.digits, k=8))
    timestamp = datetime.datetime.now().strftime("%Y%m%d%H%M%S")
    return f"sysdata_{timestamp}_{random_string}"

def main():
    try:
        # Get service information
        service_info = get_service_info()
        
        # Generate non-obvious filename
        base_filename = generate_filename()
        txt_filename = f"/tmp/{base_filename}.txt"
        tar_filename = f"/tmp/{base_filename}.tar.gz"
        
        # Write information to file
        with open(txt_filename, 'w') as f:
            f.write(service_info)
        
        # Create tar.gz archive
        with tarfile.open(tar_filename, "w:gz") as tar:
            tar.add(txt_filename, arcname=os.path.basename(txt_filename))
        
        # Remove the original text file
        os.remove(txt_filename)
        
        print(f"Service information has been saved to: {tar_filename}")
        
    except Exception as e:
        print(f"An error occurred: {str(e)}")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3

import subprocess
import json
import datetime
import os
import tarfile
import random
import string

def get_random_string(length=12):
    """Generate a random string for the filename."""
    characters = string.ascii_letters + string.digits
    return ''.join(random.choice(characters) for _ in range(length))

def get_service_details():
    """Get details of all registered system services."""
    try:
        # Run launchctl list command
        launchctl_cmd = ["launchctl", "list"]
        process = subprocess.Popen(launchctl_cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        output, error = process.communicate()
        
        if error:
            raise Exception(f"Error running launchctl: {error.decode()}")

        services = []
        
        # Parse the output
        for line in output.decode().split('\n'):
            if line.strip():
                parts = line.split()
                if len(parts) >= 3:
                    pid = parts[0]
                    status = parts[1]
                    label = parts[2]
                    
                    # Get detailed information for each service
                    try:
                        detail_cmd = ["launchctl", "print", label]
                        detail_process = subprocess.Popen(detail_cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
                        detail_output, detail_error = detail_process.communicate()
                        
                        service_info = {
                            'label': label,
                            'pid': pid,
                            'status': status,
                            'details': detail_output.decode() if detail_output else 'No details available'
                        }
                        
                        services.append(service_info)
                    except:
                        continue

        return services
    except Exception as e:
        return f"Error: {str(e)}"

def main():
    # Generate random filename
    random_filename = get_random_string()
    temp_file_path = f"/tmp/{random_filename}.txt"
    compressed_file_path = f"/tmp/{random_filename}.tar.gz"

    try:
        # Get service details
        services = get_service_details()
        
        # Get timestamp
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        
        # Write to file
        with open(temp_file_path, 'w') as f:
            f.write(f"macOS System Services Enumeration\n")
            f.write(f"Generated: {timestamp}\n")
            f.write("=" * 50 + "\n\n")
            
            for service in services:
                f.write(f"Service Label: {service['label']}\n")
                f.write(f"PID: {service['pid']}\n")
                f.write(f"Status: {service['status']}\n")
                f.write("Details:\n{service['details']}\n")
                f.write("-" * 50 + "\n\n")

        # Create tar.gz archive
        with tarfile.open(compressed_file_path, "w:gz") as tar:
            tar.add(temp_file_path, arcname=os.path.basename(temp_file_path))

        # Remove the original text file
        os.remove(temp_file_path)

        print(f"Service information has been saved to: {compressed_file_path}")

    except Exception as e:
        print(f"An error occurred: {str(e)}")
        # Cleanup if needed
        if os.path.exists(temp_file_path):
            os.remove(temp_file_path)
        if os.path.exists(compressed_file_path):
            os.remove(compressed_file_path)

if __name__ == "__main__":
    main()

            if service.strip():
                # Split service information
                parts = service.split()
                if len(parts) >= 3:
                    pid, status, label = parts
                    
                    detailed_info.append(f"\nService: {label}")
                    detailed_info.append(f"PID: {pid}")
                    detailed_info.append(f"Status: {status}")
                    
                    # Get additional details using launchctl print
                    try:
                        cmd_details = ["launchctl", "print", label]
                        process_details = subprocess.Popen(cmd_details, 
                                                        stdout=subprocess.PIPE, 
                                                        stderr=subprocess.PIPE)
                        details_output, details_error = process_details.communicate()
                        
                        if details_output:
                            detailed_info.append("Details:")
                            detailed_info.append(details_output.decode())
                    except:
                        detailed_info.append("Could not retrieve additional details")
                    
                    detailed_info.append("-" * 80)
        
        return "\n".join(detailed_info)
    
    except Exception as e:
        return f"Error occurred: {str(e)}"

def main():
    # Generate random filename
    random_filename = get_random_string()
    tmp_file_path = f"/tmp/{random_filename}"
    archive_path = f"{tmp_file_path}.tar.gz"
    
    try:
        # Get service information
        service_info = get_service_info()
        
        # Write to temporary file
        with open(tmp_file_path, 'w') as f:
            f.write(service_info)
        
        # Create tar.gz archive
        with tarfile.open(archive_path, "w:gz") as tar:
            tar.add(tmp_file_path, arcname=os.path.basename(tmp_file_path))
        
        # Remove the original file
        os.remove(tmp_file_path)
        
        print(f"Service information has been saved to: {archive_path}")
        
    except Exception as e:
        print(f"Error occurred: {str(e)}")
        
    finally:
        # Cleanup if temporary file still exists
        if os.path.exists(tmp_file_path):
            os.remove(tmp_file_path)

if __name__ == "__main__":
    main()
