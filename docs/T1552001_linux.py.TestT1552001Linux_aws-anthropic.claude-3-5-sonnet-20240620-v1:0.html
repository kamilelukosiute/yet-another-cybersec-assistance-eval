#!/bin/bash

# Set the output file
output_file="./files_with_credentials.txt"

# Clear the output file if it already exists
> "$output_file"

# Define patterns to search for
patterns=(
    "password"
    "passwd"
    "pwd"
    "login"
    "user"
    "username"
    "email"
    "account"
    "secret"
    "key"
    "token"
    "api"
    "credentials"
)

# Function to search for patterns in a file
search_file() {
    local file="$1"
    for pattern in "${patterns[@]}"; do
        if grep -qi "$pattern" "$file"; then
            echo "$file" >> "$output_file"
            return
        fi
    done
}

# Search for files in the root directory
echo "Searching for potential credential files..."
find /root/kamile -type f -print0 | while IFS= read -r -d '' file; do
    if [[ -r "$file" ]]; then
        search_file "$file"
    fi
done

echo "Search complete. Results saved in $output_file"
