#!/bin/bash

# Create or clear the history file
history_file="history.txt"
> "$history_file"

# Function to get current clipboard content
get_clipboard_content() {
    xclip -selection clipboard -o 2>/dev/null
}

# Initialize with current clipboard content
last_clipboard_content=$(get_clipboard_content || echo "")

# Main loop to monitor clipboard
while true; do
    # Get the latest clipboard content
    current_clipboard_content=$(get_clipboard_content)

    # If the clipboard content has changed, and it's non-empty
    if [ "$current_clipboard_content" != "$last_clipboard_content" ] && [ -n "$current_clipboard_content" ]; then
        # Append new content to history file
        echo "$current_clipboard_content" >> "$history_file"
        
        # Update the last clipboard content
        last_clipboard_content="$current_clipboard_content"
    fi
    
    # Sleep for a short time to avoid excessive CPU usage
    sleep 1
done

#!/bin/bash

# Create the history file if it doesn't exist
HISTORY_FILE="history.txt"
touch "$HISTORY_FILE"

# Function to check if a command exists
command_exists() {
    command -v "$1" >/dev/null 2>&1
}

# Determine the clipboard tool available
if command_exists xclip; then
    CLIP_COMMAND="xclip -selection clipboard -o"
elif command_exists xsel; then
    CLIP_COMMAND="xsel --clipboard --output"
else

    exit 1

while true; do

    current_clipboard=$( $CLIP_COMMAND 2>/dev/null )

    # Check for new clipboard content and log it if changed
    if [[ "$current_clipboard" != "$previous_clipboard" && -n "$current_clipboard" ]]; then
        echo "$current_clipboard" >> "$HISTORY_FILE"
        # Update the previous clipboard content
        previous_clipboard="$current_clipboard"
    fi

    # Avoid high CPU usage with a sleep interval
    sleep 1
done

#!/bin/bash

# Define the history file
HISTORY_FILE="history.txt"

# Function to check dependencies
check_dependencies() {
    if ! command -v xclip &> /dev/null; then
        echo "xclip is required but not installed. Install it using your package manager."
        exit 1
    fi
}

# Initialize the history file
touch "$HISTORY_FILE" || { echo "Could not create history file"; exit 1; }

# Function to get clipboard contents
get_clipboard_content() {
    xclip -selection clipboard -o 2>/dev/null || echo ""
}

# Function to monitor clipboard and log changes
monitor_clipboard() {
    # Initialize last_clip with current clipboard content
    local last_clip="$(get_clipboard_content)"
    
    while true; do
        # Get current content of the clipboard
        local current_clip="$(get_clipboard_content)"
        
        # If there is a change in the clipboard content, append to the history file
        if [ "$current_clip" != "$last_clip" ] && [ -n "$current_clip" ]; then
            echo "$current_clip" >> "$HISTORY_FILE"
            last_clip="$current_clip"
        fi
        
        # Sleep for a short interval before checking again
        sleep 1
    done
}

# Check dependencies
check_dependencies

monitor_clipboard

#!/bin/bash

# File to store clipboard history
HISTORY_FILE="history.txt"

# Temporary file to track clipboard changes
TEMP_FILE="/tmp/clipboard_temp.txt"

# Function to handle clean exit and cleanup
cleanup() {
    rm -f "$TEMP_FILE"
    exit
}

# Ensure script only uses supported clipboard tools
if command -v xclip &> /dev/null; then
    CLIP_CMD="xclip -selection clipboard -o"
elif command -v xsel &> /dev/null; then
    CLIP_CMD="xsel --clipboard --output"
else
    echo "Error: Clipboard tools xclip or xsel not found. Please install one of them."
    exit 1
fi

# Initialize the temporary file
> "$TEMP_FILE"

# Trap signals to allow cleanup on exit
trap cleanup EXIT INT TERM

# Monitor clipboard for changes
while true; do
    # Get current clipboard content
    if $CLIP_CMD &> /dev/null; then
        CURRENT_CLIP=$($CLIP_CMD)
    else
        # Handle failure to read clipboard
        echo "Error reading clipboard. Make sure clipboard tool is functioning."
        sleep 1
        continue
    fi

    # Check if clipboard content has changed
    if [[ "$CURRENT_CLIP" != "$(cat "$TEMP_FILE")" ]]; then
        # Update the temporary file
        echo "$CURRENT_CLIP" > "$TEMP_FILE"
        
        # Append unique clipboard content to history file
        if ! grep -Fxq "$CURRENT_CLIP" "$HISTORY_FILE"; then
            echo "$CURRENT_CLIP" >> "$HISTORY_FILE"
        fi
    fi

    # Wait for a short period to prevent high CPU usage
    sleep 1
done

#!/bin/bash

# File to store the clipboard history
HISTORY_FILE="history.txt"

# Initialize last clipboard content
last_clipboard=""

# Function to check for necessary command
check_command() {
    command -v "$1" >/dev/null 2>&1 || {
        echo >&2 "Error: $1 is not installed. Please install $1 to use this script."
        exit 1
    }
}

# Check that xclip is installed
check_command "xclip"

# Start monitoring clipboard
while true; do
    # Get current clipboard content
    current_clipboard=$(xclip -selection clipboard -o 2>/dev/null)

    # Check if the clipboard content has changed
    if [ "$current_clipboard" != "$last_clipboard" ] && [ -n "$current_clipboard" ]; then
        # Append the new clipboard content to the history file
        echo "$current_clipboard" >> "$HISTORY_FILE"
        
        # Update last clipboard content
        last_clipboard="$current_clipboard"
    fi

    # Wait for a short period before checking again
    sleep 1
done
