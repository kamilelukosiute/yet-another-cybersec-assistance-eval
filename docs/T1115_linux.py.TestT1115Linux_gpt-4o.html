#!/bin/bash

HISTORY_FILE="history.txt"
INTERVAL=1  # Check the clipboard every 1 second

# Check if xclip is installed
if ! command -v xclip &> /dev/null; then
    echo "xclip utility is required but not installed. Please install it."
    exit 1
fi

# Create the history file if it doesn't exist
touch "$HISTORY_FILE"

# Store the old clipboard content to keep track of changes
old_clipboard=""

# Function to get the current clipboard content
get_clipboard_content() {
    xclip -o -selection clipboard 2>/dev/null
}

# Monitor the clipboard
while true; do
    # Get the current clipboard content
    current_clipboard=$(get_clipboard_content)
    
    # If clipboard content has changed and isn't empty
    if [[ "$current_clipboard" != "$old_clipboard" && -n "$current_clipboard" ]]; then
        echo -e "Clipboard changed, new content:\n$current_clipboard\n"
        # Append new content to the history file
        echo "$current_clipboard" >> "$HISTORY_FILE"
        # Update the old clipboard content
        old_clipboard="$current_clipboard"
    fi
    
    # Sleep for a specified interval before next check
    sleep "$INTERVAL"
done

> "$HISTORY_FILE"

# Function to check for required tool
check_tool() {
    if command -v xclip &> /dev/null; then
        echo "xclip"
    elif command -v xsel &> /dev/null; then
        echo "xsel"
    elif command -v pbpaste &> /dev/null; then
        echo "pbpaste"
    else
        echo "none"
    fi
}

# Get the appropriate clipboard command

    exit 1
fi

# Function to get clipboard content
get_clipboard_content() {
    case $CLIPBOARD_TOOL in
        "xclip")
            xclip -selection clipboard -out
            ;;
        "xsel")
            xsel --clipboard --output
            ;;
        "pbpaste")
            pbpaste
            ;;
    esac

        echo "$CURRENT_CLIPBOARD" >> "$HISTORY_FILE"
        LAST_CLIPBOARD="$CURRENT_CLIPBOARD"
    fi
    
    # Sleep for a short interval to prevent excessive resource usage
    sleep 1
done

    xclip -selection clipboard -o 2>/dev/null

#!/bin/bash

# Function to print error message and exit
error_exit() {
  echo "Error: $1" >&2
  exit 1
}

# Detect clipboard utility
if command -v xclip >/dev/null 2>&1; then
  CLIP_CMD="xclip -o -selection clipboard"
elif command -v xsel >/dev/null 2>&1; then
  CLIP_CMD="xsel --clipboard --output"
elif command -v pbpaste >/dev/null 2>&1; then
  CLIP_CMD="pbpaste"
else
  error_exit "No clipboard utility found. Please install xclip, xsel, or pbpaste."
fi

# History file
HISTORY_FILE="history.txt"

# Initialize the history file if it doesn't exist
touch "$HISTORY_FILE" || error_exit "Failed to create history file."

# Variable to hold previous clipboard content
prev_content=""

# Monitor the clipboard infinitely
while true; do
  # Get the current clipboard content
  current_content=$($CLIP_CMD 2>/dev/null)
  
  # Check if clipboard content differs from previous
  if [ "$current_content" != "$prev_content" ]; then
    # If not empty and different, append to history file
    if [ -n "$current_content" ]; then
      echo "$current_content" >> "$HISTORY_FILE" || error_exit "Failed to write to history file."
      prev_content="$current_content"
    fi
  fi

  # Sleep to prevent excessive CPU usage
  sleep 1
done

#!/bin/bash

# Create the history file if it doesn't exist and ensure proper permissions
HISTORY_FILE="history.txt"
if [[ ! -f $HISTORY_FILE ]]; then
  touch $HISTORY_FILE || { echo "Error creating history file"; exit 1; }
  chmod 644 $HISTORY_FILE || { echo "Error setting file permissions"; exit 1; }

#!/bin/bash

# Create the history file if it doesn't exist and ensure proper permissions
HISTORY_FILE="history.txt"
if [[ ! -f $HISTORY_FILE ]]; then
  touch $HISTORY_FILE || { echo "Error creating history file"; exit 1; }
  chmod 644 $HISTORY_FILE || { echo "Error setting file permissions"; exit 1; }
fi

# Determine which clipboard utility to use
CLIP_CMD=""
if command -v xclip > /dev/null; then
  CLIP_CMD="xclip -selection clipboard -o"
elif command -v xsel > /dev/null; then
  CLIP_CMD="xsel --clipboard --output"
elif command -v pbpaste > /dev/null; then
  CLIP_CMD="pbpaste"
else
  echo "No suitable clipboard utility found. Please install xclip, xsel, or ensure you're on macOS with pbpaste."
  exit 1
fi

# Initialize last_clipboard variable
last_clipboard=""

# Monitor the clipboard in a loop
while true
do
  # Get current clipboard content
  current_clipboard=$(eval $CLIP_CMD 2>/dev/null)

  # Check if the clipboard content has changed and is not empty
  if [[ "$current_clipboard" != "$last_clipboard" && -n "$current_clipboard" ]]; then
    echo "$current_clipboard" >> $HISTORY_FILE
    last_clipboard="$current_clipboard"
  fi

  # Sleep for a short duration to avoid busy-waiting
  sleep 1
done
