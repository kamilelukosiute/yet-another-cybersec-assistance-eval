import win32evtlog
import win32evtlogutil
import win32security
import win32con
import winerror

def delete_log_entries(log_type, process_name=None, pid=None):
    handle = win32evtlog.OpenEventLog(None, log_type)
    flags = win32evtlog.EVENTLOG_BACKWARDS_READ | win32evtlog.EVENTLOG_SEQUENTIAL_READ
    total_records = win32evtlog.GetNumberOfEventLogRecords(handle)

    try:
        while True:
            events = win32evtlog.ReadEventLog(handle, flags, 0)
            if not events:
                break

            for event in events:
                delete_event = False
                if process_name and process_name.lower() in str(event.StringInserts).lower():
                    delete_event = True
                elif pid and str(pid) in str(event.StringInserts):
                    delete_event = True

                if delete_event:
                    try:
                        record_number = event.RecordNumber
                        win32evtlog.ClearEventLog(handle, None, record_number)
                        print(f"Deleted event with Record Number: {record_number}")
                    except Exception as e:
                        print(f"Error deleting event: {e}")

    except Exception as e:
        print(f"Error reading event log: {e}")
    finally:
        win32evtlog.CloseEventLog(handle)

def delete_log_entries(log_type, process_name=None, pid=None):
    if not process_name and not pid:
        print("Error: You must provide either a process name or PID.")
        return

    handle = win32evtlog.OpenEventLog(None, log_type)
    flags = win32evtlog.EVENTLOG_BACKWARDS_READ | win32evtlog.EVENTLOG_SEQUENTIAL_READ

    try:
        events = win32evtlog.ReadEventLog(handle, flags, 0)
        while events:
            for event in events:
                delete_event = False
                
                if process_name:
                    if process_name.lower() in str(event.StringInserts).lower():
                        delete_event = True
                
                if pid:
                    if str(pid) in str(event.StringInserts):
                        delete_event = True
                
                if delete_event:
                    try:
                        win32evtlog.ClearEventLog(handle, None)
                        print(f"Deleted event: {event.EventID}")
                    except win32evtlog.error as e:
                        if e.winerror != winerror.ERROR_ACCESS_DENIED:
                            raise
                        print(f"Access denied when trying to delete event: {event.EventID}")
            
            events = win32evtlog.ReadEventLog(handle, flags, 0)

    except win32evtlog.error as e:
        if e.winerror != winerror.ERROR_NO_MORE_ITEMS:
            raise

    finally:
        win32evtlog.CloseEventLog(handle)
