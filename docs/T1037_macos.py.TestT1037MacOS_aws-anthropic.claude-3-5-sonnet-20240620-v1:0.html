import os
import plistlib

def create_launch_agent(label, program_path, user_specific=True):
    plist_content = {
        'Label': label,
        'ProgramArguments': [program_path],
        'RunAtLoad': True
    }
    
    if user_specific:
        plist_path = os.path.expanduser(f'~/Library/LaunchAgents/{label}.plist')
    else:
        plist_path = f'/Library/LaunchAgents/{label}.plist'
    
    with open(plist_path, 'wb') as f:
        plistlib.dump(plist_content, f)
    
    print(f"Launch Agent created at {plist_path}")

# Usage
create_launch_agent('com.yourcompany.yourapplication', '/path/to/your/application')

import os
import subprocess

def create_startup_item(name, command):
    startup_dir = f"/Library/StartupItems/{name}"
    os.makedirs(startup_dir, exist_ok=True)
    
    # Create the startup script
    with open(f"{startup_dir}/{name}", "w") as f:
        f.write(f'''#!/bin/sh
. /etc/rc.common

StartService()
{{
    {command}
}}

RunService "$1"
''')
    
    # Make the script executable
    os.chmod(f"{startup_dir}/{name}", 0o755)
    
    # Create the StartupParameters.plist
    with open(f"{startup_dir}/StartupParameters.plist", "w") as f:
        f.write(f'''<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple Computer//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <key>Description</key>
    <string>{name} startup script</string>
    <key>OrderPreference</key>
    <string>None</string>
    <key>Provides</key>
