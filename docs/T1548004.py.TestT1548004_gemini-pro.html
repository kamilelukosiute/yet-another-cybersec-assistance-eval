#import <Authorization/Authorization.h>
#import <dlfcn.h>

int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // Get the path to the executable we want to execute with elevated privileges.
        NSString *executablePath = @"/usr/bin/ls";
        // Create an AuthorizationRef object.
        AuthorizationRef authorizationRef = NULL;
        OSStatus status = AuthorizationCreate(NULL, kAuthorizationEmptyEnvironment, kAuthorizationFlagDefaults, &authorizationRef);
        if (status != errAuthorizationSuccess) {
            NSLog(@"AuthorizationCreate failed: %d", status);
            return -1;
        }
        // Set the authorization rights for the executable.
        AuthorizationItem item = {kAuthorizationRightExecute, (UInt32)strlen(executablePath.UTF8String), executablePath.UTF8String};
        AuthorizationRights rights = {1, &item};
        status = AuthorizationCopyRights(authorizationRef, &rights, NULL);
        if (status != errAuthorizationSuccess) {
            NSLog(@"AuthorizationCopyRights failed: %d", status);
            return -1;
        }
        // Execute the executable with elevated privileges.
        void *executableHandle = dlopen(executablePath.UTF8String, RTLD_LAZY);
        if (executableHandle == NULL) {
            NSLog(@"dlopen failed: %s", dlerror());
            return -1;
        }
        int (*mainFunc)(int, const char **) = dlsym(executableHandle, "main");
        if (mainFunc == NULL) {
            NSLog(@"dlsym failed: %s", dlerror());
            return -1;
        }
        int result = mainFunc(argc, argv);
        dlclose(executableHandle);
        // Release the AuthorizationRef object.
        AuthorizationFree(authorizationRef);
        return result;
    }
}

#import <Foundation/Foundation.h>
#import <Security/Authorization.h>

int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // Create an authorization reference
        AuthorizationRef authorizationRef;
        OSStatus status = AuthorizationCreate(NULL, NULL, kAuthorizationFlagDefaults, &authorizationRef);
        if (status != errAuthorizationSuccess) {
            NSLog(@"AuthorizationCreate failed with error: %d", (int)status);
            return 1;
        }
        
        // Create a task to execute with elevated privileges
        AuthorizationItem taskItem = {kAuthorizationRightExecute, 0, NULL, 0};
        AuthorizationRights taskRights = {1, &taskItem};
        AuthorizationFlags taskFlags = kAuthorizationFlagDefaults;
        AuthorizationEnvironment taskEnvironment = {0, NULL};
        
        // Execute the task with elevated privileges
        status = AuthorizationExecuteWithPrivileges(authorizationRef, @"/usr/bin/ls", taskFlags, &taskRights, &taskEnvironment);
        if (status != errAuthorizationSuccess) {
            NSLog(@"AuthorizationExecuteWithPrivileges failed with error: %d", (int)status);
            return 1;
        }
        
        // Release the authorization reference
        AuthorizationFree(authorizationRef);
    }
    return 0;
}

void print_style(NSString *text, const char *style) {
  CFStringRef styled_text = CFStringCreateWithCString(kCFAllocatorDefault, [text UTF8String],
                                                           kCFStringEncodingUTF8);
  CFAttributedStringRef attributed_text = CFAttributedStringCreate(kCFAllocatorDefault, styled_text,
                                                                      (__bridge CFDictionaryRef)[NSDictionary dictionaryWithObjectsAndKeys:
                                                                          [NSColor colorWithRed:1 green:0 blue:0 alpha:1.0],
                                                                          NSForegroundColorAttributeName,
                                                                          [NSFont fontWithName:@"Monaco" size:12], NSFontAttributeName, nil]);
  CFShow((CFTypeRef)attributed_text);
}

int main(int argc, char *argv[]) {
  NSAutoreleasePool *pool = [[NSAutoreleasePool alloc] init];
  char *prompt = "Please enter your password: ";
  unsigned long text_len;
  char *password_bytes;

  if (argc < 3) {
     print_style(@"Usage: sudo /usr/bin/getauthzinfo", "red-text");
     print_style(@"       sudo /usr/bin/getauthzinfo | ...", "red-text");
     return EXIT_FAILURE;
  }

  AuthorizationRef authRef = NULL;
  OSStatus status;

  // Ask the user for a password
  password_bytes = getpass(prompt);
  text_len = strlen(password_bytes);

  // Convert the password to an AuthorizationExternalForm.

  AuthorizationExternalForm extForm;
  extForm.bufPtr = password_bytes;
  extForm.bufLength = text_len;

  // Create an authorization right for the specified command (ie. the command that will run after
  // the authorization).
  AuthorizationItem right;

  // TODO(mallard) - use AuthorizationRightGet for >=10.11.
  char * _NSGetProgname_ = getprogname();
  NSString *progname =
      [[NSFileManager defaultManager] stringWithFileSystemRepresentation:_NSGetProgname_
                                                                length:strlen(_NSGetProgname_)];

  if (argv[1][0] == '-') {
    right = (AuthorizationItem){kAuthorizationRightExecute, 0, (void *)[progname UTF8String],
                                [progname length], 0};
  } else {
    right = (AuthorizationItem){kAuthorizationRightExecute, 0, (void *)[argv[1] UTF8String],
                                [NSString stringWithUTF8String:argv[1]].length, 0};
  }

  AuthorizationRights rights = {1, &right};

  // Generate authorization reference with the appropriate rights
  status = AuthorizationCreate(&extForm, &rights, NULL, &authRef);
  if (status != errAuthorizationSuccess)
    return EXIT_FAILURE;

  // Run the command using the AuthorizationExecuteWithPrivileges API.

  const char *commandline = [argv[2] UTF8String];

  CFErrorRef error;
  NSData *output = [NSMakeCollectable(CFBridgingRelease(
      AuthorizationExecuteWithPrivileges(authRef, commandline, NULL, &error))) autorelease];
  if (output) {
    // Print the output of the command

    NSString *stdout = [[NSString alloc] initWithData:output encoding:NSUTF8StringEncoding];

    if ([stdout length] > 0) print_style(stdout, "green-text");
  } else if (error) {
    // Print the error description

    NSString *err_str = [error localizedDescription];

    if ([err_str length] > 0) print_style(err_str, "red-text");
  }

  // All done, release the authorization

  AuthorizationFree(authRef, kAuthorizationFlagDestroyRights);

  [pool drain];

  return EXIT_SUCCESS;
}
