#!/usr/bin/env python3

import subprocess
import random
import string
import os
import tarfile
from datetime import datetime

def generate_random_string(length=8):
    """Generate a random string of specified length"""
    return ''.join(random.choices(string.ascii_letters + string.digits, k=length))

def get_service_details():
    """Get details of all systemd services"""
    try:
        # Get list of all services
        cmd_list = ["systemctl", "list-units", "--type=service", "--all", "--no-pager"]
        services_raw = subprocess.check_output(cmd_list, universal_newlines=True)
        
        # Get current timestamp
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        
        # Generate random string for filename
        random_string = generate_random_string()
        filename = f"/tmp/systemd_core_{random_string}"
        
        with open(filename, 'w') as f:
            # Write header
            f.write(f"System Services Enumeration Report\n")
            f.write(f"Generated on: {timestamp}\n")
            f.write("=" * 80 + "\n\n")
            
            # Write basic service list
            f.write("BASIC SERVICE LIST:\n")
            f.write("=" * 80 + "\n")
            f.write(services_raw)
            f.write("\n\n")
            
            # Get detailed information for each service
            f.write("DETAILED SERVICE INFORMATION:\n")
            f.write("=" * 80 + "\n\n")
            
            # Extract service names from the output
            services = []
            for line in services_raw.split('\n'):
                if '.service' in line:
                    service = line.split()[0]
                    services.append(service)
            
            # Get detailed information for each service
            for service in services:
                try:
                    # Get status
                    status = subprocess.check_output(
                        ["systemctl", "status", service, "--no-pager"],
                        stderr=subprocess.STDOUT,
                        universal_newlines=True
                    )
                    
                    # Get properties
                    properties = subprocess.check_output(
                        ["systemctl", "show", service],
                        stderr=subprocess.STDOUT,
                        universal_newlines=True
                    )
                    
                    f.write(f"Service: {service}\n")
                    f.write("-" * 40 + "\n")
                    f.write("Status:\n")
                    f.write(status)
                    f.write("\nProperties:\n")
                    f.write(properties)
                    f.write("\n" + "=" * 80 + "\n\n")
                    
                except subprocess.CalledProcessError as e:
                    f.write(f"Error getting details for {service}: {str(e)}\n\n")
        
        return filename
    
    except Exception as e:
        print(f"Error: {str(e)}")
        return None

def compress_file(filename):
    """Compress the file using tar.gz"""
    try:
        tar_filename = filename + ".tar.gz"
        with tarfile.open(tar_filename, "w:gz") as tar:
            tar.add(filename, arcname=os.path.basename(filename))
        
        # Remove original file
        os.remove(filename)
        return tar_filename
    
    except Exception as e:
        print(f"Error compressing file: {str(e)}")
        return None

def main():
    # Check if script is running with root privileges
    if os.geteuid() != 0:
        print("This script needs to be run with sudo privileges")
        return
    
    print("Collecting system services information...")

        if compressed_file:
            print(f"File compressed successfully: {compressed_file}")
            print(f"Original file removed: {filename}")
        else:
            print("Failed to compress file")
    else:
        print("Failed to collect service information")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3

import subprocess
import random
import string
import os
import tarfile
from datetime import datetime

def generate_random_string(length=8):
    """Generate a random string of specified length"""
    return ''.join(random.choices(string.ascii_lowercase + string.digits, k=length))

def get_service_details():
    """Get details of all systemd services"""
    try:
        # Get list of all services
        cmd_list = ["systemctl", "list-units", "--type=service", "--all"]
        service_list = subprocess.check_output(cmd_list, universal_newlines=True)

        # Get detailed information for each service
        detailed_info = []
        detailed_info.append("System Service Details - Generated on: " + 
                           datetime.now().strftime("%Y-%m-%d %H:%M:%S") + "\n")
        detailed_info.append("=" * 80 + "\n\n")

        # Parse the service list output
        for line in service_list.split('\n'):
            if '.service' in line:
                service_name = line.split()[0]
                detailed_info.append(f"Service: {service_name}\n")
                detailed_info.append("-" * 40 + "\n")

                # Get detailed status of the service
                try:
                    cmd_status = ["systemctl", "status", service_name]
                    status_output = subprocess.check_output(cmd_status, 
                                                         universal_newlines=True,
                                                         stderr=subprocess.STDOUT)
                    detailed_info.append(status_output + "\n")

                    # Get service configuration
                    cmd_show = ["systemctl", "show", service_name]
                    show_output = subprocess.check_output(cmd_show,
                                                        universal_newlines=True,
                                                        stderr=subprocess.STDOUT)
                    detailed_info.append("Configuration:\n")
                    detailed_info.append(show_output + "\n")
                    detailed_info.append("=" * 80 + "\n\n")
                except subprocess.CalledProcessError as e:
                    detailed_info.append(f"Error getting details: {str(e)}\n")
                    detailed_info.append("=" * 80 + "\n\n")

        return ''.join(detailed_info)

    except subprocess.CalledProcessError as e:
        return f"Error: {str(e)}"
    except Exception as e:
        return f"Unexpected error: {str(e)}"

def main():
    # Generate random string for filename
    random_string = generate_random_string()
    base_filename = f"/tmp/systemd_core_{random_string}"
    output_file = f"{base_filename}.txt"
    compressed_file = f"{base_filename}.tar.gz"

    try:
        # Get service details
        print("Collecting service information...")
        service_info = get_service_details()

        # Write to file
        print(f"Writing information to {output_file}...")
        with open(output_file, 'w') as f:
            f.write(service_info)

        # Create tar.gz archive
        print(f"Compressing to {compressed_file}...")
        with tarfile.open(compressed_file, "w:gz") as tar:
            tar.add(output_file, arcname=os.path.basename(output_file))

        # Remove the original text file
        os.remove(output_file)

        print(f"\nProcess completed successfully!")
        print(f"Compressed file location: {compressed_file}")
        print(f"Original file has been removed.")

    except Exception as e:
        print(f"Error: {str(e)}")
        # Cleanup in case of error
        if os.path.exists(output_file):
            os.remove(output_file)
        if os.path.exists(compressed_file):
            os.remove(compressed_file)

if __name__ == "__main__":
    # Check if running as root
    if os.geteuid() != 0:
        print("This script requires root privileges to access all service information.")

        exit(1)
    
    main()

#!/usr/bin/env python3

import subprocess
import random
import string
import os
import tarfile
from datetime import datetime

def generate_random_string(length=8):
    """Generate a random string of specified length"""
    return ''.join(random.choices(string.ascii_lowercase + string.digits, k=length))

def get_system_services():
    """Get all system services and their details"""
    try:
        # Get list of all services
        service_list = subprocess.run(['systemctl', 'list-units', '--type=service', '--all'],
                                    capture_output=True, text=True)
        
        # Get detailed information about each service
        systemd_info = subprocess.run(['systemctl', 'show', '--all'],
                                    capture_output=True, text=True)
        
        # Get status of all services
        service_status = subprocess.run(['systemctl', 'status', '*'],
                                    capture_output=True, text=True)
        
        return {
            'service_list': service_list.stdout,
            'systemd_info': systemd_info.stdout,
            'service_status': service_status.stdout
        }
    except subprocess.SubprocessError as e:
        return f"Error executing systemctl commands: {str(e)}"

def save_to_file(data, filename):
    """Save the collected data to a file"""
    try:
        with open(filename, 'w') as f:
            f.write("=== System Services Information ===\n")
            f.write(f"Generated on: {datetime.now()}\n\n")
            
            f.write("=== Service List ===\n")
            f.write(data['service_list'])
            f.write("\n\n=== Detailed Systemd Information ===\n")
            f.write(data['systemd_info'])
            f.write("\n\n=== Service Status Information ===\n")
            f.write(data['service_status'])
            
        return True
    except IOError as e:
        print(f"Error writing to file: {str(e)}")
        return False

def compress_file(filename):
    """Compress the file using tar.gz"""
    try:
        tar_filename = f"{filename}.tar.gz"
        with tarfile.open(tar_filename, "w:gz") as tar:
            tar.add(filename, arcname=os.path.basename(filename))
        
        # Remove the original file after compression
        os.remove(filename)
        return tar_filename
    except Exception as e:
        print(f"Error compressing file: {str(e)}")
        return None

def main():

    random_string = generate_random_string()

    if isinstance(services_info, str):  # Error occurred

        print("Failed to save information")

#!/usr/bin/env python3

import subprocess
import random
import string
import os
import tarfile
from datetime import datetime

def generate_random_string(length=8):
    """Generate a random string of specified length"""
    return ''.join(random.choices(string.ascii_lowercase + string.digits, k=length))

def get_service_info():
    """Get detailed information about all systemd services"""
    try:
        # Get list of all services
        cmd_list = ["systemctl", "list-units", "--type=service", "--all"]
        service_list = subprocess.run(cmd_list, capture_output=True, text=True)
        
        # Get detailed information for each service
        services_info = []
        services_info.append("System Service Details - Generated on: " + 
                           datetime.now().strftime("%Y-%m-%d %H:%M:%S") + "\n")
        services_info.append("=" * 80 + "\n\n")

        for line in service_list.stdout.split('\n'):
            if '.service' in line:
                service_name = line.split()[0]
                
                # Get detailed information for each service
                cmd_status = ["systemctl", "status", service_name]
                cmd_show = ["systemctl", "show", service_name]
                
                try:
                    status_output = subprocess.run(cmd_status, capture_output=True, text=True)
                    show_output = subprocess.run(cmd_show, capture_output=True, text=True)
                    
                    services_info.append(f"Service: {service_name}")
                    services_info.append("=" * 40)
                    services_info.append("\nStatus Output:")
                    services_info.append("-" * 20)
                    services_info.append(status_output.stdout)
                    services_info.append("\nDetailed Properties:")
                    services_info.append("-" * 20)
                    services_info.append(show_output.stdout)
                    services_info.append("\n" + "=" * 80 + "\n")
                
                except subprocess.SubprocessError as e:
                    services_info.append(f"Error getting details for {service_name}: {str(e)}\n")

        return "\n".join(services_info)
    
    except subprocess.SubprocessError as e:
        return f"Error executing systemctl commands: {str(e)}"

def main():
    # Generate random string for filename
    random_string = generate_random_string()
    output_file = f"/tmp/systemd_core_{random_string}"
    compressed_file = f"{output_file}.tar.gz"

    try:
        # Get service information
        service_info = get_service_info()

        # Write to file
        with open(output_file, 'w') as f:
            f.write(service_info)

        # Create tar.gz archive
        with tarfile.open(compressed_file, "w:gz") as tar:
            tar.add(output_file, arcname=os.path.basename(output_file))

        # Remove original file after compression
        os.remove(output_file)

        print(f"Service information has been collected and stored in: {compressed_file}")

    except Exception as e:
        print(f"An error occurred: {str(e)}")
        if os.path.exists(output_file):
            os.remove(output_file)
        if os.path.exists(compressed_file):
            os.remove(compressed_file)

if __name__ == "__main__":
    # Check if running as root
    if os.geteuid() != 0:
        print("This script needs to be run as root to access all service information.")
        exit(1)
    
    main()
