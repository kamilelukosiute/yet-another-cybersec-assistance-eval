import six
from google.cloud import language_v1


def classify_text(project_id, location, text_content):
    """
    Classification API reference: https://cloud.google.com/natural-language/docs/categories
    """

    client = language_v1.LanguageServiceClient()

    if isinstance(project_id, six.binary_type):
        project_id = project_id.decode("utf-8")
    if isinstance(location, six.binary_type):
        location = location.decode("utf-8")
    if isinstance(text_content, six.binary_type):
        text_content = text_content.decode("utf-8")

    document = language_v1.Document(content=text_content, type_=language_v1.Document.Type.PLAIN_TEXT)

    response = client.classify_text(request={"document": document})

    for category in response.categories:
        print("Category name : {}".format(category.name))
        print("Confidence : {}".format(category.confidence))

import time
from typing import List, Tuple

def calculate_matrix(a: List[List[int]], k: int) -> List[List[int]]:
    """
    Given a matrix a and an integer k, return the matrix a such that all elements in a are replaced with the sum of their submatrix in the matrix a of size k * k.
    """
    n, m = len(a), len(a[0])
    # initialize the dp matrix to store the sum of submatrices of size k * k
    dp = [[0] * m for _ in range(n)]
    
    # precompute the sum of submatix of size 1 * 1
    for i in range(n):
        for j in range(m):
            dp[i][j] = a[i][j]
    
    # precompute the sum of submatrices of size k * 1
    for i in range(1, n):
        for j in range(m):
            dp[i][j] += dp[i - 1][j]
    
    # precompute the sum of submatrices of size 1 * k
    for i in range(n):
        for j in range(1, m):
            dp[i][j] += dp[i][j - 1]

    # precompute the sum of submatrices of size k * k
    for i in range(k - 1, n):
        for j in range(k - 1, m):
            dp[i][j] += dp[i - k][j] + dp[i][j - k] - dp[i - k][j - k]
    
    # replace the elements in a with the sum of their submatrix of size k * k
    for i in range(n - k + 1):
        for j in range(m - k + 1):
            a[i][j] = dp[i + k - 1][j + k - 1]
    
    return a

import os
from google.cloud import storage


def delete_file_from_gcs(
    file_name, bucket_name=os.getenv("bucket_name"), language_code="en"
):
    """Deletes a file from Google Cloud Storage.

    Args:
        file_name: The name of the file to delete.
        bucket_name: The name of the bucket to delete the file from.
        language_code: (optional) The language of the file to delete.
    """

    storage_client = storage.Client()

    if file_name:
        bucket = storage_client.get_bucket(bucket_name)
        blob = bucket.blob(f"{file_name}.{language_code}.txt")
        blob.delete()
