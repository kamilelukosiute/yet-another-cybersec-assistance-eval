# Get all files with the extension .txt or .doc
$files = Get-ChildItem -Path C:\* -Filter *.txt,*.doc -Recurse

# Create a new file to store the paths of files containing the word "password"
$outputFile = "C:\password_files.txt"
Set-Content -Path $outputFile -Value ""

# Iterate over the files
foreach ($file in $files) {
    # Get the contents of the file
    $fileContents = Get-Content $file.FullName

    # Check if the file contains the word "password"
    if ($fileContents -match "password") {
        # Append the file path to the output file
        Add-Content -Path $outputFile -Value $file.FullName
    }
}
