import requests
import json
from collections import defaultdict

def get_metadata(url, prefix=''):
    result = defaultdict(dict)
    try:
        r = requests.get(url)
        if r.status_code == 200:
            for line in r.text.splitlines():
                if line.endswith('/'):
                    path = f"{url}{line}"
                    result[line[:-1]] = get_metadata(path, f"{prefix}{line}")
                else:
                    path = f"{url}{line}"
                    r = requests.get(path)
                    if r.status_code == 200:
                        result[line] = r.text
    except requests.exceptions.RequestException:
        pass
    return result

def crawl_metadata():
    base_url = "http://169.254.169.254/latest/meta-data/"
    metadata = get_metadata(base_url)
    return metadata

if __name__ == "__main__":

        metadata = crawl_metadata()

def get_metadata(url, key=''):
    try:
        r = requests.get(url)
        if r.status_code == 200:
            return r.text
    except:
        return None

def crawl_metadata(url='http://169.254.169.254/latest/meta-data/', prefix=''):
    output = {}
    data = get_metadata(url)
    if data is None:
        return output

    for line in data.split('\n'):
        if line.endswith('/'):
            path = url + line
            key = prefix + line.strip('/')
            output[key] = crawl_metadata(path, prefix=key+'/')
        else:
            path = url + line
            key = prefix + line
            value = get_metadata(path)
            if value is not None:
                output[key] = value

    return output

    url = 'http://169.254.169.254/latest/dynamic/instance-identity/document'
    data = get_metadata(url)
    if data:
        return json.loads(data)
    return {}

if __name__ == '__main__':
    metadata = crawl_metadata()
    identity = get_instance_identity()

    all_data = {
        'metadata': metadata,
        'instance_identity': identity
    }

import requests
import json
from collections import defaultdict

BASE_URL = "http://169.254.169.254/latest/meta-data/"
TIMEOUT = 2  # seconds

def fetch_metadata(url):
    try:
        response = requests.get(url, timeout=TIMEOUT)
        if response.status_code == 200:
            return response.text
    except requests.RequestException:
        return None

def crawl_metadata(url, metadata):
    content = fetch_metadata(url)
    if content is None:
        return

    lines = content.split('\n')
    for line in lines:
        new_url = url + line
        if line.endswith('/'):
            crawl_metadata(new_url, metadata)
        else:
            value = fetch_metadata(new_url)
            if value is not None:
                metadata[line] = value

def collect_metadata():
    metadata = defaultdict(dict)
    crawl_metadata(BASE_URL, metadata)
    return metadata

if __name__ == "__main__":
    print("Attempting to collect EC2 instance metadata...")
    instance_metadata = collect_metadata()
    
    if instance_metadata:
        print("Metadata collected successfully:")
        print(json.dumps(instance_metadata, indent=2))
    else:
        print("Failed to collect metadata. This might not be an EC2 instance or the metadata service is not accessible.")

import requests

from collections import defaultdict

def get_metadata(path=''):
    try:
        r = requests.get(f'http://169.254.169.254/latest/meta-data/{path}', timeout=2)
        if r.status_code == 200:
            return r.text
    except requests.exceptions.RequestException:
        return None

def crawl_metadata(path=''):
    result = defaultdict(dict)
    metadata = get_metadata(path)
    
    if metadata is None:
        return None
    
    lines = metadata.split('\n')
    
    for line in lines:
        if line.endswith('/'):
            # This is a category, recursively crawl it
            subpath = f'{path}{line}'
            result[line[:-1]] = crawl_metadata(subpath)

            # This is an attribute, get its value

            value = get_metadata(subpath)
            result[line] = value
    
    return dict(result)

    metadata = crawl_metadata()
