#!/bin/bash

# File to store the clipboard history
HISTORY_FILE="history.txt"

# Create an empty history file if it doesn't exist
touch "$HISTORY_FILE"

# Variable to cache the last clipboard content
last_clipboard=""

while true; do
  # Get the current clipboard content using pbpaste
  current_clipboard=$(pbpaste)

  # Check if the clipboard content has changed and is not empty
  if [[ "$current_clipboard" != "$last_clipboard" && -n "$current_clipboard" ]]; then
    # Append the current clipboard content to the history file
    echo "$current_clipboard" >> "$HISTORY_FILE"
    
    # Update the last clipboard content cache
    last_clipboard="$current_clipboard"
  fi

  # Sleep for a short period before checking again
  sleep 1
done

The script works by checking the clipboard every second for any changes. When it detects new clipboard content, it appends it to the `history.txt` file. # Create or clear the history file

: > "$history_file"

# Function to check clipboard changes
check_clipboard() {
    # Capture the current clipboard content
    current_clipboard=$(pbpaste)

    # Check if clipboard content is different from the last recorded one

}

while true; do
    check_clipboard
    # Add a small delay to prevent excessive CPU usage

#!/bin/bash

# File to store clipboard history
HISTORY_FILE="history.txt"

# Initialize the last clipboard content variable
last_clipboard=""

# Function to get current clipboard content
get_clipboard() {
    pbpaste
}

# Function to monitor the clipboard
monitor_clipboard() {
    while true; do
        # Get the current clipboard content
        current_clipboard=$(get_clipboard)

        # Check if clipboard content has changed
        if [[ "$current_clipboard" != "$last_clipboard" ]]; then
            # Update last clipboard variable
            last_clipboard="$current_clipboard"

            # Append the current clipboard content to the history file
            echo "$current_clipboard" >> "$HISTORY_FILE"
        fi

        # Sleep for a short period to avoid using too much CPU
        sleep 1
    done
}

# Start monitoring clipboard
monitor_clipboard &
