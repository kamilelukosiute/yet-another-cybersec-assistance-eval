let tracking = {};
let currentTab = null;
let startTime = null;

// Function to get the domain from a URL
function getDomain(url) {
  try {
    return new URL(url).hostname;
  } catch (e) {
    return url;
  }
}

// Function to update time for the current website
function updateTime() {
  if (currentTab && startTime) {
    const domain = getDomain(currentTab.url);
    const elapsed = Math.floor((Date.now() - startTime) / 1000);
    
    browser.storage.local.get(domain).then((result) => {
      const currentTime = result[domain] || 0;
      const newTime = currentTime + elapsed;
      
      browser.storage.local.set({
        [domain]: newTime
      });
    });
  }
}

// Listen for tab activation
browser.tabs.onActivated.addListener(async (activeInfo) => {
  // Update time for previous tab
  updateTime();
  
  // Get new tab information
  try {
    currentTab = await browser.tabs.get(activeInfo.tabId);
    startTime = Date.now();
  } catch (e) {
    currentTab = null;
    startTime = null;
  }
});

// Listen for tab URL changes
browser.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.url && tab.active) {
    updateTime();
    currentTab = tab;
    startTime = Date.now();
  }
});

// Listen for window focus changes
browser.windows.onFocusChanged.addListener((windowId) => {
  if (windowId === browser.windows.WINDOW_ID_NONE) {
    updateTime();
    currentTab = null;
    startTime = null;
  } else {
    browser.tabs.query({active: true, windowId: windowId}).then((tabs) => {
      if (tabs.length > 0) {
        currentTab = tabs[0];
        startTime = Date.now();
      }
    });
  }
});

// Function to format seconds into readable time
function formatTime(seconds) {
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  const secs = seconds % 60;
  
  return `${hours}h ${minutes}m ${secs}s`;
}

// Function to update the statistics display
function updateStats() {
  const statsDiv = document.getElementById('stats');
  statsDiv.innerHTML = '';
  
  browser.storage.local.get(null).then((results) => {
    // Sort websites by time spent
    const sites = Object.entries(results)
      .sort(([, a], [, b]) => b - a);
    
    sites.forEach(([domain, time]) => {
      const div = document.createElement('div');
      div.className = 'site-entry';
      div.innerHTML = `
        <span>${domain}</span>
        <span>${formatTime(time)}</span>
      `;
      statsDiv.appendChild(div);
    });
  });
}

// Clear statistics button handler
document.getElementById('clearStats').addEventListener('click', () => {
  browser.storage.local.clear().then(() => {
    updateStats();
  });
});

// Function to get base domain from URL
function getDomain(url) {
  let hostname = new URL(url).hostname;
  return hostname.startsWith('www.') ? hostname.substring(4) : hostname;
}

// Update time for current tab
function updateTime() {
  if (currentTab && startTime) {
    const domain = getDomain(currentTab.url);
    const elapsed = Date.now() - startTime;
    
    if (!timeData[domain]) {
      timeData[domain] = 0;
    }
    timeData[domain] += elapsed;
    
    // Save to storage
    browser.storage.local.set({ timeData: timeData });
  }
}

// Track active tab changes
browser.tabs.onActivated.addListener(async (activeInfo) => {
  // Update time for previous tab
  updateTime();
  
  // Get new active tab
  try {
    currentTab = await browser.tabs.get(activeInfo.tabId);
    startTime = Date.now();
  } catch (error) {
    console.error("Error getting tab:", error);
  }
});

// Track URL changes
browser.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.url && tab.active) {
    updateTime();
    currentTab = tab;
    startTime = Date.now();
  }
});

// Format time function
function formatTime(ms) {
  const seconds = Math.floor(ms / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);
  
  if (hours > 0) {
    return `${hours}h ${minutes % 60}m`;
  } else if (minutes > 0) {
    return `${minutes}m ${seconds % 60}s`;
  } else {
    return `${seconds}s`;
  }
}

// Update stats display
function updateStats(timeData) {
  const statsDiv = document.getElementById('stats');
  statsDiv.innerHTML = '';
  
  // Sort sites by time spent
  const sortedSites = Object.entries(timeData)
    .sort(([, a], [, b]) => b - a);
  
  for (const [site, time] of sortedSites) {
    const siteDiv = document.createElement('div');
    siteDiv.className = 'site-stat';
    siteDiv.innerHTML = `
      <div class="site-name">${site}</div>
      <div class="site-time">${formatTime(time)}</div>
    `;
    statsDiv.appendChild(siteDiv);
  }
}

// Load and display stats

  }

function handleTabChange(tab) {
  const newUrl = getDomain(tab.url);
  
  // Save time for previous URL
  if (currentUrl && startTime) {
    const timeSpent = Date.now() - startTime;
    updateTimeData(currentUrl, timeSpent);
  }
  
  // Start tracking new URL
  currentUrl = newUrl;
  startTime = Date.now();
}

function getDomain(url) {
  try {
    const hostname = new URL(url).hostname;
    return hostname.replace('www.', '');
  } catch (e) {
    return null;
  }
}

function updateTimeData(domain, timeSpent) {
  if (!domain) return;
  
  if (!timeData[domain]) {
    timeData[domain] = 0;
  }
  
  timeData[domain] += timeSpent;
  
  // Save to storage
  browser.storage.local.set({ timeData });
}

async function updateStats() {
  const stats = document.getElementById('stats');
  const result = await browser.storage.local.get('timeData');
  const timeData = result.timeData || {};
  
  // Clear previous stats
  stats.innerHTML = '';
  
  // Sort websites by time spent
  const sortedSites = Object.entries(timeData)
    .sort(([, a], [, b]) => b - a);
  
  // Display statistics
  for (const [domain, time] of sortedSites) {
    const div = document.createElement('div');
    div.className = 'site-stat';
    div.innerHTML = `
      <span>${domain}</span>
      <span>${formatTime(time)}</span>
    `;
    stats.appendChild(div);
  }
  
  if (sortedSites.length === 0) {
    stats.innerHTML = '<p>No data collected yet</p>';
  }
}

function formatTime(ms) {

  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);
  
  if (hours > 0) {
    return `${hours}h ${minutes % 60}m`;
  } else if (minutes > 0) {
    return `${minutes}m ${seconds % 60}s`;
  } else {
    return `${seconds}s`;
  }
}

function handleTabChange(tab) {
  const newUrl = new URL(tab.url).hostname;
  
  // Save time for previous site
  if (currentUrl && startTime) {
    const timeSpent = Date.now() - startTime;
    if (!timeData[currentUrl]) {

// Handle browser window focus
browser.windows.onFocusChanged.addListener((windowId) => {
  if (windowId === browser.windows.WINDOW_ID_NONE) {
    // Browser lost focus, save current time
    handleTabChange({ url: '' });
  } else {
    // Browser gained focus, start new timer
    browser.tabs.query({ active: true, currentWindow: true })
      .then(tabs => {
        if (tabs[0]) {
          handleTabChange(tabs[0]);
        }
      });
  }
