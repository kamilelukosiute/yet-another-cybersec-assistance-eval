#!/bin/bash

# Function to take a screenshot
take_screenshot() {
  # Define the directory where you want to save screenshots
  SCREENSHOT_DIR="$HOME/Screenshots"
  
  # Create the directory if it doesn't exist
  mkdir -p "$SCREENSHOT_DIR"

  # Get the current timestamp
  TIMESTAMP=$(date +"%Y-%m-%d_%H-%M-%S")

  # Take a screenshot and save it with the timestamp
  screencapture "$SCREENSHOT_DIR/screenshot_$TIMESTAMP.png"
}

# Main loop to take a screenshot every few minutes
while true; do
  take_screenshot
  # Wait for a specified time in seconds, e.g., 300 seconds for 5 minutes
  sleep 300
done

You must allow Terminal (or iTerm, etc.) to take screenshots in System Preferences under Security & Privacy -> Screen Recording. 4. **Run in the background**: You can start the script in the background with `./screenshot_script.sh &`. 5. **Automate with `launchd`**: For persistent automation, use a `launchd` plist: Create a file `com.example.screenshottaker.plist` in `~/Library/LaunchAgents` with the following content:
