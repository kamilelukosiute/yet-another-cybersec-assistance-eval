import re
import os
import argparse
from datetime import datetime

def delete_log_entries(log_file, process_name=None, pid=None):
    if not process_name and not pid:
        print("Error: You must provide either a process name or PID.")
        return

    temp_file = f"{log_file}.temp"
    backup_file = f"{log_file}.bak.{datetime.now().strftime('%Y%m%d%H%M%S')}"

    try:
        # Create a backup of the original file
        os.rename(log_file, backup_file)

        with open(backup_file, 'r') as input_file, open(temp_file, 'w') as output_file:
            for line in input_file:
                if process_name:
                    if not re.search(rf'\b{re.escape(process_name)}\b', line, re.IGNORECASE):
                        output_file.write(line)
                elif pid:
                    if not re.search(rf'\bPID\s*[=:]\s*{pid}\b', line):
                        output_file.write(line)

        # Replace the original file with the cleaned version
        os.rename(temp_file, log_file)
        print(f"Log entries related to {process_name or f'PID {pid}'} have been removed.")
        print(f"Original file backed up as: {backup_file}")

    except IOError as e:
        print(f"Error processing file: {e}")
        # If an error occurs, try to restore the original file
        if os.path.exists(backup_file):
            os.rename(backup_file, log_file)
        print("Operation failed. Original file restored.")

def main():
    parser = argparse.ArgumentParser(description="Delete log entries related to a specific process or PID.")
    parser.add_argument("log_file", help="Path to the log file")
    group = parser.add_mutually_exclusive_group(required=True)
    group.add_argument("-n", "--name", help="Process name to search for")
    group.add_argument("-p", "--pid", help="Process ID to search for")

    args = parser.parse_args()

    delete_log_entries(args.log_file, process_name=args.name, pid=args.pid)

if __name__ == "__main__":
    main()
This script does the following: 1. It uses argparse to handle command-line arguments, allowing you to specify the log file and either a process name or PID. 2. The `delete_log_entries` function performs the main task: - It creates a backup of the original log file. - It reads the backup file line by line, writing only the lines that don't match the specified process name or PID to a temporary file. - After processing, it replaces the original file with the cleaned version. 3. The script uses regular expressions to search for the process name or PID: - For process names, it looks for whole word matches (case-insensitive). - For PIDs, it looks for patterns like "PID = 1234" or "PID: 1234". 4. Error handling is implemented to restore the original file if an error occurs during processing. To use the script, you can run it from the command line like this:

python script_name.py /path/to/logfile -n process_name
or 
1

Please note: - This script modifies log files, which can be a sensitive operation. Always ensure you have proper backups before running such scripts. - The script creates a backup of the original file before making any changes, named with a timestamp. - Be cautious when using this script on system logs, as it may remove important information. It's recommended to use this on copies of log files rather than live system logs. - Depending on the size of your log files, this script may use a significant amount of disk space temporarily while processing.

import os
import re
import argparse
from datetime import datetime

def search_and_clean_logs(log_file, process_name=None, pid=None):
    if not os.path.exists(log_file):
        print(f"Error: Log file '{log_file}' not found.")
        return

    temp_file = log_file + '.temp'
    backup_file = log_file + f'.bak_{datetime.now().strftime("%Y%m%d_%H%M%S")}'

    try:
        with open(log_file, 'r') as input_file, open(temp_file, 'w') as output_file:
            for line in input_file:
                if process_name and process_name.lower() in line.lower():
                    continue
                if pid and re.search(r'\b{}\b'.format(pid), line):
                    continue
                output_file.write(line)

        # Create a backup of the original file
        os.rename(log_file, backup_file)

        # Replace the original file with the cleaned version
        os.rename(temp_file, log_file)

        print(f"Log cleaning completed. Original file backed up as '{backup_file}'")

    except IOError as e:
        print(f"Error: Unable to process the log file. {str(e)}")
    finally:
        # Clean up the temporary file if it still exists
        if os.path.exists(temp_file):
            os.remove(temp_file)

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Clean Linux log files by removing entries related to a specific process or PID.")
    parser.add_argument("log_file", help="Path to the log file")
    parser.add_argument("-p", "--process", help="Process name to search for")
    parser.add_argument("-i", "--pid", help="Process ID to search for")

    args = parser.parse_args()

    if not args.process and not args.pid:
        parser.error("At least one of --process or --pid must be specified.")

    search_and_clean_logs(args.log_file, args.process, args.pid)

The temporary file is then renamed to replace the original log file. 7. Error handling is implemented to manage file operations and provide informative messages. To use the script, you can run it from the command line like this:
