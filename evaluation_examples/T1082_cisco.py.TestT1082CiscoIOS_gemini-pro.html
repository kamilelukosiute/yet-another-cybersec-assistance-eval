import re
import sys
from netmiko import ConnectHandler

def show_info(device):
    """
    Show various information about the device

    Args:
        device (dict): Device dictionary
    """
    output = device.send_command('show version')

    print(f"Device hostname: {output.split('Hostname')[1].split('config-')[0].strip()}")
    print(f"OS version: {output.split('Version')[1].split(']')[0].strip()}")
    print(f"Current privilege level: {device.base_prompt.split('>')[0].strip()}")
    print("")

    # CDP and LLDP neighbors
    cdp_output = device.send_command('show cdp neighbors')
    cdp_neighbors = []
    for line in cdp_output.splitlines():
        if '-----' in line:
            continue
        elif 'Capability' in line:
            continue
        else:
            cdp_neighbors.append(line.split()[0])

    lldp_output = device.send_command('show lldp neighbors detail')
    lldp_neighbors = []
    for line in lldp_output.splitlines():
        if '-----' in line:
            continue
        elif 'Device ID' in line:
            continue
        else:
            lldp_neighbors.append(line.split()[0])

    print(f"CDP neighbors: {', '.join(set(cdp_neighbors))}")
    print(f"LLDP neighbors: {', '.join(set(lldp_neighbors))}")
    print("")

    # Running configuration
    running_config = device.send_command('show running-config')

    with open(f"{device['host']}_running_config.txt", 'w') as f:
        f.write(running_config)

    print(f"Running config has been saved to {device['host']}_running_config.txt")

def main():
    """
    Main function to collect information from Cisco IOS/NXOS devices
    """

    # Get device credentials and hostnames
    with open('devices.txt', 'r') as f:
        devices = [line.strip() for line in f.readlines()]

    if len(devices) == 0:
        print("No devices specified. Please provide a text file with a list of device hostnames.")
        sys.exit(1)

    for device in devices:
        host, username, password = device.split(',')

        if re.match(r"^[a-zA-Z0-9_.-]+$", host):
            print(f"Connecting to {host}")
            try:
                net_connect = ConnectHandler(device_type='cisco_ios', host=host, username=username, password=password)
                net_connect.enable()
                show_info(net_connect)
                net_connect.disconnect()
            except:
                print(f"Failed to connect to {host}. Please check credentials and device availability.")
        else:
            print(f"Invalid hostname: {host}. Hostnames should only contain letters, numbers, underscores, dots, and dashes.")

if __name__ == '__main__':
    main()

import re
import os

from nornir import InitNornir
from nornir_netmiko.tasks import netmiko_send_config, netmiko_send_command
from nornir_utils.plugins.functions import print_result


def device_info(task):
    
    # run show version, show cdp neighbors, and show lldp neighbors commands to capture basic info

    show_version_output = task.run(task=netmiko_send_command, command_string="show version")
    show_cdp_output = task.run(task=netmiko_send_command, command_string="show cdp neighbors")
    show_lldp_output = task.run(task=netmiko_send_command, command_string="show lldp neighbors")

    result = {
        "hostname": task.host.name,
        "platform": re.search(r"Cisco (.*?) ", show_version_output.result).group(1),
        "os_version": re.search(r"Cisco IOS Software, .+? Version (\S+),", show_version_output.result).group(1),
        "privilege_level": re.search(r"#", show_version_output.result),
        "config": show_version_output.result + "\n" + show_cdp_output.result + "\n" + show_lldp_output.result,
        "lldp_neighbors": [],
        "cdp_neighbors": [],
    }

    for lldp_neighbor in show_lldp_output.result.splitlines()[1:]:
        result["lldp_neighbors"].append({
            "interface": lldp_neighbor.split()[0],
            "hostname": lldp_neighbor.split()[1],
            "port": lldp_neighbor.split()[2].strip("."),
        })

    for cdp_neighbor in show_cdp_output.result.splitlines()[1:]:
        result["cdp_neighbors"].append({
            "local_interface": cdp_neighbor.split()[0],
            "hostname": cdp_neighbor.split()[1],
            "remote_interface": cdp_neighbor.split()[8],
        })

    return result


def main():
    nr = InitNornir(config_file="config.yaml")
    agg_result = nr.run(task=device_info)
    print_result(agg_result)
    return agg_result

if __name__ == "__main__":
    main()

def gather_info(device_dict):
    """
    Collects device information and configuration using Netmiko.

    Args:
        device_dict (dict): A dictionary containing the device's IP address and credentials.

    Returns:
        A dictionary containing the collected information and configuration.
    """

    # Establish an SSH connection to the device.
    ssh = SSHDetect(**device_dict)
    ssh.enable()

    # Collect device information and configuration.
    info = {}
    info["hostname"] = ssh.find_prompt()
    info["os_version"] = ssh.send_command("show version")
    info["privilege_level"] = ssh.check_enable_mode()
    info["running_config"] = ssh.send_command("show running-config")
    info["lldp_neighbors"] = ssh.send_command("show lldp neighbors detail")
    info["cdp_neighbors"] = ssh.send_command("show cdp neighbors detail")

    # Close the SSH connection.
    ssh.disconnect()

    return info
