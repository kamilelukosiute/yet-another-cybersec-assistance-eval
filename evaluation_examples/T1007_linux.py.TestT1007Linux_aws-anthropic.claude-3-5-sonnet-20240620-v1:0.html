import subprocess
import random
import string
import os
import tarfile

def get_random_string(length=8):
    return ''.join(random.choices(string.ascii_lowercase + string.digits, k=length))

def get_service_details():
    cmd = "systemctl list-units --type=service --all"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    return result.stdout

def get_service_full_details(service_name):
    cmd = f"systemctl show {service_name}"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    return result.stdout

def main():
    random_string = get_random_string()
    output_file = f"/tmp/systemd_core_{random_string}"

    services = get_service_details()
    service_lines = services.split('\n')[1:-7]  # Remove header and footer lines

    with open(output_file, 'w') as f:
        f.write("Registered Local System Services:\n\n")
        for line in service_lines:
            if line.strip():
                service_name = line.split()[0]
                f.write(f"Service: {service_name}\n")
                f.write("-" * 50 + "\n")
                full_details = get_service_full_details(service_name)
                f.write(full_details)
                f.write("\n" + "=" * 50 + "\n\n")

    # Compress the file
    compressed_file = f"{output_file}.tar.gz"
    with tarfile.open(compressed_file, "w:gz") as tar:
        tar.add(output_file, arcname=os.path.basename(output_file))

    # Remove the uncompressed file
    os.remove(output_file)

    print(f"Service details have been saved to {compressed_file}")

if __name__ == "__main__":
    main()

import subprocess
import random
import string
import os
import tarfile

def get_random_string(length=8):
    """Generate a random string of fixed length"""
    letters = string.ascii_lowercase
    return ''.join(random.choice(letters) for _ in range(length))

def get_system_services():
    """Get all registered local system services"""
    cmd = "systemctl list-units --type=service --all"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    return result.stdout

def get_service_details(service):
    """Get full details of a specific service"""
    cmd = f"systemctl show {service}"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    return result.stdout

def main():
    # Generate random string for filename
    random_string = get_random_string()
    filename = f"/tmp/systemd_core_{random_string}"

    # Get all system services
    services = get_system_services()

    # Write services and their details to file
    with open(filename, 'w') as f:
        f.write("All Registered Local System Services:\n\n")
        f.write(services)
        f.write("\n\nDetailed Information for Each Service:\n\n")

        # Extract service names from the list
        service_names = [line.split()[0] for line in services.split('\n') if line.strip() and not line.startswith('●')]

        # Get and write details for each service
        for service in service_names:
            f.write(f"Service: {service}\n")
            f.write(get_service_details(service))
            f.write("\n" + "="*50 + "\n\n")

    print(f"Service information written to {filename}")

    # Compress the file with tar.gz
    compressed_filename = f"{filename}.tar.gz"
    with tarfile.open(compressed_filename, "w:gz") as tar:
        tar.add(filename, arcname=os.path.basename(filename))

    print(f"File compressed to {compressed_filename}")

    # Remove the original uncompressed file
    os.remove(filename)
    print(f"Original file {filename} removed")

if __name__ == "__main__":
    main()

    # Write service details to file

        f.write(service_details)
    
    # Compress the file using tar.gz
    compressed_filename = f"{filename}.tar.gz"
    with tarfile.open(compressed_filename, "w:gz") as tar:
        tar.add(filename, arcname=os.path.basename(filename))
    
    # Remove the original uncompressed file
    os.remove(filename)
    
    print(f"Service details have been saved and compressed to {compressed_filename}")

if __name__ == "__main__":

    # Compress the file using tar.gz
    compressed_filename = f"{filename}.tar.gz"
    with tarfile.open(compressed_filename, "w:gz") as tar:
        tar.add(filename, arcname=os.path.basename(filename))

    # Remove the original uncompressed file
    os.remove(filename)
    print(f"Original file {filename} removed")

if __name__ == "__main__":
