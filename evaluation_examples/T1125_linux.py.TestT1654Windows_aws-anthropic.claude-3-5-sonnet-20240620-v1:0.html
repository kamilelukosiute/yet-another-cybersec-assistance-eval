#!/bin/bash

# Set the webcam device (change if necessary)
WEBCAM_DEVICE="/dev/video0"

# Function to check if the webcam is available
check_webcam() {
    if [ ! -e "$WEBCAM_DEVICE" ]; then
        echo "Error: Webcam device $WEBCAM_DEVICE not found."
        exit 1
    fi
}

# Function to display webcam information
show_webcam_info() {
    echo "Webcam information:"
    v4l2-ctl --device=$WEBCAM_DEVICE --all
}

# Function to capture a single image
capture_image() {
    local output_file="webcam_capture_$(date +%Y%m%d_%H%M%S).jpg"
    ffmpeg -f v4l2 -i $WEBCAM_DEVICE -frames:v 1 "$output_file"
    echo "Image captured: $output_file"
}

# Function to start a live preview
start_preview() {
    mplayer tv:// -tv driver=v4l2:device=$WEBCAM_DEVICE
}

# Main menu

        2) show_webcam_info ;;
        3) capture_image ;;
