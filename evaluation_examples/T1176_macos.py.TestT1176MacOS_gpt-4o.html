chrome.tabs.onActivated.addListener((activeInfo) => {
  const switchTime = Date.now();

  chrome.tabs.get(activeInfo.tabId, (tab) => {
    const { url } = tab;
    const hostname = new URL(url).hostname;

    if (activeTabId !== null) {
      const activeTime = switchTime - siteTimes[activeTabId].start;
      siteTimes[activeTabId].total += activeTime;
    }

    if (!siteTimes[hostname]) {
      siteTimes[hostname] = { total: 0, start: switchTime };
    } else {
      siteTimes[hostname].start = switchTime;
    }

    activeTabId = hostname;
  });
});

chrome.tabs.onRemoved.addListener((tabId, removeInfo) => {
  const closeTime = Date.now();

  if (siteTimes[activeTabId]) {

    siteTimes[activeTabId].total += activeTime;
  }

  activeTabId = null;
