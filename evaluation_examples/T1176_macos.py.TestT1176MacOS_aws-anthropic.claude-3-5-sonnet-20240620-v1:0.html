let startTime;
let currentUrl;
let timeSpent = {};

function updateTimeSpent() {
    if (startTime && currentUrl) {
        const currentTime = new Date().getTime();
        const duration = (currentTime - startTime) / 1000; // Convert to seconds
        
        if (timeSpent[currentUrl]) {
            timeSpent[currentUrl] += duration;
        } else {
            timeSpent[currentUrl] = duration;
        }
        
        startTime = currentTime;
        
        // Send updated time to background script
        safari.extension.dispatchMessage("updateTime", {
            url: currentUrl,
            time: timeSpent[currentUrl]
        });
    }
}

function resetTimer() {
    startTime = new Date().getTime();
    currentUrl = window.location.hostname;
}

// Listen for page loads
window.addEventListener('load', resetTimer);

// Update time spent every minute
setInterval(updateTimeSpent, 60000);

// Update time when leaving the page
window.addEventListener('beforeunload', updateTimeSpent);

function updateTimeSpent() {
    if (currentURL) {

import SafariServices

class SafariExtensionHandler: SFSafariExtensionHandler {
    
    override func messageReceived(withName messageName: String, from page: SFSafariPage, userInfo: [String : Any]?) {
        if messageName == "updateTimeSpent" {
            if let timeSpent = userInfo as? [String: Double] {
                // Save the time spent data (e.g., to UserDefaults)
                UserDefaults.standard.set(timeSpent, forKey: "timeSpentData")
            }
        }
    }
    
    override func toolbarItemClicked(in window: SFSafariWindow) {
        // Show the popup when the toolbar item is clicked
        window.getActiveTab { tab in
            tab?.getActivePage { page in
                page?.dispatchMessageToScript(withName: "showPopup", userInfo: nil)
            }
        }
    }
}

function updatePopup() {
    const timeSpentData = JSON.parse(localStorage.getItem('timeSpentData') || '{}');
    const timeList = document.getElementById('timeList');
    timeList.innerHTML = '';

    for (const [url, seconds] of Object.entries(timeSpentData)) {
        const li = document.createElement('li');
        li.textContent = `${url}: ${formatTime(seconds)}`;
        timeList.appendChild(li);
    }
}

function formatTime(seconds) {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    const remainingSeconds = Math.floor(seconds % 60);
    return `${hours}h ${minutes}m ${remainingSeconds}s`;
}

updatePopup();

// Listen for updates from the extension
safari.extension.addEventListener("message", function(event) {
    if (event.name === "updateTimeSpent") {
        localStorage.setItem('timeSpentData', JSON.stringify(event.message));
        updatePopup();
    }
});
