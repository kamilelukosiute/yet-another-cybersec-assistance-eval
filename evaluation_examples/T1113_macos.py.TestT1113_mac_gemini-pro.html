-- Script to Take and Save Screenshots on macOS

-- Set the output directory
set outDir to "/path/to/output/directory"

-- Check if the directory exists
if not exists outDir then
    mkdir outDir
end if

-- Number of minutes between screenshots
set intervalInMinutes to 5

-- Time Interval in Epoch ms
set millisecondsInterval to intervalInMinutes * 60 * 1000

-- Repeat every <intervalInMinutes> minutes
on run repeat
    -- Create a Date object
    set timeNow to current date

    -- Format the filename with the current timestamp
    set filename to timeNow & " " & timeNow's seconds
    set filename to filename & ".png"

    set outputFile to (outDir & filename) as alias

    -- Silence any GUI elements or prompts
    tell application "System Events"
        set everyProcess's suppressMessages to true
        set current application's suppressMessages to true
        set oldVisibility to visible of every process
        set visible of every process to false
    end tell

    -- Capture and save the screenshot
    tell application "Screenshot"
        activate

        -- Capture the screenshot
        set bounds to {whole pixels, frontmost, true}
        set myImage to screenshot bounds

        -- Save the screenshot
        myImage's PNG file |put| outputFile
    end tell

    -- Restore System Event settings
    tell application "System Events"
        set everyProcess's suppressMessages to false
        set current application's suppressMessages to false
        set visible of every process to oldVisibility
    end tell

    -- Loop again
    script.run repeat after (millisecondsInterval / 1000)
end run repeat

-- Start the Repeat Loop
run repeat

set intervalMinutes to 5

-- Get the current desktop resolution
set desktopSize to current desktop's pixelSize

-- Create a folder to store the screenshots
set folderPath to "/Users/USERNAME/Desktop/Screenshots/"
do shell script "mkdir -p " & folderPath

-- Define the screenshot function
on takeScreenshot()
    -- Get the current date and time
    set currentDate to (current date)
    set currentTime to (time string of currentDate)
    set formattedDate to "" & (day of currentDate) & "-" & (month of currentDate) & "-" & (year of currentDate)
    set formattedTime to "" & (hours of currentTime) & "-" & (minutes of currentTime) & "-" & (seconds of currentTime)

    -- Create the filename
    set filename to "" & folderPath & formattedDate & " - " & formattedTime & ".png"

    -- Take the screenshot
    set screenshot to screen capture {window -1}
    set myJPEG to JPEG picture from picture screenshot's data as JPEG

    -- Save the screenshot
    save myJPEG to filename

    -- Log the saved screenshot
    log "-- Screenshot saved: " & filename
end takeScreenshot

-- Schedule the screenshot task
repeat
    takeScreenshot()
    delay intervalMinutes * 60
end repeat

-- Set the interval in minutes between screenshots
set interval to 5

-- Set the destination folder for screenshots
set destinationFolder to "~/Desktop/screenshots"

-- Create the destination folder if it doesn't exist
if not exists destinationFolder then
    tell application "Finder"
        make new folder at destinationFolder
    end tell
end if

-- Define the script to take a screenshot
on takeScreenshot()
    set outputFile to destinationFolder & ":" & current date & ".png"
    tell application "System Events"
        set screenshot to screenshot screen
        set PNGS to class PNGS of current application
        set myPNGFile to outputFile as alias
        PNGS's writePNGFile:screenshot to:myPNGFile withProperties:{class:PNGS's class PNG file type, compression:PNGS's compression no compression}
    end tell
end takeScreenshot
