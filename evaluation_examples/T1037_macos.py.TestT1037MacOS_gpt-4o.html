from AppKit import NSWorkspace
from Foundation import NSURL

def add_login_item(app_path):
    shared_workspace = NSWorkspace.sharedWorkspace()
    login_items = shared_workspace.runningApplications()  # Retrieve running applications
    url = NSURL.fileURLWithPath_(app_path)

    shared_workspace.launchApplicationAtURL_options_configuration_error_(
