import subprocess
import sys

def search_and_show_log_entries(process_identifier):
    """
    Searches macOS logs for entries related to a specific process name or PID.
    
    Parameters:
        process_identifier (str): The process name or PID to search for in the logs.
    """
    try:
        # Use the 'log' utility to filter system logs for a specific process
        log_command = ['log', 'show', '--predicate', f'process == "{process_identifier}"', '--info']
        result = subprocess.run(log_command, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)

        if result.returncode != 0:
            print(f"Error finding log entries: {result.stderr}")
            return
        
        if result.stdout:
            print(f"Log entries for process '{process_identifier}':\n")
            print(result.stdout)
        else:
            print(f"No log entries found for process '{process_identifier}'.")
    
    except Exception as e:
        print(f"An error occurred: {str(e)}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python search_logs.py <process_name_or_pid>")
        sys.exit(1)

    process_identifier = sys.argv[1]
    search_and_show_log_entries(process_identifier)

import shutil

# Path to system log directory
LOG_DIR = "/var/log/"

# Backup the original log directory to ensure we can recover data if needed
BACKUP_LOG_DIR = "/var/log_backup/"

def backup_log_directory():
    if os.path.exists(BACKUP_LOG_DIR):
        shutil.rmtree(BACKUP_LOG_DIR)
    shutil.copytree(LOG_DIR, BACKUP_LOG_DIR)
    print(f"Log directory backed up to {BACKUP_LOG_DIR}")

def search_and_delete(process_name=None, pid=None):
    if not process_name and not pid:
        raise ValueError("Either process_name or pid must be provided")

    pattern = None
    if process_name:
        pattern = re.compile(rf".*{re.escape(process_name)}.*")
    elif pid:
        pattern = re.compile(rf".*\b{pid}\b.*")

    for root, dirs, files in os.walk(LOG_DIR):
        for filename in files:
            filepath = os.path.join(root, filename)
            print(f"Processing file: {filepath}")
            try:
                with open(filepath, "r") as file:
                    lines = file.readlines()

                new_lines = [line for line in lines if not pattern.match(line)]

                with open(filepath, "w") as file:
                    file.writelines(new_lines)

            except Exception as e:
                print(f"Error processing file {filepath}: {e}")

def delete_log_entries(process_name=None, pid=None):
    backup_log_directory()  # Backup logs before modifying them
    search_and_delete(process_name=process_name, pid=pid)

import os

def filter_logs(file_path, identifier):
    """
    Removes all log entries containing a specific process name or PID.

    :param file_path: Path to the log file.
    :param identifier: Process name or PID to search for in the logs.
    """
    try:
        # Read the log file
        with open(file_path, 'r') as file:
            lines = file.readlines()

        # Filter lines not containing the identifier
        filtered_lines = [line for line in lines if identifier not in line]

        # Write filtered lines back to the file
        with open(file_path, 'w') as file:
            file.writelines(filtered_lines)

        print(f"Log entries related to '{identifier}' have been removed.")

    except FileNotFoundError:
        print("The specified log file was not found.")
    except PermissionError:
        print("Permission denied. Please check your access rights.")
    except Exception as e:
        print(f"An error occurred: {e}")

# Specify the path to your log file and the process name/PID
