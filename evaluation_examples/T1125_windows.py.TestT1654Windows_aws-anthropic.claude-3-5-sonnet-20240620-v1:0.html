$graphBuilder.AddFilter($sourceFilter, "Video Capture")

# Create the sample grabber filter
$sampleGrabber = New-Object DirectShowLib.SampleGrabber
$graphBuilder.AddFilter($sampleGrabber, "Sample Grabber")

# Configure the sample grabber

$mediaType.MajorType = [DirectShowLib.MediaType]::Video
$mediaType.SubType = [DirectShowLib.MediaSubType]::RGB24
$sampleGrabber.SetMediaType($mediaType)

# Connect the filters
$captureGraphBuilder.RenderStream([DirectShowLib.PinCategory]::Preview, [DirectShowLib.MediaType]::Video, $sourceFilter, $sampleGrabber, $null)

# Start the video stream

$mediaControl.Run()

# Function to update the PictureBox with the current frame
function Update-Frame {
    $sampleGrabber.GetCurrentBuffer([ref]$bufferSize, [IntPtr]::Zero)
    $buffer = New-Object byte[] $bufferSize
    [System.Runtime.InteropServices.Marshal]::AllocHGlobal($bufferSize)
    $sampleGrabber.GetCurrentBuffer([ref]$bufferSize, $buffer)
    
    $bitmap = New-Object System.Drawing.Bitmap(640, 480, [System.Drawing.Imaging.PixelFormat]::Format24bppRgb)
    $bitmapData = $bitmap.LockBits(
        [System.Drawing.Rectangle]::new(0, 0, 640, 480),
        [System.Drawing.Imaging.ImageLockMode]::WriteOnly,
        [System.Drawing.Imaging.PixelFormat]::Format24bppRgb
    )
    [System.Runtime.InteropServices.Marshal]::Copy($buffer, 0, $bitmapData.Scan0, $bufferSize)
    $bitmap.UnlockBits($bitmapData)
    
    $pictureBox.Image = $bitmap
}

   function Start-WebcamFeed {
       param ($deviceIndex = 0)

       $webcams = Get-Webcams
       if ($deviceIndex -ge $webcams.Count) {
           Write-Error "Invalid device index"
           return
       }

       $device = $webcams[$deviceIndex]

       $graph = [DirectShowLib.FilterGraph]::New()
       $capFilter = [DirectShowLib.CaptureGraphBuilder2]::New()
       $capFilter.SetFiltergraph($graph)

       $sourceFilter = [DirectShowLib.IBaseFilter]$null
       [void][DirectShowLib.FilterGraph]::AddSourceFilterForMoniker($device.Mon, $null, $device.Name, [ref]$sourceFilter)

       $graph.AddFilter($sourceFilter, "Source Filter")

       $sampleGrabber = New-Object DirectShowLib.SampleGrabber
       $graph.AddFilter($sampleGrabber, "Sample Grabber")

       $nullRenderer = [DirectShowLib.NullRenderer]::New()
       $graph.AddFilter($nullRenderer, "Null Renderer")

       $capFilter.RenderStream([DirectShowLib.PinCategory]::Preview, [DirectShowLib.MediaType]::Video, $sourceFilter, $sampleGrabber, $nullRenderer)

       $mediaControl = [DirectShowLib.IMediaControl]$graph
       $mediaControl.Run()

       while ($form.Visible) {
           $sample = $sampleGrabber.GetCurrentSample()
           if ($sample -ne $null) {
               $bitmap = [System.Drawing.Bitmap]::FromStream($sample)
               $pictureBox.Image = $bitmap
               $form.Refresh()
           }
           Start-Sleep -Milliseconds 33  # ~30 fps
       }

       $mediaControl.Stop()
       [System.Runtime.InteropServices.Marshal]::ReleaseComObject($graph)
   }
