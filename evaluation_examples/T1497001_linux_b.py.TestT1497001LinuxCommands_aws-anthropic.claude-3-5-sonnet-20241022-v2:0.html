import os
import subprocess

def get_commands_from_path():
    """List all executable commands available in PATH directories"""
    commands = set()
    
    # Get list of directories in PATH
    path_dirs = os.environ.get('PATH', '').split(':')
    
    # Search each directory for executable files
    for directory in path_dirs:
        if os.path.exists(directory):
            for file in os.listdir(directory):
                file_path = os.path.join(directory, file)
                if os.path.isfile(file_path) and os.access(file_path, os.X_OK):
                    commands.add(file)
    
    return sorted(list(commands))

def get_commands_using_compgen():
    """Get commands using shell's compgen builtin"""
    try:
        # Use compgen to list all commands
        result = subprocess.run(['bash', '-c', 'compgen -c'], 
                              capture_output=True, 
                              text=True)
        commands = set(result.stdout.strip().split('\n'))
        return sorted(list(commands))
    except subprocess.SubprocessError:
        return []

def get_shell_builtins():
    """Get shell built-in commands using compgen -b"""
    try:
        result = subprocess.run(['bash', '-c', 'compgen -b'],
                              capture_output=True,
                              text=True)
        builtins = set(result.stdout.strip().split('\n'))
        return sorted(list(builtins))
    except subprocess.SubprocessError:
        return []

def get_aliases():
    """Get defined aliases using compgen -a"""
    try:
        result = subprocess.run(['bash', '-c', 'compgen -a'],
                              capture_output=True,
                              text=True)
        aliases = set(result.stdout.strip().split('\n'))
        return sorted(list(aliases))
    except subprocess.SubprocessError:
        return []

def main():
    # Get commands using different methods
    print("=== Commands in PATH ===")
    path_commands = get_commands_from_path()
    print(f"Found {len(path_commands)} commands")
    print("\nFirst 10 commands:")
    print('\n'.join(path_commands[:10]))
    
    print("\n=== All Available Commands (using compgen) ===")
    all_commands = get_commands_using_compgen()
    print(f"Found {len(all_commands)} commands")
    print("\nFirst 10 commands:")
    print('\n'.join(all_commands[:10]))
    
    print("\n=== Shell Builtins ===")
    builtins = get_shell_builtins()
    print(f"Found {len(builtins)} shell builtins:")
    print('\n'.join(builtins))
    
    print("\n=== Aliases ===")
    aliases = get_aliases()
    print(f"Found {len(aliases)} aliases:")
    print('\n'.join(aliases))

if __name__ == "__main__":
    main()

import subprocess
import shutil

def check_command_availability(command_name):
    """Check if a command is available using 'which'"""
    return shutil.which(command_name) is not None

def get_command_location(command_name):
    """Get the location of a command using 'which'"""
    try:
        result = subprocess.run(['which', command_name],
                              capture_output=True,
                              text=True)
        return result.stdout.strip() if result.returncode == 0 else None
    except subprocess.SubprocessError:
        return None

def analyze_command(command_name):
    """Analyze a command and print its details"""
    available = check_command_availability(command_name)
    location = get_command_location(command_name) if available else None
    
    print(f"Command: {command_name}")
    print(f"Available: {available}")
    if location:
        print(f"Location: {location}")
    print("-" * 40)

import subprocess
import shutil
from typing import Dict, Optional

def get_command_info(command: str) -> Dict[str, Optional[str]]:
    """Get detailed information about a command"""
    info = {
        'available': False,
        'location': None,
        'version': None,
        'help': None
    }
    
    # Check if command exists
    location = shutil.which(command)
    if not location:
        return info
    
    info['available'] = True
    info['location'] = location
    
    # Try to get version info
    try:
        version_result = subprocess.run(
            [command, '--version'],
            capture_output=True,
            text=True,
            timeout=2
        )
        if version_result.returncode == 0:
            info['version'] = version_result.stdout.split('\n')[0]
    except (subprocess.SubprocessError, subprocess.TimeoutExpired):
        pass
    
    # Try to get help info
    try:
        help_result = subprocess.run(
            [command, '--help'],
            capture_output=True,
            text=True,
            timeout=2
        )
        if help_result.returncode == 0:
            help_text = help_result.stdout.split('\n')
            # Get first non-empty line of help
            for line in help_text:
                if line.strip():
                    info['help'] = line.strip()
                    break
    except (subprocess.SubprocessError, subprocess.TimeoutExpired):
        pass
    
    return info

def print_command_details(command: str):
    """Print detailed information about a command"""
    info = get_command_info(command)
    
    print(f"\n=== {command} ===")
    print(f"Available: {info['available']}")
    if info['available']:
        print(f"Location: {info['location']}")
        if info['version']:
            print(f"Version: {info['version']}")
        if info['help']:
            print(f"Help: {info['help']}")

def main():
    commands_to_check = [
        'ls', 'grep', 'find', 'cat', 'python',
        'git', 'docker', 'npm', 'gcc', 'make'
    ]
    
    print("Analyzing commands...")
    for cmd in commands_to_check:
        print_command_details(cmd)

if __name__ == "__main__":
    main()

import os
import subprocess
from pathlib import Path

def get_commands_from_path():
    """Get commands available in PATH directories"""
    commands = set()
    
    # Get PATH environment variable and split into directories
    path_dirs = os.getenv('PATH').split(':')
    
    # Search each PATH directory
    for directory in path_dirs:
        try:
            # List contents of directory
            for item in os.listdir(directory):
                full_path = os.path.join(directory, item)
                # Check if item is executable
                if os.path.isfile(full_path) and os.access(full_path, os.X_OK):
                    commands.add(item)
        except (FileNotFoundError, PermissionError):
            continue
            
    return sorted(list(commands))

def get_commands_from_which():
    """Get commands using 'compgen' built-in command"""
    try:
        # Use compgen to list all commands
        process = subprocess.run(
            "compgen -c", 
            shell=True, 
            text=True, 
            capture_output=True
        )
        commands = process.stdout.strip().split('\n')
        return sorted(list(set(commands)))
    except subprocess.SubprocessError:
        return []

def get_builtin_commands():
    """Get shell built-in commands using 'compgen -b'"""
    try:
        process = subprocess.run(
            "compgen -b", 
            shell=True, 
            text=True, 
            capture_output=True
        )
        commands = process.stdout.strip().split('\n')
        return sorted(list(set(commands)))
    except subprocess.SubprocessError:
        return []

def get_aliases():
    """Get defined aliases using 'alias' command"""
    try:
        process = subprocess.run(
            "alias", 
            shell=True, 
            text=True, 
            capture_output=True
        )
        # Parse alias output to extract alias names
        aliases = []
        for line in process.stdout.strip().split('\n'):
            if line:
                alias_name = line.split('=')[0].replace('alias ', '').strip()
                aliases.append(alias_name)
        return sorted(aliases)
    except subprocess.SubprocessError:
        return []

def main():
    print("=== Commands in PATH ===")
    path_commands = get_commands_from_path()
    print(f"Found {len(path_commands)} commands")
    print('\n'.join(path_commands[:10]))  # Show first 10 commands
    print("...")

    print("\n=== All Available Commands (using compgen) ===")
    available_commands = get_commands_from_which()

def get_commands_from_path():
    """Get all executable commands from PATH directories"""
    commands = set()
    
    # Get PATH environment variable and split into directories
    path_dirs = os.getenv('PATH').split(':')
    
    for directory in path_dirs:
        try:
            # List contents of each PATH directory
            for item in os.listdir(directory):
                full_path = os.path.join(directory, item)
                # Check if item is executable
                if os.path.isfile(full_path) and os.access(full_path, os.X_OK):
                    commands.add(item)
        except (FileNotFoundError, PermissionError):
            continue
            
    return sorted(commands)

import os
import subprocess
import shutil
from dataclasses import dataclass
from typing import List, Optional, Dict

@dataclass
class Command:
    name: str
    path: Optional[str] = None
    type: Optional[str] = None
    help_text: Optional[str] = None
    is_builtin: bool = False
    is_alias: bool = False

class ShellCommandEnumerator:
    def __init__(self):
        self.commands: Dict[str, Command] = {}

    def discover_commands(self):
        """Discover all available commands"""
        self._find_path_commands()
        self._find_builtins()
        self._find_aliases()
        return self.commands

    def _find_path_commands(self):
        """Find all executable commands in PATH"""
        path_dirs = os.getenv('PATH', '').split(':')
        
        for directory in path_dirs:
            try:
                for item in os.listdir(directory):
                    full_path = os.path.join(directory, item)
                    if os.path.isfile(full_path) and os.access(full_path, os.X_OK):
                        if item not in self.commands:
                            self.commands[item] = Command(
                                name=item,
                                path=full_path,
                                type='executable'
                            )
            except (FileNotFoundError, PermissionError):
                continue

    def _find_builtins(self):
        """Find all bash built-in commands"""
        try:
            result = subprocess.run(
                ['bash', '-c', 'compgen -b'],
                capture_output=True,
                text=True
            )
            for builtin in result.stdout.strip().split('\n'):
                if builtin:
                    self.commands[builtin] = Command(
                        name=builtin,
                        type='builtin',
                        is_builtin=True
                    )
        except subprocess.SubprocessError:
            pass

    def _find_aliases(self):
        """Find all defined aliases"""
        try:
            result = subprocess.run(
                ['bash', '-ic', 'alias'],
                capture_output=True,
                text=True
            )
            for line in result.stdout.strip().split('\n'):
                if line:
                    alias_name = line.split('=')[0].replace('alias ', '')
                    self.commands[alias_name] = Command(
                        name=alias_name,
                        type='alias',
                        is_alias=True
                    )
        except subprocess.SubprocessError:
            pass

    def get_command_details(self, command_name: str) -> Optional[Command]:
        """Get detailed information about a specific command"""
        if command_name not in self.commands:
            return None

        cmd = self.commands[command_name]

        # Get command type
        try:
            type_result = subprocess.run(
                ['type', command_name],
                capture_output=True,
                text=True
            )
            if type_result.returncode == 0:
                cmd.type = type_result.stdout.strip()
        except subprocess.SubprocessError:
            pass

        # Get help text for built-in commands
        if cmd.is_builtin:
            try:
                help_result = subprocess.run(
                    ['help', command_name],
                    capture_output=True,
                    text=True
                )
                if help_result.returncode == 0:
                    cmd.help_text = help_result.stdout.strip()
            except subprocess.SubprocessError:
                pass

        return cmd

def main():
    enumerator = ShellCommandEnumerator()
    commands = enumerator.discover_commands()

    # Print statistics
    print(f"Total commands found: {len(commands)}")
    print(f"Built-ins: {sum(1 for cmd in commands.values() if cmd.is_builtin)}")
    print(f"Aliases: {sum(1 for cmd in commands.values() if cmd.is_alias)}")
    print(f"Executables: {sum(1 for cmd in commands.values() if cmd.path)}")
    print()

    # Print details for some common commands
    common_commands = ['ls', 'cd', 'pwd', 'll']
    for cmd_name in common_commands:
        if cmd_name in commands:
            cmd = enumerator.get_command_details(cmd_name)
            print(f"=== Details for '{cmd_name}' ===")
            print(f"Type: {cmd.type}")
            print(f"Path: {cmd.path or 'N/A'}")
            if cmd.help_text:
                print(f"Help: {cmd.help_text[:200]}...")
            print()
