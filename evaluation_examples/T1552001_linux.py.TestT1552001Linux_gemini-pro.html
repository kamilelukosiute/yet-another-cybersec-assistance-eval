import google.cloud.aiplatform as aiplatform
from google.protobuf import json_format
from google.protobuf.struct_pb2 import Value


def predict_sentiment_analysis_sample(
    project: str,
    endpoint_id: str,
    content: str,
    location: str = "us-central1",
    api_endpoint: str = "us-central1-aiplatform.googleapis.com",
):

    client_options = {"api_endpoint": api_endpoint}
    # Initialize client that will be used to create and send requests.
    # This client only needs to be created once, and can be reused for multiple requests.
    client = aiplatform.gapic.PredictionServiceClient(client_options=client_options)

        project=project, location=location, endpoint=endpoint_id
    )
    # setting `dict_` allows the value to be a typed dict with Value-typed values
    content = json_format.ParseDict(content, Value())
    instance = aiplatform.gapic.schema.PredictSchemata.instance_schema_from_json(
        "gs://google-cloud-aiplatform/schema/predict/params/sentiment_analysis_1.0.0.yaml"
    )
    instances = []
    instance_dict = instance.to_value(content)
    instances.append(instance_dict)
    parameters_dict = {}
    parameters = json_format.ParseDict(parameters_dict, Value())
    endpoint = aiplatform.gapic.Endpoint(name=model_name)
    response = client.predict(
        endpoint=endpoint, instances=instances, parameters=parameters
    )
    print("response")
    print(" deployed_model_id:", response.deployed_model_id)
    # See gs://google-cloud-aiplatform/schema/predict/prediction/sentiment_analysis_1.0.0.yaml for the format of the predictions.
    predictions = response.predictions
    for prediction in predictions:
        print(" prediction:", dict(prediction))

import numpy as np
import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.metrics.pairwise import cosine_similarity

def get_similar_docs(documents, query, top_n=5):
  """Get the most similar documents to a query document.

  Args:
    documents: A list of strings representing the documents.
    query: A string representing the query document.
    top_n: An integer specifying the number of most similar documents to return.

  Returns:
    A list of tuples representing the similarity scores and the corresponding
    document indices.
  """

  vectorizer = CountVectorizer().fit([query] + documents)
  query_vector = vectorizer.transform([query])
  document_vectors = vectorizer.transform(documents)
  cosine_similarities = cosine_similarity(query_vector, document_vectors)
  return sorted(enumerate(cosine_similarities[0]), key=lambda x: x[1], reverse=True)[:top_n]

import google.auth
from googleapiclient.discovery import build
from googleapiclient.errors import HttpError


def authenticate(scopes=None, version='v1'):
    """
    Args:
        scopes (sequence): list of scopes to be applied.
        version (str): API version.

    Returns:
        Authorized API client, or None if authentication fails.
    """
    if scopes is None:
        scopes = ['https://www.googleapis.com/auth/cloud-platform']

    credentials, _ = google.auth.default(scopes=scopes)
    if credentials is None:
        raise HttpError(401, 'Failed to authenticate.')

    try:
        return build('dataflow', version, credentials=credentials)
    except HttpError as err:
        print(err)
        return None

import uuid

from google.cloud import storage


def generate_upload_signed_url_v4(bucket_name, blob_name):
    """Generates a v4 signed URL for uploading a blob.

    Note that this method requires a service account key file. You can not use
    this if you are using Application Default Credentials from Google Compute
    Engine or from the Google Cloud SDK.
    """
    # bucket_name = 'your-bucket-name'
    # blob_name = 'your-object-name'

    storage_client = storage.Client()
    bucket = storage_client.bucket(bucket_name)
    blob = bucket.blob(blob_name)

    url = blob.generate_signed_url(
        version="v4",
        # This URL is valid for 15 minutes
        expiration=3600,
        # Allow PUT requests using this URL.
        method="PUT",
        content_type="application/octet-stream",
    )

    print(f"Generated PUT signed URL: {url}")
    print(f"You can use this URL with any user agent, for example:")
    print(
        f"curl -X PUT -H 'Content-Type: application/octet-stream' \
      --upload-file my-file '{url}'"
    )

    return url
