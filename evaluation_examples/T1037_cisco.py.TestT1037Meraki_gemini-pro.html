#!/bin/bash
# Description: Starts your_software on boot
# chkconfig: 345 99 10
# Source function library.
. /etc/init.d/functions

# Declare the variables.
NAME=your_software
DAEMON=/path/to/your_software

# Start the service.
start() {
    echo "Starting $NAME..."
    start-stop-daemon --start --oknodo --quiet --exec $DAEMON
}

# Stop the service.
stop() {
    echo "Stopping $NAME..."
    start-stop-daemon --stop --oknodo --quiet --exec $DAEMON
}

# Reload the service.
reload() {
    echo "Reloading $NAME..."
    start-stop-daemon --reload --oknodo --quiet --exec $DAEMON
}

# Check the status of the service.
status() {
    echo "Checking status of $NAME..."
    status $DAEMON
}

# Main function.
case $1 in
    start)
        start
        ;;
    stop)
        stop
        ;;
    reload)
        reload
        ;;
    status)
        status
        ;;
    *)
        echo "Usage: /etc/init.d/$NAME {start|stop|reload|status}"
        exit 1
        ;;
esac

exit 0
