#!/bin/bash

# Redirect stderr to stdout
exec 2>&1

# Initialize timestamp to prevent initial processing
prev_timestamp="0"

# Main Loop to Continuously Monitor Clipboard
while true; do

    # Only process if Clipboard contents has changed
    current_timestamp=`xsel -b -o`
    if [[ "$current_timestamp" != "$prev_timestamp" ]]; then

        # Store Timestamp to Check against next Cycle
        prev_timestamp=$current_timestamp

        # Retrieve Current Contents of Clipboard
        content=`xsel -b -o`

        # Ensure history File Is Created or Is Empty
        echo -n "" > history.txt

        # Append Current Clipboard Contents To history File
        echo "$content" >> history.txt
    fi

    sleep 1  # Sleep For 1 Second before Next Check
done

#!/bin/bash

# Create the history file if it doesn't exist
touch history.txt

# Continuously monitor the clipboard for changes
while true; do
  # Get the current clipboard contents
  clipboard=$(xclip -o -selection clipboard)
  
  # Check if the clipboard contents have changed
  if [ "$clipboard" != "$last_clipboard" ]; then
    # Update the history file
    echo "$clipboard" >> history.txt

    # Update the last_clipboard variable
    last_clipboard="$clipboard"
  fi

  # Sleep for 1 second
  sleep 1
done

#!/bin/bash

# Create history file
touch history.txt
chmod 755 history.txt

# Continuously monitor clipboard for changes
while true; do
    # Get current clipboard contents
    clipboard_contents=$(xclip -o)

    # Check if clipboard contents have changed
    if grep -q "$clipboard_contents" history.txt; then
        # Clipboard contents haven't changed, so do nothing
        continue
    else
        # Clipboard contents have changed, so save them to history file
        echo "$clipboard_contents" >> history.txt
    fi

    # Sleep for 1 second before checking again
    sleep 1
done

#!/bin/bash

# Create history file
touch history.txt

# Infinite loop to monitor clipboard
while true; do
  # Get clipboard contents
  clipText=$(xclip -selection clipboard -o)

  # Compare with previous clipboard contents
  if [[ "$clipText" != "$prevClipText" ]]; then
    # Save new text to history file
    echo "$clipText" >> history.txt
    # Update previous clipboard contents
    prevClipText="$clipText"
  fi

  # Adjust sleep time based on clipboard activity
  if [[ -n "$clipText" ]]; then
    sleep 0.1
  else
    sleep 1
  fi
done
