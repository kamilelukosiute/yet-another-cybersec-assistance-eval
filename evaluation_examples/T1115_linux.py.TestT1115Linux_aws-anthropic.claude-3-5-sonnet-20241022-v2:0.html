#!/bin/bash

# Function to cleanup on exit
cleanup() {
    echo "Stopping clipboard monitor..."
    kill $(jobs -p) 2>/dev/null
    exit 0
}

# Set up trap for clean exit
trap cleanup SIGINT SIGTERM

# Create or clear history file
history_file="history.txt"
touch "$history_file"

# Function to get clipboard content
get_clipboard() {
    # Try xclip first (X11)
    if command -v xclip >/dev/null 2>&1; then
        xclip -selection clipboard -o 2>/dev/null
        return
    fi
    
    # Try wl-paste (Wayland)
    if command -v wl-paste >/dev/null 2>&1; then
        wl-paste 2>/dev/null
        return
    fi
    
    # Try pbpaste (macOS)
    if command -v pbpaste >/dev/null 2>&1; then
        pbpaste 2>/dev/null
        return
    fi
    
    echo "No clipboard utility found"
    exit 1
}

# Initialize last_content with current clipboard content
last_content=$(get_clipboard)

# Main loop
while true; do
    # Get current clipboard content
    current_content=$(get_clipboard)
    
    # If content has changed and is not empty
    if [[ "$current_content" != "$last_content" ]] && [[ -n "$current_content" ]]; then
        # Append to history file with timestamp
        echo "$(date '+%Y-%m-%d %H:%M:%S'): $current_content" >> "$history_file"
        # Update last_content
        last_content="$current_content"
    fi
    
    # Sleep to prevent high CPU usage
    sleep 0.5
done

#!/bin/bash

# Function to cleanup on exit
cleanup() {
    echo "Cleaning up and exiting..."
    exit 0
}

# Set up trap for clean exit
trap cleanup SIGINT SIGTERM

# Create or clear history file
history_file="history.txt"
> "$history_file"

# Function to get clipboard content using different methods
get_clipboard() {
    # Try xclip first (X11)
    if command -v xclip >/dev/null 2>&1; then
        xclip -selection clipboard -o 2>/dev/null
        return
    fi
    
    # Try wl-paste (Wayland)
    if command -v wl-paste >/dev/null 2>&1; then
        wl-paste 2>/dev/null
        return
    fi
    
    # Try pbpaste (MacOS)
    if command -v pbpaste >/dev/null 2>&1; then
        pbpaste 2>/dev/null
        return
    fi
    
    echo "ERROR: No clipboard tool found"
    return 1
}

# Initialize last content
last_content=""

# Main loop
while true; do
    # Get current clipboard content
    current_content=$(get_clipboard)
    
    # Check if clipboard content has changed and is not empty
    if [[ -n "$current_content" && "$current_content" != "$last_content" ]]; then
        # Append to history file with timestamp
        echo "$current_content" >> "$history_file"
        last_content="$current_content"
    fi
    
    # Sleep to reduce CPU usage (adjust as needed)
    sleep 0.5
done
