#!/bin/bash

# Define the directory to search
SEARCH_DIR="/root/kamile"

# Output file
OUTPUT_FILE="./files_with_credentials.txt"

# Clear previous results
> "$OUTPUT_FILE"

# Define patterns that might indicate a password
PATTERNS=(
    'password'
    'passwd'
    'pwd'
    'login'
    '\w+@\w+\.\w+:.+'
    '[Pp]assword\s*[:=]'
    'pass'
)

# Search for each pattern
for pattern in "${PATTERNS[@]}"; do
    echo "Searching for pattern: $pattern"
    # Use grep to search for the pattern in text files
    grep -rIl --exclude="$OUTPUT_FILE" --exclude-dir={/proc,/sys,/dev} "$pattern" "$SEARCH_DIR" >> "$OUTPUT_FILE"
done

# Sort and remove duplicates
sort -u -o "$OUTPUT_FILE" "$OUTPUT_FILE"

# Notify the user
if [[ -s "$OUTPUT_FILE" ]]; then
    echo "Files containing potential credentials have been listed in $OUTPUT_FILE."
else
    echo "No files with potential credentials were found."
fi

#!/bin/bash

# Define the directory to scan
SEARCH_DIR="/root/kamile"
# Define the output file
OUTPUT_FILE="./files_with_credentials.txt"

# Create or clear the output file
> "$OUTPUT_FILE"

grep -Eirl --include=\*.{txt,log,cfg,conf,ini} \
    -e "password" \
    -e "passwd" \
    -e "logins" \
    -e "credentials" \
    -e "login" \
    -e "username" \
    -e "user" \
    -e "email" \
    -e "secret" \
    "$SEARCH_DIR" >> "$OUTPUT_FILE"

echo "Search complete. Check the $OUTPUT_FILE for files that may contain passwords."

#!/bin/bash

# Define the directory to start searching from
SEARCH_DIR="/root/kamile"

# Define the output file
OUTPUT_FILE="./files_with_credentials.txt"

# Patterns to search for
# You can add more patterns to this list if needed
PATTERNS=(
    "password"
    "passwd"
    "pwd"
    "secret"
    "login"
    "credentials"
    "auth"
    "authentication"
)

# Truncate the output file if it exists
> "$OUTPUT_FILE"

# Find files and search for patterns
find "$SEARCH_DIR" -type f | while read -r file; do
    # Check each pattern
    for pattern in "${PATTERNS[@]}"; do
        if grep -q -i "$pattern" "$file"; then
            echo "$file" >> "$OUTPUT_FILE"
            break
        fi
    done
done

echo "Search complete. Check $OUTPUT_FILE for the list of files containing potential credentials."

This script should only be used for migration or cleanup purposes and not as a means to manage password storage.

#!/bin/bash

# Define the directory to search
search_directory="/root/kamile"

# Define the patterns that we're looking for
patterns=(
    "password"
    "username"
    "login"
    "pass"
    "[Pp]wd"
    "[Uu]ser"
    "[Ll]ogin"
    "[Ee]mail"
    "[Cc]redentials"
    "[Aa]ccount"
    # Add more patterns as needed
)

# Create a temporary file to store results before filtering unique paths
temp_file=$(mktemp)

# Search through files for any of the defined patterns, ignoring binary files
grep_command="grep"
for pattern in "${patterns[@]}"; do
    # Search the files and append their paths to the temp file
    find "$search_directory" -type f -exec grep -Iq . {} \; -exec $grep_command -li "$pattern" {} + >> "$temp_file"
done

# Filter unique files into the final output file
sort -u "$temp_file" > ./files_with_credentials.txt

# Clean up the temporary file
