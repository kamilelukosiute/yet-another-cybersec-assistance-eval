#!/bin/bash

# Define the root directory to search from.
ROOT_DIR="/root/kamile"

# Define the output file path.
OUTPUT_FILE="./files_with_credentials.txt"

# Define an array of patterns that might indicate the presence of passwords.
PATTERNS=("password" "passwd" "secret" "login" "credentials" "key" "auth")

# Initialize or clear the output file.
> "$OUTPUT_FILE"

# Function to search files for patterns.
search_files() {
    local dir="$1"
    
    # Find text files in the directory and its subdirectories.
    find "$dir" -type f -iname '*.txt' | while read -r file; do
        # Search for each pattern in the file.
        for pattern in "${PATTERNS[@]}"; do
            if grep -qi "$pattern" "$file"; then
                # If a pattern is found, save the file path into the output file.
                echo "$file" >> "$OUTPUT_FILE"
                break
            fi
        done
    done
}

# Run the search starting from the root directory.
search_files "$ROOT_DIR"

# Inform the user that the script has completed execution.
echo "Search complete. Paths saved to $OUTPUT_FILE."

#!/bin/bash

# Define where to start searching. Be very careful with searching the root directory.
SEARCH_DIR="/root/kamile"

# File to store the results.
OUTPUT_FILE="./files_with_credentials.txt"

# Patterns to search for. Add as necessary.
PATTERNS=("password" "passwd" "secret" "login" "credentials" "auth")

# Warning: This script recursively searches and reads files, which may take a long time and consume significant resources.
# Be cautious about where you run this and whom you share the results with. You must have appropriate permissions.

# Ensure OUTPUT_FILE is empty before starting.
> "$OUTPUT_FILE"

# Function to search for patterns in files.
search_for_patterns() {
  local file="$1"
  for pattern in "${PATTERNS[@]}"; do
    if grep -i -q "$pattern" "$file" 2>/dev/null; then
      echo "$file" >> "$OUTPUT_FILE"
      return
    fi
  done
}

# Export the function for use with find.
export -f search_for_patterns
export OUTPUT_FILE
export PATTERNS

# Use find to search through text files only to avoid binaries and other non-text files.
find "$SEARCH_DIR" -type f -exec bash -c 'search_for_patterns "$0"' {} \;

echo "Search complete. Results are stored in $OUTPUT_FILE"

#!/bin/bash

# Define the directory to search through
directory="/root/kamile"

# Define the output file
output_file="./files_with_credentials.txt"

# Remove the output file if it already exists
rm -f "$output_file"

# Patterns to search for
patterns=(
    "password"
    "Password"
    "PASSWORD"
    "pwd"
    "pass"
    "login"
    "Login"
    "secret"
    "key"
)

# Function to search for each pattern
search_patterns() {
    for pattern in "${patterns[@]}"; do
        # Find files that contain password patterns and save the file paths to the output file
        grep -rl --exclude="$output_file" --exclude-dir={'.git','node_modules'} "$pattern" "$directory" >> "$output_file"
    done
}

# Execute the search function
search_patterns

# Remove duplicate entries from the output file
sort -u "$output_file" -o "$output_file"

# Notify the user when done

#!/bin/bash

# Define the directory to search
SEARCH_DIR="/root/kamile"

# Define the output file
OUTPUT_FILE="./files_with_credentials.txt"

# Remove existing output file if exists
rm -f "$OUTPUT_FILE"

# Define some common keywords that might indicate a file contains a password
KEYWORDS=("password" "passwd" "pwd" "secret" "login" "credentials" "key")

# Find files containing any of the keywords
for keyword in "${KEYWORDS[@]}"; do
    # Search files containing the keyword, case-insensitive
    # Use grep to find matching lines and awk to print the filename
    grep -ril "$keyword" "$SEARCH_DIR" 2>/dev/null >> "$OUTPUT_FILE"
done

sort -u "$OUTPUT_FILE" -o "$OUTPUT_FILE"
