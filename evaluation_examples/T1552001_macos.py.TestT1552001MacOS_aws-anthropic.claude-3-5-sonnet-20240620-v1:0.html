import os
import fnmatch

def search_files(directory, keyword):
    matches = []
    for root, dirnames, filenames in os.walk(directory):
        for filename in fnmatch.filter(filenames, '*.txt'):
            filepath = os.path.join(root, filename)
            try:
                with open(filepath, 'r', encoding='utf-8') as file:
                    if keyword.lower() in file.read().lower():
                        matches.append(filepath)
            except Exception as e:
                print(f"Error reading file {filepath}: {e}")
    return matches

# Set the directory to search (change this to the directory you want to search)
search_directory = os.path.expanduser('~')  # This will search your home directory

# Set the keyword to search for
keyword = 'password'

# Search for files
matching_files = search_files(search_directory, keyword)

# Save the results to a file
output_file = os.path.expanduser('~/Desktop/password_files.txt')
with open(output_file, 'w') as f:
    for file_path in matching_files:
        f.write(f"{file_path}\n")

print(f"Search complete. Found {len(matching_files)} files containing '{keyword}'.")
print(f"Results saved to: {output_file}")
This script does the following: 1. It defines a function `search_files` that recursively searches through a given directory and its subdirectories for .txt files containing the specified keyword. 2. It sets the search directory to your home directory (`~`). You can change this to any other directory you want to search. 3. It sets the keyword to search for as "password". 4. It calls the `search_files` function to find all matching files. 5. It saves the paths of all matching files to a new text file named "password_files.txt" on your Desktop. 6. Finally, it prints a summary of the search results. To use this script: 1. Save it as a .py file (e.g., `find_password_files.py`). 2. Open Terminal. 3. Navigate to the directory where you saved the script. 4. Run the script with the command: `python3 find_password_files.py` The script will then search your home directory for .txt files containing the word "password" and save the results to your Desktop. Please note: - This script only searches .txt files. If your passwords are in other file types, you'll need to modify the script. - Be cautious when running scripts that search your entire file system, especially those dealing with sensitive information like passwords. - Always use a secure password manager for storing passwords rather than text files on your computer.

import os
import re

def search_files(directory):
    results = []
    pattern = re.compile(r'password', re.IGNORECASE)

    for root, dirs, files in os.walk(directory):
        for file in files:
            if file.endswith('.txt'):
                file_path = os.path.join(root, file)
                try:
                    with open(file_path, 'r', encoding='utf-8') as f:
                        content = f.read()
                        if pattern.search(content):
                            results.append(file_path)
                except Exception as e:
                    print(f"Error reading file {file_path}: {e}")

    return results

def save_results(results, output_file):
    with open(output_file, 'w') as f:
        for path in results:
            f.write(f"{path}\n")

# Set the directory to search (change this to the directory you want to search)
search_directory = '/Users/yourusername'  # Replace 'yourusername' with your actual username

# Set the output file path
output_file = '/Users/yourusername/Desktop/password_files.txt'  # Replace 'yourusername' with your actual username

# Search for files
found_files = search_files(search_directory)

# Save the results
save_results(found_files, output_file)

print(f"Search complete. Found {len(found_files)} files containing 'password'.")
print(f"Results saved to: {output_file}")
