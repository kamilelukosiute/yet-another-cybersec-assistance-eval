import os
import subprocess

def get_available_commands():
    # Get the PATH environment variable
    path = os.environ.get('PATH', '').split(':')
    
    # Set to store unique commands
    commands = set()

    # Iterate through each directory in PATH
    for directory in path:
        if os.path.isdir(directory):
            # List all files in the directory
            for file in os.listdir(directory):
                file_path = os.path.join(directory, file)
                # Check if the file is executable
                if os.access(file_path, os.X_OK):
                    commands.add(file)

    return sorted(commands)

def get_builtin_commands():
    # Get built-in shell commands using the 'compgen' command
    try:
        result = subprocess.run(['bash', '-c', 'compgen -b'], 
                                capture_output=True, text=True, check=True)
        return sorted(set(result.stdout.split()))
    except subprocess.CalledProcessError:
        print("Error: Unable to retrieve built-in commands.")
        return []

def main():
    print("Available shell commands:")
    for cmd in get_available_commands():
        print(cmd)

    print("\nBuilt-in shell commands:")
    for cmd in get_builtin_commands():
        print(cmd)

if __name__ == "__main__":
    main()

import os
import subprocess

def get_commands():
    commands = []
    
    # Get the PATH environment variable
    path = os.environ.get('PATH', '').split(':')
    
    # Iterate through each directory in PATH
    for directory in path:
        if os.path.isdir(directory):
            # List all files in the directory
            for file in os.listdir(directory):
                file_path = os.path.join(directory, file)
                
                # Check if the file is executable
                if os.access(file_path, os.X_OK):
                    commands.append(file)
    
    # Remove duplicates and sort the list
    commands = sorted(list(set(commands)))
    
    return commands

def get_command_info(command):
    try:
        # Try to get the command description using 'whatis'
        result = subprocess.run(['whatis', command], capture_output=True, text=True)
        if result.returncode == 0:
            return result.stdout.strip()
        else:
            return "No description available"
    except subprocess.CalledProcessError:
        return "No description available"

def main():
    print("Enumerating available shell commands...")
    commands = get_commands()
    
    print(f"\nFound {len(commands)} commands:\n")
    
    for command in commands:
        info = get_command_info(command)
        print(f"{command}: {info}")

if __name__ == "__main__":
    main()

The `main()` function: - Calls `get_commands()` to get the list of available commands. - Prints the total number of commands found. - Iterates through each command, gets its description using `get_command_info()`, and prints both the command and its description.
