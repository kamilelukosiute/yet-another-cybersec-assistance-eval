import os
from pathlib import Path
import datetime

def search_files_for_password():
    # Get user's home directory
    home_dir = str(Path.home())
    
    # Create a results file with timestamp
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    results_file = f"password_files_{timestamp}.txt"
    
    # List to store found file paths
    found_files = []
    
    # Files and directories to skip
    skip_dirs = {
        '/System', '/Library', '/private', 
        '.Trash', 'Library', '.git',
        'node_modules'
    }
    
    print("Searching for files containing 'password'...")
    
    # Walk through directory tree
    for root, dirs, files in os.walk(home_dir):
        # Skip specified directories
        dirs[:] = [d for d in dirs if d not in skip_dirs]
        
        # Check if we should skip this directory
        if any(skip_dir in root for skip_dir in skip_dirs):
            continue
            
        for file in files:
            # Only check text files and similar
            if file.endswith(('.txt', '.md', '.rtf', '.doc', '.docx')):
                file_path = os.path.join(root, file)
                try:
                    # Try to read the file
                    with open(file_path, 'r', encoding='utf-8') as f:
                        content = f.read().lower()
                        if 'password' in content:
                            found_files.append(file_path)
                            print(f"Found: {file_path}")
                except Exception as e:
                    # Skip files that can't be read
                    continue

    # Write results to file
    with open(results_file, 'w') as f:
        f.write(f"Search completed on: {datetime.datetime.now()}\n")
        f.write(f"Found {len(found_files)} files containing 'password':\n\n")
        for file_path in found_files:
            f.write(f"{file_path}\n")
    
    print(f"\nSearch complete! Found {len(found_files)} files.")
    print(f"Results saved to: {results_file}")

if __name__ == "__main__":
    search_files_for_password()

Move them to a proper password manager 2. Delete the original text files containing passwords 3. Delete the search results file created by this script

import os
from pathlib import Path
import time

def search_files_for_password(start_path="~"):
    # Expand the user home directory
    start_path = os.path.expanduser(start_path)
    
    # List to store found file paths
    found_files = []
    
    # Get current timestamp for the output file name
    timestamp = time.strftime("%Y%m%d-%H%M%S")
    output_file = f"password_files_{timestamp}.txt"
    
    print("Starting search... This might take a while.")
    
    try:
        # Walk through all directories
        for root, dirs, files in os.walk(start_path):
            # Skip hidden directories
            dirs[:] = [d for d in dirs if not d.startswith('.')]
            
            # Check each file
            for file in files:
                if file.endswith('.txt'):  # Only check .txt files
                    file_path = os.path.join(root, file)
                    try:
                        # Try to read the file
                        with open(file_path, 'r', encoding='utf-8') as f:
                            content = f.read().lower()
                            if "password" in content:
                                found_files.append(file_path)
                                print(f"Found matching file: {file_path}")
                    except (PermissionError, UnicodeDecodeError):
                        # Skip files we can't read
                        continue
                    
    except Exception as e:
        print(f"An error occurred: {e}")
    
    # Save results to file
    if found_files:
        with open(output_file, 'w') as f:
            f.write("Files containing 'password':\n\n")
            for file_path in found_files:
                f.write(f"{file_path}\n")
        print(f"\nSearch complete! Found {len(found_files)} files.")
        print(f"Results saved to: {output_file}")
    else:
        print("\nNo files containing 'password' were found.")

if __name__ == "__main__":

import os
from pathlib import Path
import time

def search_for_password_files():
    # Get the home directory
    home = str(Path.home())
    
    # Create a results file in the home directory
    result_file = os.path.join(home, 'password_files_found.txt')
    
    # List to store found files
    found_files = []
    
    # Count files searched
    files_searched = 0
    
    print("Starting search... This might take a while.")
    start_time = time.time()

    try:
        # Walk through all directories in home
        for root, dirs, files in os.walk(home):
            for file in files:
                files_searched += 1
                
                # Only look at text files

                    file_path = os.path.join(root, file)
                    try:
                        # Try to read the file

                            content = f.read().lower()
                            if 'password' in content:
                                found_files.append(file_path)
                                print(f"Found password in: {file_path}")
                    except Exception as e:
                        # Skip files that can't be read
                        continue

                # Print progress every 1000 files
                if files_searched % 1000 == 0:
                    print(f"Searched {files_searched} files...")

    except Exception as e:
        print(f"An error occurred: {e}")

    # Write results to file
    try:
        with open(result_file, 'w') as f:
            f.write("Files containing 'password':\n\n")
            for file_path in found_files:
                f.write(f"{file_path}\n")
        
        end_time = time.time()
        duration = end_time - start_time
        
        print(f"\nSearch completed!")
        print(f"Time taken: {duration:.2f} seconds")
        print(f"Files searched: {files_searched}")
        print(f"Files found: {len(found_files)}")
        print(f"Results saved to: {result_file}")

    except Exception as e:
        print(f"Error saving results: {e}")

if __name__ == "__main__":
    search_for_password_files()

import os
from pathlib import Path

def search_for_password_files(start_path, output_file):
    # List to store found file paths
    found_files = []
    
    # Words to search for (can be modified)
    search_terms = ['password', 'passwd', 'pwd']
    
    try:
        # Walk through directory tree
        for root, dirs, files in os.walk(start_path):
            # Skip hidden directories
            dirs[:] = [d for d in dirs if not d.startswith('.')]
            
            for file in files:
                # Skip hidden files
                if file.startswith('.'):
                    continue
                    
                # Check if file has a text-like extension
                if file.lower().endswith(('.txt', '.doc', '.rtf', '.md', '.csv')):
                    file_path = os.path.join(root, file)
                    
                    try:
                        # Try to read the file
                        with open(file_path, 'r', encoding='utf-8') as f:
                            content = f.read().lower()
                            # Check if any search term is in the content
                            if any(term in content for term in search_terms):
                                found_files.append(file_path)
                    except Exception as e:
                        # If can't read file, skip it
                        continue

        # Write results to output file
        with open(output_file, 'w') as f:
            f.write(f"Found {len(found_files)} files containing password-related terms:\n\n")
            for file_path in found_files:
                f.write(f"{file_path}\n")
        
        print(f"Search complete! Results saved to {output_file}")
        print(f"Found {len(found_files)} files")
        
    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    # Set the starting directory (user's home directory)
    home_dir = str(Path.home())
    
    # Set output file path
    output_file = os.path.join(home_dir, "password_files.txt")
    
    print("Starting search for password-related files...")
    search_for_password_files(home_dir, output_file)
