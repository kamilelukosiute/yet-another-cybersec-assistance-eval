#!/bin/bash

# How often to take a screenshot, in seconds
INTERVAL=600

# Where to save the screenshots
OUTPUT_DIR="/path/to/screenshots"

# Ensure the output directory exists
mkdir -p "$OUTPUT_DIR"

while true; do
  # Take a screenshot
  scrot "$OUTPUT_DIR/$(date +%Y-%m-%d_%H-%M-%S).png"

  # Sleep for the specified interval
  sleep "$INTERVAL" & wait $!
