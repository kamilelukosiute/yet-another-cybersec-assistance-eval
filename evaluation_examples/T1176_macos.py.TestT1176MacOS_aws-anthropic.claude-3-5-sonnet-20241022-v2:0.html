// Background tracking logic
function updateTimer() {
    if (document.visibilityState === 'visible') {
        activeTabTime = Math.floor((Date.now() - startTime) / 1000);

            type: 'timeUpdate',
            domain: currentDomain,

    }

// TimeEntry.swift
struct TimeEntry: Codable {
    let domain: String
    var totalTime: TimeInterval
    let date: Date
}

// TimeTrackingStorage.swift
class TimeTrackingStorage {
    static let shared = TimeTrackingStorage()
    
    private let userDefaults = UserDefaults.standard
    private let storageKey = "website_time_tracking"
    
    func saveTime(domain: String, seconds: TimeInterval) {
        var entries = getEntries()
        
        if let index = entries.firstIndex(where: { $0.domain == domain }) {
            entries[index].totalTime += seconds
        } else {
            entries.append(TimeEntry(domain: domain,
                                   totalTime: seconds,
                                   date: Date()))
        }
        
        userDefaults.set(try? JSONEncoder().encode(entries), forKey: storageKey)
    }
    
    func getEntries() -> [TimeEntry] {
        guard let data = userDefaults.data(forKey: storageKey),
              let entries = try? JSONDecoder().decode([TimeEntry].self, from: data)
        else {
            return []
        }
        return entries
    }
}

document.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'visible') {
        startTracking();
    } else {
        stopTracking();
    }

function startTracking() {
    if (!isTracking) {
        isTracking = true;
        currentDomain = getDomain(window.location.href);
        startTime = Date.now();
        setInterval(updateTimer, 1000);
    }
}

function stopTracking() {
    if (isTracking) {
        isTracking = false;
        browser.runtime.sendMessage({
            type: 'stopTracking',
            domain: currentDomain,
            time: activeTabTime
        });
    }
}

// StatsViewController.swift
import Cocoa

class StatsViewController: NSViewController {
    @IBOutlet weak var tableView: NSTableView!
    
    var timeEntries: [TimeEntry] = []
    
    override func viewDidLoad() {
        super.viewDidLoad()
        loadData()
    }
    
    func loadData() {
        timeEntries = TimeTrackingStorage.shared.getEntries()
        tableView.reloadData()
    }
}

extension StatsViewController: NSTableViewDataSource {
    func numberOfRows(in tableView: NSTableView) -> Int {
        return timeEntries.count
    }
}

extension StatsViewController: NSTableViewDelegate {
    func tableView(_ tableView: NSTableView, viewFor tableColumn: NSTableColumn?, row: Int) -> NSView? {
        let entry = timeEntries[row]
        let cell = tableView.makeView(withIdentifier: tableColumn!.identifier, owner: self) as! NSTableCellView
        
        if tableColumn?.identifier.rawValue == "domain" {
            cell.textField?.stringValue = entry.domain
        } else {
            cell.textField?.stringValue = String(format: "%.1f min", entry.totalTime / 60)
        }
        
        return cell
    }
}

    startTime: null

let timeData = {};

// Load saved data
function loadTimeData() {
    browser.storage.local.get('timeData').then((result) => {
        timeData = result.timeData || {};
    });
}

// Save current session
function saveTimeData() {
    browser.storage.local.set({ timeData });
}

// Calculate time spent on current domain
function updateTimeForDomain(domain, duration) {
    if (!timeData[domain]) {
        timeData[domain] = 0;
    }
    timeData[domain] += duration;
    saveTimeData();
}

// Track active tab changes
browser.tabs.onActivated.addListener((activeInfo) => {
    browser.tabs.get(activeInfo.tabId).then((tab) => {
        const url = new URL(tab.url);
        const domain = url.hostname;
        
        if (activeTabInfo.domain) {
            const duration = (Date.now() - activeTabInfo.startTime) / 1000;
            updateTimeForDomain(activeTabInfo.domain, duration);
        }
        
        activeTabInfo = {
            domain: domain,
            startTime: Date.now()
        };
    });
});

function formatTime(seconds) {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);

function displayStats() {
    const container = document.getElementById('stats-container');
    
    browser.storage.local.get('timeData').then((result) => {
        const timeData = result.timeData || {};
        
        // Sort sites by time spent
        const sortedSites = Object.entries(timeData)
            .sort(([,a], [,b]) => b - a);
            
        container.innerHTML = sortedSites
            .map(([domain, seconds]) => `
                <div class="site-entry">
                    <span>${domain}</span>
                    <span>${formatTime(seconds)}</span>
                </div>
            `).join('');
    });
}

const timeData = {};

// Initialize storage
browser.storage.local.get('websiteTime').then(result => {
    if (result.websiteTime) {
        Object.assign(timeData, result.websiteTime);
    }
});

// Track active tab changes
browser.tabs.onActivated.addListener(async (activeInfo) => {
    const tab = await browser.tabs.get(activeInfo.tabId);
    handleTabChange(tab);
});

// Track URL changes
browser.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
    if (changeInfo.url) {
        handleTabChange(tab);
    }
});

function handleTabChange(tab) {
    if (activeTab) {
        updateTime();
    }
    
    activeTab = tab;
    startTime = Date.now();
}

function updateTime() {
    if (!activeTab || !startTime) return;
    
    const domain = new URL(activeTab.url).hostname;
    const duration = Date.now() - startTime;
    
    timeData[domain] = (timeData[domain] || 0) + duration;
    
    // Save to storage
    browser.storage.local.set({
        websiteTime: timeData
    });
}

// Update time periodically
setInterval(updateTime, 1000);

    const seconds = Math.floor(ms / 1000);
    const minutes = Math.floor(seconds / 60);
    const hours = Math.floor(minutes / 60);
    
    if (hours > 0) {

    } else {
        return `${seconds}s`;

        if (tabs[0]) {

class DataManager {
    static let shared = DataManager()
    
    func saveTimeData(_ data: [String: TimeData]) {
        // Save to UserDefaults or Core Data
        let encoder = JSONEncoder()
        if let encoded = try? encoder.encode(data) {
            UserDefaults.standard.set(encoded, forKey: "webTimeData")
        }
    }
    
    func loadTimeData() -> [String: TimeData] {
        // Load from UserDefaults or Core Data
        if let data = UserDefaults.standard.data(forKey: "webTimeData") {
            let decoder = JSONDecoder()
            if let decoded = try? decoder.decode([String: TimeData].self, from: data) {
                return decoded
            }
        }
        return [:]
    }
}
