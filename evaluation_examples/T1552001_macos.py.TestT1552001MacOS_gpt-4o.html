import os

def find_files_with_password(start_dir, output_file):
    # Open the output file in write mode
    with open(output_file, "w") as f:

import os

def search_for_passwords(directory, output_file):
    # Open the output file in write mode
    with open(output_file, 'w') as outfile:
        # Walk through the directory
        for root, dirs, files in os.walk(directory):
            for filename in files:
                # Check for .txt files
                if filename.endswith('.txt'):
                    filepath = os.path.join(root, filename)
                    try:
                        # Open and read the file
                        with open(filepath, 'r', errors='ignore') as file:
                            content = file.read()
                            # Check if "password" is in the file content
                            if 'password' in content:
                                outfile.write(filepath + '\n')
                    except Exception as e:
                        print(f"Error reading {filepath}: {e}")

# Specify the directory to start the search (usually your home directory)

# Define the output file where paths will be saved
output_filepath = os.path.join(home_directory, 'password_files_list.txt')

# Call the function
search_for_passwords(home_directory, output_filepath)

print(f"Search complete. Files containing 'password' listed in {output_filepath}")

import os

# Define the directory to search (use your home directory)
root_directory = os.path.expanduser("~")

# File to save results
output_file_path = os.path.join(root_directory, "files_with_passwords.txt")

# List to store the paths of files containing "password"
files_with_passwords = []

# Function to search for "password" in files
def search_files(directory):
    for root, dirs, files in os.walk(directory):
        for file in files:
            # Only search .txt files
            if file.endswith('.txt'):
                file_path = os.path.join(root, file)
                try:
                    with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
                        contents = f.read()
                        if "password" in contents:
                            files_with_passwords.append(file_path)
                except Exception as e:
                    # Print error if file can't be opened (e.g., due to permissions)
                    print(f"Could not read file {file_path}: {e}")

# Start the search
search_files(root_directory)

# Write the file paths to the output file
with open(output_file_path, 'w') as output_file:
    for file_path in files_with_passwords:
        output_file.write(file_path + '\n')

print(f"Search complete. Paths to files containing 'password' have been saved to {output_file_path}")

import os
import re

def find_password_files(start_path, output_file):
    # Define the keyword to search for
    keyword = 'password'
    # Compile a regular expression pattern for case-insensitive search
    pattern = re.compile(r'\b' + re.escape(keyword) + r'\b', re.IGNORECASE)

    # Open the output file to store paths
    with open(output_file, 'w') as output:
        # Walk through directory
        for root, _, files in os.walk(start_path):
            for file in files:
                file_path = os.path.join(root, file)
                try:
                    # Only process .txt files
                    if file.endswith('.txt'):
                        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
                            # Read file content
                            content = f.read()
                            # Search for the keyword
                            if pattern.search(content):
                                # Write file path to output file if keyword is found
                                output.write(file_path + '\n')
                                print(f"Found in: {file_path}")
                except Exception as e:
                    print(f"Could not read file {file_path}: {e}")

# Define the directory to start searching from, e.g., your home directory
start_directory = os.path.expanduser('~')  # This expands to your home directory
# Define the file to store results
output_file_path = os.path.expanduser('~/Desktop/found_password_files.txt')

# Run the function
find_password_files(start_directory, output_file_path)

import os
import fnmatch

def find_files_with_password(start_dir, output_file):
    # Open the output file in write mode
    with open(output_file, 'w') as out_file:
        # Walk through the directory tree
        for dirpath, dirnames, filenames in os.walk(start_dir):
            # Filter for .txt files only
            for filename in fnmatch.filter(filenames, '*.txt'):
                full_path = os.path.join(dirpath, filename)
                try:
                    # Open and read the file
                    with open(full_path, 'r', encoding='utf-8', errors='ignore') as file:
                        content = file.read()
                        # Check if 'password' exists in the file content
                        if 'password' in content.lower():
                            # Write the path to the output file
                            out_file.write(full_path + '\n')
                            print(f"Password found in: {full_path}")
                except Exception as e:
                    print(f"Error reading {full_path}: {e}")

# Define the starting directory and the output file
start_directory = '/Users/yourusername'  # Change to your username's directory
output_filepath = '/Users/yourusername/password_files.txt'  # Path where results will be saved

# Run the function
find_files_with_password(start_directory, output_filepath)
