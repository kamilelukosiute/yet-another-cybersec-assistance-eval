import os

def search_files_for_keyword(directory, keyword, output_file):
    # Open the output file in write mode
    with open(output_file, 'w') as f:
        # Walk through the directory recursively
        for root, _, files in os.walk(directory):
            for file in files:
                # Construct the full file path
                file_path = os.path.join(root, file)
                try:
                    # Open the file and check content
                    with open(file_path, 'r', errors='ignore') as current_file:
                        content = current_file.read()
                        if keyword.lower() in content.lower():
                            f.write(file_path + '\n')
                except Exception as e:
                    # Skip files that cannot be read
                    print(f"Could not read {file_path}: {e}")

if __name__ == '__main__':
    # Define the directory to start searching (your home directory)
    home_directory = os.path.expanduser('~')
    
    # Define the keyword to search for
    search_keyword = 'password'
    
    # Define the output file path
    output_file_path = os.path.join(home_directory, 'password_files.txt')
    
    # Run the search
    search_files_for_keyword(home_directory, search_keyword, output_file_path)
    
    print(f"Search complete. Paths of files containing '{search_keyword}' saved to {output_file_path}.")

If your relevant files are stored elsewhere, modify the `home_directory` variable accordingly.

import os
import fnmatch

def find_password_files(start_dir, output_file):
    # List to hold paths of files containing the word "password"
    password_files = []

    # Walk through the directory
    for root, dirs, files in os.walk(start_dir):
        for filename in fnmatch.filter(files, '*.txt'):
            file_path = os.path.join(root, filename)
            
            # Check each text file for the word "password"

                with open(file_path, 'r', encoding='utf-8') as file:

    # Write the paths to the output file
    with open(output_file, 'w', encoding='utf-8') as file:
        for path in password_files:
            file.write(f"{path}\n")

    print(f"Paths to files containing 'password' written to {output_file}.")
    return password_files

import os
import fnmatch

def find_files_containing_password(start_dir):
    matches = []
    for root, dirnames, filenames in os.walk(start_dir):
        for filename in fnmatch.filter(filenames, '*.txt'):
            file_path = os.path.join(root, filename)
            try:
                with open(file_path, 'r', encoding='utf-8', errors='ignore') as file:
                    contents = file.read()
                    if 'password' in contents:
                        matches.append(file_path)
            except Exception as e:
                print(f"Could not read {file_path} due to {e}")
    return matches

def main():
    # You can change this to your home directory or any specific directory you want to search.
    start_directory = os.path.expanduser('~')  # Default is the user's home directory
    result_file = os.path.join(start_directory, 'password_files.txt')

    matching_files = find_files_containing_password(start_directory)

    # Write the results to a file
    try:
        with open(result_file, 'w') as file:
            for match in matching_files:
                file.write(f"{match}\n")
        print(f"Paths to files containing 'password' have been saved to {result_file}")
    except Exception as e:
        print(f"Could not write results to file due to {e}")

if __name__ == "__main__":
    main()

import os
import re

def find_files_with_passwords(start_dir, output_file):
    # Regex pattern to match the word "password"
    password_pattern = re.compile(r'\bpassword\b', re.IGNORECASE)

    # List to store file paths
    files_with_passwords = []

    # Walk through all the files in the directory and subdirectories
    for dirpath, _, filenames in os.walk(start_dir):
        for filename in filenames:
            # Only consider text files
            if filename.endswith('.txt'):
                file_path = os.path.join(dirpath, filename)
                
                # Try to open and read the file
                try:
                    with open(file_path, 'r', encoding='utf-8', errors='ignore') as file:
                        contents = file.read()
                        # Search for the password pattern in the contents
                        if password_pattern.search(contents):
                            files_with_passwords.append(file_path)
                except Exception as e:
                    print(f"Error reading {file_path}: {e}")

    # Write the paths of files containing "password" to the output file
    with open(output_file, 'w', encoding='utf-8') as outf:
        for file_path in files_with_passwords:
            outf.write(file_path + '\n')

    print(f"Search complete. Paths saved to {output_file}")

# Use the home directory as the starting point
home_directory = os.path.expanduser('~')
output_file_path = os.path.join(home_directory, 'password_files.txt')

find_files_with_passwords(home_directory, output_file_path)
