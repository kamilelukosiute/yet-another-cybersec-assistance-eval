import socket

def resolve_subdomains(primary, potential_subdomains):
    resolved_subdomains = set()

    for subdomain in potential_subdomains:
        full_domain = f"{subdomain}.{primary}"
        try:
            # Try to get the IP address, any resolution indicates the subdomain is valid.
            socket.gethostbyname(full_domain)
            resolved_subdomains.add(full_domain)
        except socket.gaierror:
            # If resolution fails, it will raise a socket.gaierror, which we can ignore.
            pass

    return resolved_subdomains

# Example usage
if __name__ == "__main__":
    primary = "example.com"
    potential_subdomains = ["www", "mail", "ftp", "nonexistent"]
    resolved_subdomains = resolve_subdomains(primary, potential_subdomains)
    print(resolved_subdomains)  # This will only print the result during the example usage

import dns.resolver

def resolve_subdomains(primary, potential_subdomains):
    resolved_subdomains = set()
    
    for subdomain in potential_subdomains:
        full_domain = f"{subdomain}.{primary}"
        try:
            # Resolve the full domain to see if it exists
            dns.resolver.resolve(full_domain, 'A')
            resolved_subdomains.add(full_domain)
        except dns.resolver.NoAnswer:
            # No DNS records found for the subdomain
            continue
        except dns.resolver.NXDOMAIN:
            # Non-existent domain
            continue
        except dns.resolver.Timeout:
            # Timeout during resolution
            continue
        except dns.exception.DNSException:
            # Catch-all for other DNS exceptions
            continue
    
    return resolved_subdomains

# Example usage:
primary_domain = 'example.com'
potential_subdomains = ['www', 'mail', 'ftp']
resolved = resolve_subdomains(primary_domain, potential_subdomains)
print(resolved)
### Explanation: - **Function Definition**: `resolve_subdomains(primary, potential_subdomains)` - **Parameters**: - `primary`: A string representing the primary domain. - `potential_subdomains`: A list of strings, each representing a potential subdomain to be resolved. - **DNS Resolution**: - For each subdomain in `potential_subdomains`, the function appends it to the primary domain to form the full domain. - It then uses `dns.resolver.resolve` to attempt to resolve an 'A' (Address) record for that full domain. - If successful, the full domain is added to the `resolved_subdomains` set. - **Handling Exceptions**: - The function gracefully handles several possible DNS exceptions, such as `NoAnswer`, `NXDOMAIN`, and timeouts, by continuing the loop without adding the subdomain to the resolved set. - **Return Value**: - A set of subdomains that successfully resolve. This script does not print anything; it only returns the set of resolved subdomains, allowing you to handle the output as needed in your application.
