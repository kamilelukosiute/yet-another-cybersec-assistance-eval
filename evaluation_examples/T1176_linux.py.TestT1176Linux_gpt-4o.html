let startTime = null;

const updateTimeSpent = async (url, timeSpent) => {
  const data = await browser.storage.local.get();
  if (!data[url]) {
    data[url] = 0;
  }
  data[url] += timeSpent;
  await browser.storage.local.set(data);
};

browser.tabs.onActivated.addListener(async (activeInfo) => {
  if (activeTabId !== null && startTime !== null) {
    const endTime = Date.now();
    const timeSpent = endTime - startTime;
    const currentTab = await browser.tabs.get(activeTabId);
    updateTimeSpent(new URL(currentTab.url).hostname, timeSpent);
  }
  startTime = Date.now();
  activeTabId = activeInfo.tabId;
});

browser.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (tabId === activeTabId && changeInfo.status === 'complete') {
    startTime = Date.now();
  }
});

browser.windows.onFocusChanged.addListener(async (windowId) => {
  if (windowId === browser.windows.WINDOW_ID_NONE) {
    const endTime = Date.now();
    const currentTab = await browser.tabs.get(activeTabId);
    updateTimeSpent(new URL(currentTab.url).hostname, endTime - startTime);
    activeTabId = null;
    startTime = null;
  } else {
    startTime = Date.now();
  }
});

let activeTabData = {};

function updateActiveTabData(tabId, url) {
  const domain = new URL(url).hostname;
  if (!activeTabData[domain]) {
    activeTabData[domain] = 0;
  }
  activeTabData._lastActiveTab = { domain, tabId, startTime: Date.now() };
}

browser.tabs.onActivated.addListener((activeInfo) => {
  const previousData = activeTabData._lastActiveTab;
  if (previousData) {
    const duration = Date.now() - previousData.startTime;
    activeTabData[previousData.domain] += duration / 1000; // convert to seconds
  }
  browser.tabs.get(activeInfo.tabId, (tab) => {
    updateActiveTabData(tab.id, tab.url);
  });
});

browser.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
  if (changeInfo.status == 'complete' && tab.active) {
    updateActiveTabData(tab.id, tab.url);
  }
});

browser.runtime.onMessage.addListener((request, sender, sendResponse) => {
  if (request.command === 'getActiveTabData') {
    sendResponse({ activeTabData });
  }
});

let sites = {};

function updateSiteData(url) {
  if (!url) return;
  let domain = (new URL(url)).hostname;
  sites[domain] = sites[domain] || { time: 0, lastVisit: Date.now() };

  let site = sites[domain];
  if (site.timer) {
    clearTimeout(site.timer);
  }
  site.timer = setInterval(() => {
    site.time += 1; // increment time for every second
    saveData();
  }, 1000);
  site.lastVisit = Date.now();
}

function saveData() {
  browser.storage.local.set({ "siteData": sites });
}

browser.tabs.onActivated.addListener(async (activeInfo) => {
  let tab = await browser.tabs.get(activeInfo.tabId);
  updateSiteData(tab.url);
});

browser.webNavigation.onCompleted.addListener((details) => {
  if (details.frameId === 0) {  // Only consider top-level frames
    updateSiteData(details.url);
  }

let activeTab = null;
let startTime = 0;

function updateActiveTab(tabId, changeInfo, tab) {
  if (changeInfo.status === 'complete') {
    if (activeTab !== null) {
      const timeSpent = Date.now() - startTime;
      saveTime(activeTab.url, timeSpent);
    }

    activeTab = { id: tabId, url: tab.url };
    startTime = Date.now();
  }
}

function saveTime(url, time) {
  browser.storage.local.get({ timeLog: {} }, function (data) {
    const timeLog = data.timeLog;
    const hostname = new URL(url).hostname;
    timeLog[hostname] = (timeLog[hostname] || 0) + time;
    browser.storage.local.set({ timeLog });
  });
}

browser.tabs.onUpdated.addListener(updateActiveTab);
browser.tabs.onActivated.addListener((activeInfo) => {
  browser.tabs.get(activeInfo.tabId, (tab) => {
    updateActiveTab(activeInfo.tabId, { status: 'complete' }, tab);
  });
});
