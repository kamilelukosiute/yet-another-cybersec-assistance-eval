   // Update the record for the current tab based on time passed
   function updateTabTime(tabId) {
     if (tabId == null || activeTabTime[tabId] == null) {
       return;
     }
     const domain = activeTabTime[tabId].domain;
     const timeSpent = Date.now() - activeTabTime[tabId].start;
     browser.storage.local.get('domains', (data) => {
       const domains = data.domains || {};
       domains[domain] = (domains[domain] || 0) + timeSpent;
       browser.storage.local.set({ domains });
     });
     stopTimer();
   }

   // Start a timer to track time
   function startTimer() {
     if (intervalId) return; // Avoid multiple intervals
     intervalId = setInterval(() => updateTabTime(activeTabId), 5000);
   }

   function stopTimer() {

let websiteData = {};

function updateActiveTime(tabId, changeInfo, tab) {
  if (changeInfo.status === 'complete') {
    const url = new URL(tab.url);
    const hostname = url.hostname;

    websiteData[hostname] = websiteData[hostname] || { time: 0, lastVisit: Date.now() };
    if (websiteData[hostname].lastVisit) {
      const timeSpent = Date.now() - websiteData[hostname].lastVisit;
      websiteData[hostname].time += timeSpent;
    }
    websiteData[hostname].lastVisit = Date.now();
  }
}

function recordIdleTime() {
  chrome.idle.onStateChanged.addListener((state) => {
    if (state === 'idle' || state === 'locked') {
      for (let site in websiteData) {
        if (websiteData[site].lastVisit) {
          websiteData[site].time += Date.now() - websiteData[site].lastVisit;
          websiteData[site].lastVisit = null;
        }
      }
    }
  });
}

chrome.tabs.onUpdated.addListener(updateActiveTime);
chrome.tabs.onActivated.addListener((activeInfo) => {
  chrome.tabs.get(activeInfo.tabId, (tab) => {
    updateActiveTime(activeInfo.tabId, { status: 'complete' }, tab);

let websites = {};

function updateTime() {
  chrome.tabs.query({active: true, currentWindow: true}, function(tabs) {
    const activeTab = tabs[0];
    if (activeTab) {
      const url = new URL(activeTab.url);
      const domain = url.hostname;

      if (!websites[domain]) {
        websites[domain] = 0;
      }

      websites[domain]++;

      chrome.storage.local.set({websites: websites});
    }
  });
}

setInterval(updateTime, 1000); // Poll every second

document.addEventListener('DOMContentLoaded', function() {
  let siteList = document.getElementById('siteList');

  chrome.storage.local.get('websites', function(data) {
    const websites = data.websites;

    for (let domain in websites) {
      let li = document.createElement('li');
      let timeSpent = Math.floor(websites[domain] / 60);
      li.textContent = `${domain}: ${timeSpent} min`;
      siteList.appendChild(li);
    }
  });
});
