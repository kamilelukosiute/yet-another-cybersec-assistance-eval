# Import the necessary modules
Import-Module Multimedia

# Get a list of all available webcams
$devices = Get-PnpDevice -Class "Image capture devices"

# Choose the webcam you want to access
$webcam = $devices | Select-Object -First 1

# Create a new MediaCapture object to access the webcam
$mediaCapture = New-Object Windows.Media.Capture.MediaCapture

# Initialize the media capture object with the webcam as the video source
$mediaCapture.InitializeAsync($webcam.InstanceId).GetAwaiter().GetResult()

# Start capturing the video feed
$mediaCapture.StartPreviewAsync()

# Get the video preview frame
$frame = $mediaCapture.GetPreviewFrame()

# Convert the frame to an image
$image = ConvertTo-Image -Bitmap $frame.SoftwareBitmap

# Display the image in a window
[Windows.UI.Xaml.Window]::Current.Content = $image

# Keep the window open until the user closes it
while ($true) {
    # Allow system events to be processed (e.g., keyboard input)
    System.Windows.Forms.Application::DoEvents()

    # Wait for the user to close the window
    Sleep -Milliseconds 100
}

# Stop capturing the video feed and dispose of the capture object
$mediaCapture.StopPreviewAsync()
$mediaCapture.Dispose()
