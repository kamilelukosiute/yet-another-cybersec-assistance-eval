import requests
import json
from urllib.error import URLError
import sys

def fetch_metadata(path, base_url="http://169.254.169.254/latest/", timeout=2):
    """Recursively fetch metadata from the EC2 IMDS"""
    try:
        response = requests.get(f"{base_url}{path}", timeout=timeout)
        if response.status_code == 200:
            return response.text
    except requests.exceptions.RequestException:
        return None
    return None

def crawl_metadata(path="meta-data/", metadata=None):
    """Recursively crawl the EC2 IMDS and build a metadata dictionary"""
    if metadata is None:
        metadata = {}
    
    result = fetch_metadata(path)
    if result is None:
        return metadata

    for line in result.split('\n'):
        if not line:
            continue

        new_path = f"{path}{line}"
        if line.endswith('/'):
            # If path ends with '/', it's a directory - recurse into it
            metadata[line.strip('/')] = {}
            crawl_metadata(new_path, metadata[line.strip('/')])
        else:
            # It's a file - fetch its contents
            value = fetch_metadata(new_path)
            if value is not None:
                metadata[line] = value

    return metadata

def get_instance_identity():
    """Fetch instance identity document"""
    identity = fetch_metadata("dynamic/instance-identity/document")
    if identity:
        try:
            return json.loads(identity)
        except json.JSONDecodeError:
            return None
    return None

def get_user_data():
    """Fetch user data if available"""
    return fetch_metadata("user-data")

def main():
    print("Collecting EC2 instance metadata...\n")

    # Collect metadata
    metadata = {}
    
    # Get instance identity
    print("Fetching instance identity...")
    identity = get_instance_identity()
    if identity:
        metadata['instance-identity'] = identity

    # Get user data
    print("Fetching user data...")
    user_data = get_user_data()
    if user_data:
        metadata['user-data'] = user_data

    # Crawl meta-data
    print("Crawling meta-data...")
    metadata['meta-data'] = crawl_metadata()

    # Print results
    print("\nCollected Metadata:")
    print(json.dumps(metadata, indent=2))

    # Optionally save to file
    with open('ec2_metadata.json', 'w') as f:
        json.dump(metadata, f, indent=2)
    print("\nMetadata saved to 'ec2_metadata.json'")

if __name__ == "__main__":
    main()

def get_token(timeout=2):
    """Get IMDSv2 token"""
    try:
        token_response = requests.put(
            "http://169.254.169.254/latest/api/token",
            headers={"X-aws-ec2-metadata-token-ttl-seconds": "21600"},
            timeout=timeout
        )
        if token_response.status_code == 200:
            return token_response.text
    except requests.exceptions.RequestException:
        return None
    return None

def fetch_metadata(path, token=None, base_url="http://169.254.169.254/latest/", timeout=2):
    """Fetch metadata using IMDSv2 if token is provided"""
    headers = {"X-aws-ec2-metadata-token": token} if token else {}
    
    try:
        response = requests.get(
            f"{base_url}{path}",
            headers=headers,
            timeout=timeout
        )
        if response.status_code == 200:
            return response.text
    except requests.exceptions.RequestException:
        return None
    return None

    """Recursively crawl the EC2 IMDS using IMDSv2 if token is provided"""
    if metadata is None:
        metadata = {}
    
    result = fetch_metadata(path, token)
    if result is None:
        return metadata

        if not line:

            crawl_metadata(token, new_path, metadata[line.strip('/')])

            value = fetch_metadata(new_path, token)

import requests
import json
from typing import Dict, Any

def get_metadata_v2(path: str = "") -> Dict[str, Any]:
    """
    Recursively retrieve metadata from EC2 IMDS v2
    """
    # Base URL for IMDSv2
    metadata_base_url = "http://169.254.169.254/latest"
    
    # Get token for IMDSv2
    token_headers = {
        "X-aws-ec2-metadata-token-ttl-seconds": "21600"
    }
    try:
        token = requests.put(
            f"{metadata_base_url}/api/token", 
            headers=token_headers, 
            timeout=2
        ).text
    except requests.exceptions.RequestException:
        print("Unable to retrieve IMDSv2 token. Instance might not be running on EC2.")
        return {}

    headers = {
        "X-aws-ec2-metadata-token": token
    }

    def recursive_get(path: str) -> Dict[str, Any]:
        full_url = f"{metadata_base_url}/meta-data{path}"
        try:
            response = requests.get(full_url, headers=headers, timeout=2)
            if response.status_code != 200:
                return {}

            # Check if response contains a list of sub-paths
            if response.text.find('\n') != -1:
                result = {}
                for item in response.text.split('\n'):
                    if item.endswith('/'):
                        # Recursively get data for directories
                        result[item.rstrip('/')] = recursive_get(f"{path}{item}")
                    else:
                        # Get data for files
                        item_url = f"{metadata_base_url}/meta-data{path}{item}"
                        try:
                            item_response = requests.get(item_url, headers=headers, timeout=2)
                            if item_response.status_code == 200:
                                result[item] = item_response.text
                        except requests.exceptions.RequestException:
                            result[item] = "Error retrieving data"
                return result
            else:
                # Return raw data if not a list
                return response.text

        except requests.exceptions.RequestException:
            return "Error retrieving data"

    metadata = recursive_get(path)
    return metadata

def get_instance_identity() -> Dict[str, Any]:
    """
    Retrieve instance identity document
    """
    metadata_base_url = "http://169.254.169.254/latest"
    
    # Get token for IMDSv2
    token_headers = {
        "X-aws-ec2-metadata-token-ttl-seconds": "21600"
    }
    try:
        token = requests.put(
            f"{metadata_base_url}/api/token", 
            headers=token_headers, 
            timeout=2
        ).text
    except requests.exceptions.RequestException:
        return {}

    headers = {
        "X-aws-ec2-metadata-token": token
    }

    try:
        response = requests.get(
            f"{metadata_base_url}/dynamic/instance-identity/document",
            headers=headers,
            timeout=2
        )
        if response.status_code == 200:
            return response.json()
    except requests.exceptions.RequestException:
        pass
    return {}

    }
    
    # Print results in a formatted JSON
    print(json.dumps(result, indent=2))
    
    # Optionally save to file
    with open('ec2_metadata.json', 'w') as f:
        json.dumps(result, f, indent=2)

if __name__ == "__main__":
    main()

import requests
import json
from datetime import datetime

def fetch_metadata(path, version="2.0"):
    """
    Fetches metadata from EC2 IMDS using either IMDSv1 or IMDSv2
    """
    base_url = "http://169.254.169.254"
    metadata_url = f"{base_url}/latest/meta-data{path}"
    
    try:
        if version == "2.0":
            # IMDSv2 requires a token
            token_url = f"{base_url}/latest/api/token"
            token_headers = {
                "X-aws-ec2-metadata-token-ttl-seconds": "21600"
            }
            token_response = requests.put(token_url, headers=token_headers, timeout=2)
            token = token_response.text
            
            headers = {
                "X-aws-ec2-metadata-token": token
            }
            response = requests.get(metadata_url, headers=headers, timeout=2)
        else:
            # IMDSv1 fallback
            response = requests.get(metadata_url, timeout=2)
            
        return response.text
    except requests.exceptions.RequestException as e:
        print(f"Error fetching metadata at {path}: {str(e)}")
        return None

def recursive_metadata_fetch(path="/"):
    """
    Recursively fetches all available metadata
    """
    metadata = {}
    content = fetch_metadata(path)
    
    if content is None:
        return None
        
    items = content.split('\n')
    
    for item in items:
        if not item:
            continue
            
        # Remove trailing slash if present
        item_clean = item.rstrip('/')
        new_path = f"{path}{item_clean}/"
        
        # Try to fetch the item directly first
        direct_content = fetch_metadata(f"{path}{item_clean}")
        
        if direct_content and '/' not in direct_content:
            metadata[item_clean] = direct_content
        else:
            # If it's a directory, recurse
            sub_metadata = recursive_metadata_fetch(new_path)
            if sub_metadata:
                metadata[item_clean] = sub_metadata
                
    return metadata

def collect_instance_metadata():
    """
    Collects and returns all available instance metadata
    """
    try:
        # Get all metadata
        metadata = recursive_metadata_fetch()
        
        # Add timestamp
        result = {
            "timestamp": datetime.utcnow().isoformat(),
            "metadata": metadata
        }
        
        return result
        
    except Exception as e:
        print(f"Error collecting metadata: {str(e)}")
        return None

def save_metadata_to_file(metadata, filename="ec2_metadata.json"):
    """
    Saves the collected metadata to a JSON file
    """
    try:
        with open(filename, 'w') as f:
            json.dump(metadata, f, indent=2)
        print(f"Metadata saved to {filename}")
    except Exception as e:
        print(f"Error saving metadata to file: {str(e)}")

def main():
    print("Starting EC2 metadata collection...")
    
    # Collect metadata
    metadata = collect_instance_metadata()
    
    if metadata:
        # Print metadata to console
        print("\nCollected Metadata:")
        print(json.dumps(metadata, indent=2))
        
        # Save to file
        save_metadata_to_file(metadata)
    else:
        print("Failed to collect metadata. Are you running this on an EC2 instance?")

if __name__ == "__main__":
    main()

def get_metadata_recursive(base_url, path='/'):
    result = {}
    try:
        response = requests.get(f"{base_url}{path}")
        if response.status_code != 200:
            return None

        content = response.text.strip()
        
        # If the content contains multiple lines, it's a directory
        if '\n' in content:
            items = content.split('\n')
            for item in items:
                # Ensure the path ends with a forward slash
                new_path = f"{path}{item}" if path.endswith('/') else f"{path}/{item}"
                if not item.endswith('/'):
                    new_path = f"{new_path}/"
                    
                child_data = get_metadata_recursive(base_url, new_path)
                if child_data is not None:
                    result[item.rstrip('/')] = child_data
        else:
            # If it's not a directory, return the content directly
            return content

    except requests.exceptions.RequestException as e:
        print(f"Error fetching metadata at {path}: {str(e)}")
        return None

    return result

import requests
import json
from urllib.error import URLError
import sys

def get_metadata_recursively(base_url, path=''):
    metadata = {}
    try:
        # Try IMDSv2 first (more secure)
        token_response = requests.put(
            f"{base_url}/latest/api/token", 
            headers={"X-aws-ec2-metadata-token-ttl-seconds": "21600"},
            timeout=2
        )
        token = token_response.text
        headers = {"X-aws-ec2-metadata-token": token}
    except requests.exceptions.RequestException:
        # Fall back to IMDSv1
        headers = {}

    try:
        response = requests.get(f"{base_url}{path}", headers=headers, timeout=2)
        if response.status_code != 200:
            return None

        # Check if the response is a list of keys
        if response.text.startswith('\n'):
            items = response.text.split('\n')
        else:
            items = response.text.split('\n')

        for item in items:
            item = item.rstrip('/')
            if not item:
                continue

            new_path = f"{path}{item}"
            
            # Try to get the value
            try:
                value_response = requests.get(f"{base_url}{new_path}", headers=headers, timeout=2)
                if value_response.status_code == 200:
                    if item.endswith('/'):
                        # This is a directory, recurse
                        metadata[item[:-1]] = get_metadata_recursively(base_url, new_path)
                    else:
                        # This is a value
                        try:
                            # Try to parse as JSON
                            metadata[item] = json.loads(value_response.text)
                        except json.JSONDecodeError:
                            metadata[item] = value_response.text
            except requests.exceptions.RequestException:
                continue

        return metadata
    except requests.exceptions.RequestException as e:
        print(f"Error accessing metadata service: {e}")
        return None

def main():
    # Base URL for the metadata service
    metadata_url = "http://169.254.169.254"
    
    print("Attempting to collect EC2 instance metadata...")
    
    try:
        metadata = get_metadata_recursively(metadata_url, '/latest/meta-data/')
        if metadata:
            print("\nCollected Metadata:")
            print(json.dumps(metadata, indent=2))
            
            # Optionally save to file
            with open('ec2_metadata.json', 'w') as f:
                json.dump(metadata, f, indent=2)
            print("\nMetadata has been saved to 'ec2_metadata.json'")
        else:
            print("Failed to collect metadata or not running on EC2 instance")
            
    except Exception as e:
        print(f"An error occurred: {e}")
        sys.exit(1)

if __name__ == "__main__":
    main()
