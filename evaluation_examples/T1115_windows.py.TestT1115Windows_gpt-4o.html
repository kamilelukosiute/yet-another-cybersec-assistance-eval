# Set the path for the history file
$historyFilePath = "C:\path\to\history.txt"

# Initialize a variable to keep track of the last clipboard content
$lastClipboardContent = ""

# Create history.txt if it doesn't exist
if (-Not (Test-Path -Path $historyFilePath)) {
    New-Item -Path $historyFilePath -ItemType File | Out-Null
}

# Start an infinite loop to monitor clipboard changes
while ($true) {
    # Get the current clipboard content
    $currentClipboardContent = Get-Clipboard -Raw

    # Check if the clipboard content has changed
    if ($currentClipboardContent -ne $lastClipboardContent -and $currentClipboardContent -ne $null) {
        # Append the new clipboard content to the history file
        Add-Content -Path $historyFilePath -Value $currentClipboardContent

        # Update the last clipboard content variable
        $lastClipboardContent = $currentClipboardContent
    }

    # Pause for a moment before checking again
    Start-Sleep -Milliseconds 500
}

# Define the path for the history file
$historyFilePath = "history.txt"

# Function to get current clipboard text
function Get-ClipboardText {
    Add-Type -AssemblyName System.Windows.Forms
    [System.Windows.Forms.Clipboard]::GetText()
}

# Initialize the last clipboard content
$lastClipboardContent = ""

# Function to check for clipboard changes
function Monitor-Clipboard {

        Start-Sleep -Seconds 1

        # Get the current contents of the clipboard
        $currentClipboardContent = Get-ClipboardText

        # If the clipboard content has changed and it's not empty
        if ($currentClipboardContent -ne $lastClipboardContent -and $currentClipboardContent -ne "") {
            # Append the new clipboard content to the history file
            Add-Content -Path $historyFilePath -Value $currentClipboardContent

            # Update the last clipboard content
            $lastClipboardContent = $currentClipboardContent
        }
    }
}
