# File Name: ClipboardMonitor.ps1

# Define path for the history file
$historyFile = "history.txt"

# Initialize the last clipboard content
$lastClipboardContent = ""

# Function to get the current clipboard content
function Get-ClipboardContent {
    Add-Type -AssemblyName PresentationCore
    [Windows.Clipboard]::GetText()
}

# Function to monitor clipboard changes
function Monitor-Clipboard {
    while ($true) {
        # Get current clipboard content
        $currentClipboardContent = Get-ClipboardContent

        # If the content has changed, append to file and update the last known content
        if ($currentClipboardContent -ne $lastClipboardContent) {
            $currentClipboardContent | Out-File -FilePath $historyFile -Append
            $lastClipboardContent = $currentClipboardContent
        }

        # Wait for a short period before checking again
        Start-Sleep -Seconds 1
    }
}

# Filename: Monitor-Clipboard.ps1

# Define the path for the history file
$historyFilePath = "$PSScriptRoot\history.txt"

# Initialize last clipboard content as an empty string
$lastClipboardContent = ""

# Function to get clipboard content
function Get-ClipboardContent {
    Add-Type -AssemblyName System.Windows.Forms
    [System.Windows.Forms.Clipboard]::GetText()
}

# Run an infinite loop to continuously check clipboard content
while ($true) {
    try {
        # Get current clipboard content
        $currentClipboardContent = Get-ClipboardContent

        # Check if clipboard content has changed
        if ($currentClipboardContent -ne $lastClipboardContent -and $currentClipboardContent -ne "") {
            # Update last clipboard content
            $lastClipboardContent = $currentClipboardContent

            # Append the new clipboard content to the history file
            Add-Content -Path $historyFilePath -Value $currentClipboardContent

            # Print the captured content for verification (logging)
            Write-Output "Captured: $currentClipboardContent"
        }

        # Wait for a short time before checking again to reduce CPU usage
        Start-Sleep -Milliseconds 500
    } catch {
        Write-Error "An error occurred: $_"
    }
}

# PowerShell script to monitor clipboard and save history to history.txt
$historyFile = "history.txt"
$previousClipboardText = ""

# Create the history file if it doesn't exist
if (-Not (Test-Path $historyFile)) {
    New-Item -Path $historyFile -ItemType File -Force | Out-Null
}

while ($true) {
    try {
        # Get the current clipboard text
        $currentClipboardText = Get-Clipboard -ErrorAction Stop

        # Check if the clipboard text has changed and is not empty
        if ($currentClipboardText -ne $previousClipboardText -and -not [string]::IsNullOrWhiteSpace($currentClipboardText)) {
            # Log the new clipboard text to the history file
            Add-Content -Path $historyFile -Value $currentClipboardText

            # Update the previous clipboard text
            $previousClipboardText = $currentClipboardText
        }
    } catch {
        # Handle any errors (especially when clipboard is not ready or empty)
        Start-Sleep -Milliseconds 500
    }

    # Delay before checking again
    Start-Sleep -Milliseconds 500
}
