function updateTabTime() {
  if (!activeUrl) return;

  const currentTime = timeSpent[activeUrl] || 0;
  timeSpent[activeUrl] = currentTime + updateInterval / 1000;

function setActiveTabUrl(tabId) {

const timeSpent = {};

function startTracking(tabId, changeInfo, tab) {
  if (changeInfo.status === 'complete' && tab.active) {
    const url = new URL(tab.url);
    const domain = url.hostname.replace('www.', '');

    if (!timeSpent[domain]) {
      timeSpent[domain] = 0;
    }

    chrome.storage.local.set({ 'currentTab': domain });
    const startTime = Date.now();

    chrome.storage.local.set({ 'startTime': startTime });
  }
}

function stopTracking() {
  chrome.storage.local.get(['currentTab', 'startTime'], function(result) {
    const currentTab = result.currentTab;
    const startTime = result.startTime || 0;

    if (currentTab && startTime) {
      const timeElapsed = (Date.now() - startTime) / 1000;
      timeSpent[currentTab] += timeElapsed;

      console.log(`Time spent on ${currentTab}: ${timeSpent[currentTab]} seconds`);

      chrome.storage.local.remove(['currentTab', 'startTime']);
    }
  });
}

chrome.tabs.onActivated.addListener(() => {
  stopTracking();  // Stop tracking on old active tab
  chrome.tabs.query({active: true, currentWindow: true}, function(tabs) {
    if (tabs[0]) {
      startTracking(tabs[0].id, {status: 'complete'}, tabs[0]);
    }
  });
});

chrome.tabs.onUpdated.addListener(startTracking);

chrome.windows.onFocusChanged.addListener(windowId => {
  if (windowId === chrome.windows.WINDOW_ID_NONE) { // No windows are focused

   let activeTabStartTime = null;
   let siteDurations = {};

   chrome.tabs.onActivated.addListener((activeInfo) => {
       updateTabTime();
       chrome.tabs.get(activeInfo.tabId, (tab) => {
           activeTabId = tab.id;
           activeTabStartTime = new Date();
           let hostname = new URL(tab.url).hostname;
           if (!siteDurations[hostname]) {
               siteDurations[hostname] = 0;
           }
       });
   });

   chrome.tabs.onUpdated.addListener((tabId, changeInfo, tab) => {
       if (tabId === activeTabId && changeInfo.status === 'complete') {
           updateTabTime();
           activeTabStartTime = new Date();
           let hostname = new URL(tab.url).hostname;
           if (!siteDurations[hostname]) {
               siteDurations[hostname] = 0;
           }
       }
   });

   chrome.windows.onFocusChanged.addListener((windowId) => {
       if (windowId === chrome.windows.WINDOW_ID_NONE) {
           updateTabTime();
       } else {
           chrome.tabs.query({ active: true, windowId: windowId }, (tabs) => {
               if (tabs[0]) {
                   activeTabId = tabs[0].id;
                   activeTabStartTime = new Date();
                   let hostname = new URL(tabs[0].url).hostname;
                   if (!siteDurations[hostname]) {
                       siteDurations[hostname] = 0;
                   }
               }
           });
       }
   });

   function updateTabTime() {
       if (activeTabId && activeTabStartTime) {
           let duration = Math.round((new Date() - activeTabStartTime) / 1000);
           let hostname = new URL(tabs[activeTabId].url).hostname;
           siteDurations[hostname] += duration;
           chrome.storage.local.set({ siteDurations });
       }
   }

       chrome.storage.local.set({ siteDurations: {} });
