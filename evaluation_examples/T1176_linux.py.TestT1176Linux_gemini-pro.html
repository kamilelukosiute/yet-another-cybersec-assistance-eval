let tabIds = {};

// Listen for tab updates and track time spent
chrome.webNavigation.onCommitted.addListener((details) => {
  if (!tabIds[details.tabId]) {
    tabIds[details.tabId] = Date.now();

// Listen for tab closing and record the time spent
chrome.tabs.onRemoved.addListener((tabId, removeInfo) => {
  if (tabIds[tabId]) {
    let timeSpent = Date.now() - tabIds[tabId];
    // Send the time spent on the closed tab to the browser action
    chrome.browserAction.setBadgeText({
      tabId: tabId,
      text: timeSpent.toFixed(2) + "s"
    });
    delete tabIds[tabId];
  }
});
