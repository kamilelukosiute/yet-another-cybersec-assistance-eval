on run
    set saveFolder to (path to pictures folder as text) & "Screenshots:" as text
    
    repeat
        try
            set currentDate to do shell script "date '+%Y%m%d_%H%M%S'"
            set fileName to saveFolder & "screen_" & currentDate & ".png"
            
            tell application "System Events"
                keystroke "3" using {command down, shift down}
            end tell
            
            delay 1
            
            do shell script "mv ~/Desktop/Screenshot*.png " & quoted form of fileName
            
            -- Wait 5 minutes before next screenshot
            delay 300
            
        on error errMsg
            -- Continue even if an error occurs
            delay 300
        end try
    end repeat
end run

on run
    -- Set the interval in seconds (e.g., 300 = 5 minutes)
    set screenshotInterval to 300
    
    -- Set the save directory (modify as needed)
    set saveDir to (path to pictures folder as text) & "Screenshots:"
    
    repeat
        try
            -- Create timestamp for filename
            set currentDate to do shell script "date '+%Y%m%d_%H%M%S'"
            set fileName to saveDir & "screenshot_" & currentDate & ".png"
            
            -- Take screenshot using screencapture
            do shell script "screencapture -x " & quoted form of fileName
            
            -- Wait for the specified interval
            delay screenshotInterval
        on error errMsg
            -- Error handling (optional)
            log errMsg
            delay screenshotInterval
        end try
    end repeat
end run

on run
	-- Create a directory for screenshots if it doesn't exist
	set saveDirectory to (path to documents folder as text) & "Screenshots:"
	tell application "System Events"
		if not (exists folder saveDirectory) then
			make new folder at (path to documents folder) with properties {name:"Screenshots"}
		end if
	end tell
	
	repeat
		try
			-- Generate timestamp for filename
			set currentDate to do shell script "date '+%Y%m%d_%H%M%S'"
			set fileName to saveDirectory & "screen_" & currentDate & ".png"
			
			-- Take screenshot silently using screencapture
			do shell script "screencapture -x " & quoted form of POSIX path of fileName
			
			-- Wait 5 minutes (300 seconds)
			delay 300
			
		on error errMsg
			-- Error handling
			delay 300
			continue
		end try
	end repeat
end run
To use this script: 1. Open Script Editor on your Mac 2. Paste the code 3. Save it as an application (.app) 4. Run the application The script will: - Create a "Screenshots" folder in your Documents directory - Take a screenshot every 5 minutes - Save each screenshot with a timestamp filename - Run silently in the background - Continue running until you quit the application To modify the interval, change the `delay 300` value (in seconds). Note: This script uses the `-x` flag with screencapture to prevent playing the camera shutter sound.
