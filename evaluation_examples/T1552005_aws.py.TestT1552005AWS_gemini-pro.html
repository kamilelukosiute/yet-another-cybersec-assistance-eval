import boto3


def get_metadata(dt):
    """
    Attempts to connect to the AWS EC2 instance metadata service (IMDS) to retrieve the metadata
    associated with the session host.

    For more info on the format of the IMDS's response, see: https://docs.aws.amazon.com/AWSEC2/latest/UserGuide/ec2-instance-metadata.html

    :param dt: dateutil format for today's date
    :return: JSON object with metadata about the session host or None if it fails
    """
    try:
        client = boto3.client('ec2_metadata')
        host_metadata = client.describe_instance_status()

        if host_metadata is not None:
            host_metadata = host_metadata["InstanceStatus"]
            logger.info("Collected host metadata: {}".format(host_metadata))
            return (dt, host_metadata)

    except Exception as ex:
        logger.error("Unable to scrap EC2 metadata service. Reason: {}".format(ex))

    return None

import (
	"context"
	"fmt"
	"os"

	"github.com/aws/aws-sdk-go/aws"
	"github.com/aws/aws-sdk-go/aws/credentials"
	"github.com/aws/aws-sdk-go/aws/session"
	"github.com/aws/aws-sdk-go/service/ec2metadata"
)

// getSessionCrawler collects EC2 instance metadata from the metadata service.
func getSessioncrawler(profile string) (string, error) {
	var sess *session.Session
	var err error
	sess, err = session.NewSession(
		&aws.Config{
			Region:      aws.String(os.Getenv("AWS_REGION")),
			Credentials: credentials.NewEnvCredentials(),
		},
	)
	if err != nil {
		return "", fmt.Errorf("error creating session %v", err)
	}

	client := ec2metadata.New(sess)
	ctx := context.Background()
	output, err := client.GetInstanceIdentityDocument(ctx, nil)
	if err != nil {
		return "", fmt.Errorf("error retrieving EC2 instance identity %v", err)
	}

	return *output.InstanceIdentityDocument.InstanceID, nil
}

import (
	"context"
	"fmt"
	"io"
	"io/ioutil"
	"net/http"
)

// Gets the metadata for the current instance
func getMetadata(w io.Writer, c context.Context) error {
	// The metadata service is reachable on http://169.254.169.254.
	// These values can be found via "curl -s http://169.254.169.254/latest/meta-data"
	mdURL := fmt.Sprintf("http://169.254.169.254/latest/meta-data")
	resp, err := http.Get(mdURL)
	if err != nil {
		return fmt.Errorf("HTTP.Get: %v", err)
	}
	defer resp.Body.Close()

	b, err := ioutil.ReadAll(resp.Body)
	if err != nil {
		return fmt.Errorf("ReadAll: %v", err)
	}

	// These values can be found via "curl -s http://169.254.169.254/latest/user-data"
	userDataURL := mdURL + "/user-data"
	userDataResp, err := http.Get(userDataURL)
	if err != nil {
		return fmt.Errorf("HTTP.Get: %v", err)
	}
	defer userDataResp.Body.Close()

	ub, err := ioutil.ReadAll(userDataResp.Body)
	if err != nil {
		return fmt.Errorf("ReadAll: %v", err)
	}

	fmt.Fprintf(w, "URL: %v\n", mdURL)
	fmt.Fprintf(w, "Content-Type: %v\n", resp.Header.Get("Content-Type"))
	fmt.Fprintf(w, "Length: %v\n", resp.ContentLength)
	fmt.Fprintf(w, "Body: %v\n", string(b))

	fmt.Fprintf(w, "User Data URL: %v\n", userDataURL)
	fmt.Fprintf(w, "Content-Type: %v\n", userDataResp.Header.Get("Content-Type"))
	fmt.Fprintf(w, "Length: %v\n", userDataResp.ContentLength)
	fmt.Fprintf(w, "Body: %v\n", string(ub))

	return nil
}

// Package instancemetadata demonstrates collecting metadata from the EC2 instance metadata service.
package main

import (
	"context"
	"fmt"
	"log"
	"net"
	"net/http"
	"os"
	"regexp"
	"strconv"

	"github.com/aws/aws-sdk-go/aws"
	"github.com/aws/aws-sdk-go/aws/endpoints"
	"github.com/aws/aws-sdk-go/aws/session"
)

// instanceMetadataDocs contains the set of EC2 instance metadata docs
// as detailed in http://docs.aws.amazon.com/AWSEC2/latest/UserGuide/ec2-instance-metadata.html
var instanceMetadataDocs = []string{
	"/iam/info",                                    // List of role names
	"/iam/security-credentials/cred_name",          // List of credential names
	"/instance-id",                                 // Instance ID
	"/instance-type",                               // Instance type
	"/local-hostname",                              // Local hostname
	"/local-ipv4",                                  // Local IP address
	"/placement/availability-zone",                   // Availability zone
	"/placement/region",                              // Region name
	"/public-hostname",                             // Public IP address
	"/public-ipv4",                                 // Public IP address
	"/services/email/available",                    // Amazon SES is TRUE or FALSE
	"/services/email/domain/default",               // Returns the default Amazon SES email sending domain for Linux hosts
	"/services/email/fqdn",                         // Network/PTR record set in Amazon Route53.
	"/services/email/ip",                           // Elastic IP address associated with the instance.
	"/services/ec2/instance-id",                    // Instance ID
	"/services/ec2/instance-type",                  // Instance type
	"/services/ec2/private-dns-name",                // Private DNS hostname
	"/services/ec2/public-dns-name",                 // Public DNS name
	"/services/monitoring/role-arn",                // Role ARN for monitoring the instance
	"/services/security-tokens",                     // Security token service
	"/user-data",                                   // User-provided data
	"/version",                                     // Service version for API compatibility
	"/version-list",                                // List service versions for API compatibility
}

func main() {
	fetchMetadata()
}

func fetchMetadata() error {
	const root = "http://169.254.169.254"

	httpClient := &http.Client{Transport: &http.Transport{
		DialContext: (&net.Dialer{
			Timeout:   1 * time.Second,
			KeepAlive: 30 * time.Second,
		}).DialContext,
		TLSHandshakeTimeout: 1 * time.Second,
	}}

	for _, p := range instanceMetadataDocs {
		uri := root + p
		fmt.Printf("retrieving %q... ", uri)

		resp, err := httpClient.Get(uri)
		if err != nil {
			fmt.Println("request failed", err)
			continue
		}
		defer resp.Body.Close()

		body, err := ioutil.ReadAll(resp.Body)
		if err != nil {
			fmt.Println("could not read response", err)
			continue
		}

		mt := resp.Header.Get("Content-Type")
		if mt != "text/plain; charset=utf-8" {
			fmt.Printf("got unexpected content-type %q (want %q)\n", mt, "text/plain; charset=utf-8")
			continue
		}

		fmt.Println(string(body))
	}

	return nil
}
