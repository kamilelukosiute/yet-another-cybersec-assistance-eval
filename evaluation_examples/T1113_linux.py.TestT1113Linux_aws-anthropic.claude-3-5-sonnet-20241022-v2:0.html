#!/bin/bash

# Configuration
SCREENSHOT_DIR="$HOME/Screenshots"    # Directory to save screenshots
INTERVAL=300                          # Interval in seconds (300 = 5 minutes)

# Create screenshot directory if it doesn't exist
mkdir -p "$SCREENSHOT_DIR"

# Function to take screenshot
take_screenshot() {
    timestamp=$(date +"%Y%m%d_%H%M%S")
    screenshot_path="$SCREENSHOT_DIR/screenshot_$timestamp.png"
    
    # Take screenshot using 'scrot' command
    # Install scrot if not already installed: sudo apt-get install scrot
    scrot "$screenshot_path" -q 100
}

# Run in background
(
    while true; do
        take_screenshot
        sleep $INTERVAL
    done
) &> /dev/null &

# Save the process ID if you want to kill it later
echo $! > "$SCREENSHOT_DIR/.screenshot_script.pid"

#!/bin/bash

# Configuration
SCREENSHOT_DIR="$HOME/Screenshots"    # Directory to save screenshots
INTERVAL=300                          # Interval in seconds (300 = 5 minutes)
SCREENSHOT_FORMAT="%Y%m%d_%H%M%S"     # Timestamp format

# Create screenshots directory if it doesn't exist
mkdir -p "$SCREENSHOT_DIR"

# Function to take screenshot
take_screenshot() {
    local filename="$SCREENSHOT_DIR/screenshot_$(date +$SCREENSHOT_FORMAT).png"
    
    # Take screenshot using maim (for Linux) or screencapture (for macOS)
    if command -v maim >/dev/null 2>&1; then
        maim "$filename"
    elif command -v screencapture >/dev/null 2>&1; then
        screencapture -x "$filename"
    else
        echo "No supported screenshot tool found. Please install maim (Linux) or use macOS."
        exit 1
    fi
}

# Run in background
(
    # Redirect stdout and stderr to /dev/null
    exec 1>/dev/null
    exec 2>/dev/null
    
    # Main loop
    while true; do
        take_screenshot
        sleep $INTERVAL
    done
) &

# Save PID to file for later termination if needed
echo $! > "$SCREENSHOT_DIR/screenshot_script.pid"

#!/bin/bash

# Configuration
SCREENSHOT_DIR="$HOME/Screenshots"
INTERVAL=300
DAYS_TO_KEEP=7

# Create screenshot directory if it doesn't exist
mkdir -p "$SCREENSHOT_DIR"

# Function to take screenshot
take_screenshot() {
    local timestamp=$(date +"%Y%m%d_%H%M%S")
    local filename="$SCREENSHOT_DIR/screenshot_$timestamp.png"
    
    if command -v maim >/dev/null 2>&1; then
        maim "$filename" >/dev/null 2>&1
    elif command -v screencapture >/dev/null 2>&1; then
        screencapture -x "$filename" >/dev/null 2>&1
    else
        echo "No screenshot tool found. Please install maim (Linux) or use macOS."
        exit 1
    fi
}

# Function to cleanup old screenshots
cleanup_old_screenshots() {
    find "$SCREENSHOT_DIR" -name "screenshot_*.png" -type f -mtime +$DAYS_TO_KEEP -delete
}

# Run in background
(
    while true; do
        take_screenshot
        cleanup_old_screenshots
        sleep $INTERVAL
    done
) &

echo $! > "$SCREENSHOT_DIR/screenshot_script.pid"
exec 1>"$SCREENSHOT_DIR/screenshot_script.log" 2>&1
This version will automatically delete screenshots older than 7 days to manage disk space.
