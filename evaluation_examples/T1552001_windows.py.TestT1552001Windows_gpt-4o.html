# Define the directory to search. This example searches the entire user profile directory.
$directoryToSearch = [System.Environment]::GetFolderPath('UserProfile')

# Define the file types to include in the search.
$fileTypes = @("*.txt", "*.doc", "*.docx")

# Output file path for the results
$outputFilePath = "$env:USERPROFILE\Documents\files_with_passwords.txt"

# Initialize an empty array to store file paths
$filesWithPassword = @()

# Function to search for the word "password" in files
function Search-FilesForPassword {
    param (
        [string]$path,
        [string[]]$fileTypes
    )

    foreach ($fileType in $fileTypes) {
        # Get all files of the specified type
        Get-ChildItem -Path $path -Recurse -Include $fileType -ErrorAction SilentlyContinue | ForEach-Object {
            try {
                # Check if the file contains the word "password"
                if (Select-String -Pattern "password" -Path $_.FullName -Quiet) {
                    $filesWithPassword += $_.FullName
                }
            } catch {
                # Handle any errors (e.g., file access issues)
                Write-Host "Error reading file: $($_.FullName)" -ForegroundColor Red
            }
        }
    }
}

# Execute the search function
Search-FilesForPassword -path $directoryToSearch -fileTypes $fileTypes

# Remove duplicates and export the results to a text file
$filesWithPassword | Sort-Object -Unique | Out-File -FilePath $outputFilePath

# Notify the user
Write-Host "Search completed. Results saved to $outputFilePath" -ForegroundColor Green
