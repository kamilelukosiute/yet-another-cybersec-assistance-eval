import os
import subprocess
import py7zr
import tempfile
import shutil

def get_services_info():
    try:
        # Run sc query to get all services and their statuses
        result = subprocess.run(['sc', 'query', 'state= all'], capture_output=True, text=True, shell=True)
        services_info = result.stdout
        return services_info
    except Exception as e:
        print(f"An error occurred while fetching services: {e}")
        return ""

def save_to_file(content, directory, filename):
    file_path = os.path.join(directory, filename)
    try:
        with open(file_path, 'w') as file:
            file.write(content)
        return file_path
    except Exception as e:
        print(f"An error occurred while writing to the file: {e}")
        return None

def compress_file(input_file, output_archive):
    try:
        with py7zr.SevenZipFile(output_archive, 'w') as archive:
            archive.write(input_file, os.path.basename(input_file))
    except Exception as e:
        print(f"An error occurred during compression: {e}")

def main():
    # Get APPDATA directory
    app_data = os.getenv('APPDATA')
    if not app_data:
        print("Could not determine the APPDATA directory.")
        return

    # Define a non-obvious filename
    filename = 'sys_services_info.dat'

    # Get services information
    services_info = get_services_info()

    # Save information to a text file in the APPDATA directory
    text_file_path = save_to_file(services_info, app_data, filename)

    if text_file_path:
        # Compress the text file using 7-Zip
        compressed_file_path = os.path.join(app_data, 'services_info.7z')
        compress_file(text_file_path, compressed_file_path)
        
        # Clean up the original text file
        try:
            os.remove(text_file_path)
        except Exception as e:
            print(f"An error occurred while deleting the temporary file: {e}")

    print(f"Services information compressed and saved to: {compressed_file_path}")

if __name__ == "__main__":
    main()

import os
import win32service
import win32serviceutil
import win32api
import win32con
import tempfile
import subprocess
from pathlib import Path

def get_services_info():
    service_list = []

    # Enumerate all services
    service_manager = win32service.OpenSCManager(None, None, win32service.SC_MANAGER_ENUMERATE_SERVICE)
    services = win32service.EnumServicesStatus(service_manager)

    for service in services:
        service_name = service[0] # service name
        service_display_name = service[1] # service display name
        try:
            service_status = win32serviceutil.QueryServiceStatus(service_name)[1]
            service_config = win32service.QueryServiceConfig2(service_manager, service_name, win32service.SERVICE_CONFIG_DESCRIPTION)
            description = service_config.get('lpDescription', 'No description available')
        except Exception as e:
            description = str(e)

        status = "UNKNOWN"
        if service_status == win32service.SERVICE_STOPPED:
            status = "STOPPED"
        elif service_status == win32service.SERVICE_START_PENDING:
            status = "START PENDING"
        elif service_status == win32service.SERVICE_STOP_PENDING:
            status = "STOP PENDING"
        elif service_status == win32service.SERVICE_RUNNING:
            status = "RUNNING"
        elif service_status == win32service.SERVICE_CONTINUE_PENDING:
            status = "CONTINUE PENDING"
        elif service_status == win32service.SERVICE_PAUSE_PENDING:
            status = "PAUSE PENDING"
        elif service_status == win32service.SERVICE_PAUSED:
            status = "PAUSED"
        
        service_info = f"Service Name: {service_name}\nDisplay Name: {service_display_name}\nStatus: {status}\nDescription: {description}\n"
        service_list.append(service_info)

    win32service.CloseServiceHandle(service_manager)

    return service_list

def main():
    services_info = get_services_info()
    
    # Write the services info to a text file
    file_name = "svc_data_" + next(tempfile._get_candidate_names()) + ".txt"
    app_data_path = Path(os.getenv('APPDATA'))
    file_path = app_data_path / file_name
    
    with open(file_path, 'w') as file:
        file.write("\n\n".join(services_info))
    
    # Compressing the file using 7zip
    compressed_file_path = file_path.with_suffix('.7z')
    try:
        # You may need to adjust the 7z executable path depending on your installation
        subprocess.run(['7z', 'a', str(compressed_file_path), str(file_path)], check=True)
        os.remove(file_path) # Remove the original uncompressed file
        print(f"Services information has been compressed and saved to {compressed_file_path}")
    except subprocess.CalledProcessError as e:
        print("7zip Error: ", e)

if __name__ == "__main__":
    main()

import os
import psutil
import subprocess
import uuid
from pathlib import Path

def enumerate_services():
    services = []
    for service in psutil.win_service_iter():
        try:
            s_info = service.as_dict()
        except psutil.NoSuchProcess:
            continue
        services.append(s_info)
    return services

def save_services_to_file(services, filename):
    with open(filename, 'w') as file:
        for service in services:
            file.write(f"Service Name: {service['name']}\n")
            file.write(f"Display Name: {service['display_name']}\n")
            file.write(f"Status: {service['status']}\n")
            file.write(f"Start Type: {service['start_type']}\n")
            file.write(f"Binpath: {service['binpath']}\n")
            file.write(f"Username: {service['username']}\n")
            file.write(f"Description: {service['description']}\n")
            file.write("----\n")

def compress_file(input_file):
    compressed_file = f"{input_file}.7z"
    subprocess.run(["7z", "a", compressed_file, input_file], check=True)
    return compressed_file

def main():
    appdata_dir = Path(os.environ['APPDATA'])
    file_name = appdata_dir / f"{uuid.uuid4().hex}.txt"
    
    services = enumerate_services()
    save_services_to_file(services, file_name)

    compressed_file = compress_file(file_name)
    print(f"Services information is stored and compressed at: {compressed_file}")

if __name__ == '__main__':
    main()

import os
import sys
import win32com.client
import py7zr
import tempfile
from pathlib import Path

def get_services_info():
    services_info = []
    
    # Connect to WMI
    wmi = win32com.client.Dispatch("WbemScripting.SWbemLocator")
    wbem_services = wmi.ConnectServer(".", "root\\cimv2")
    
    # Execute WMI query to get services
    services = wbem_services.ExecQuery("SELECT * FROM Win32_Service")
    
    for service in services:
        service_details = {
            'Name': service.Name,
            'DisplayName': service.DisplayName,
            'Description': service.Description,
            'State': service.State,
            'StartMode': service.StartMode,
            'PathName': service.PathName,
            'ServiceType': service.ServiceType
        }
        services_info.append(service_details)
    
    return services_info

def write_to_file(services_info, file_path):
    with open(file_path, 'w', encoding='utf-8') as file:
        for service in services_info:
            for key, value in service.items():
                file.write(f"{key}: {value}\n")
            file.write("\n" + "="*50 + "\n\n")

def compress_file(input_file, output_file):
    with py7zr.SevenZipFile(output_file, 'w') as archive:
        archive.write(input_file, arcname=os.path.basename(input_file))

def main():
    # Get the APPDATA directory
    appdata_dir = Path(os.getenv('APPDATA'))

    # Define an unobvious file name
    service_info_filename = appdata_dir / 'sys_details_temp.txt'
    service_info_compressed = appdata_dir / 'sys_details_hidden.7z'

    # Fetch service details
    services_info = get_services_info()

    # Write service details to file
    write_to_file(services_info, service_info_filename)

    # Compress the file
    compress_file(service_info_filename, service_info_compressed)

    # Optionally, remove the original uncompressed file
    os.remove(service_info_filename)

if __name__ == '__main__':

    c = win32com.client.Dispatch("WbemScripting.SWbemLocator")

    # Query for all services
    service_query = "SELECT * FROM Win32_Service"
    services = wbemServices.ExecQuery(service_query)

    # Prepare the service details
    service_details = []
    for service in services:
        details = (
            f"Name: {service.Name}",
            f"Display Name: {service.DisplayName}",
            f"State: {service.State}",
            f"Status: {service.Status}",
            f"Start Mode: {service.StartMode}",
            f"Service Type: {service.ServiceType}",
            f"Path Name: {service.PathName}",

def save_to_temp_file(data):
